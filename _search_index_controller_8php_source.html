38  /**
39  * @see BatchController::getWorkPackage()
40  */
41  protected function getWorkPackage($number) {
42  if ($number == 0) {
43  $search = ObjectFactory::getInstance('search');
44  if ($search instanceof IndexedSearch) {
45  // get all types to index
46  $types = array();
47  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
48  foreach ($persistenceFacade->getKnownTypes() as $type) {
49  $tpl = $persistenceFacade->create($type);
50  if ($search->isSearchable($tpl)) {
51  $types[] = $type;
52  }
53  }
54  $search->resetIndex();
55  return array('name' => Message::get('Collect objects'), 'size' => 1, 'oids' => $types, 'callback' => 'collect');
56  }
57  else {
58  // no index to be updated
59  return null;
60  }
61  }
62  else {
63  return null;
64  }
65  }
66 
67  /**
68  * Collect all oids of the given types
69  * @param $types The types to process
70  * @note This is a callback method called on a matching work package @see BatchController::addWorkPackage()
71  */
72  protected function collect($types) {
73  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
74  foreach ($types as $type) {
75  $oids = $persistenceFacade->getOIDs($type);
76  if (sizeof($oids) == 0) {
77  $oids = array(1);
78  }
79  $this->addWorkPackage(Message::get('Indexing %0%', array($type)), 10, $oids, 'index');
80  }
81  }
82 
83  /**
84  * Create the lucene index from the given objects
85  * @param $oids The oids to process
86  * @note This is a callback method called on a matching work package @see BatchController::addWorkPackage()
87  */
88  protected function index($oids) {
89  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
90  $search = ObjectFactory::getInstance('search');
91  foreach($oids as $oid) {
92  if (ObjectId::isValid($oid)) {
93  $obj = $persistenceFacade->load($oid);
94  $search->addToIndex($obj);
95  }
96  }
97  $search->commitIndex(false);
98 
99  if ($this->getStepNumber() == $this->getNumberOfSteps()) {
100  $this->addWorkPackage(Message::get('Optimizing index'), 1, array(0), 'optimize');
101  }
102  }
103 
104  /**
105  * Optimize the search index
106  * @param $oids The oids to process
107  * @note This is a callback method called on a matching work package @see BatchController::addWorkPackage()
108  */
109  protected function optimize($oids) {
110  $search = ObjectFactory::getInstance('search');
111  $search->optimizeIndex();
112  }
