26 /**
27  * DefaultLocalization is a Localization implementation that saves translations
28  * in the store. Entity instances are localized value by value, where a
29  * translation of a value of one instance into a specific language
30  * is represented by one instance of the translation entity type (e.g. Translation).
31  *
32  * The translation entity type must have the attributes 'objectid', 'attribute',
33  * 'translation', 'language' with the appropriate getter and setter methods. It
34  * is defined calling the Localization::setTranslationType() method.
35  *
36  * The default language is defined in the configuration key 'defaultLanguage'
37  * in section 'localization'.
38  *
39  * All languages available for translation are either defined in the configuration
40  * section 'languages', where each language has it's own entry: e.g. en = English
41  * or in a language entity type (e.g. Language). The language entity type must
42  * have the attributes 'code' and 'name' with the appropriate getter and setter
43  * methods. It is defined calling the Localization::setLanguageType() method.
44  * If translation entity type and configuration section are defined, the
45  * configuration section is preferred.
46  *
47  * To exclude values of a special type (like date values) from the translation,
48  * they may be omitted in the array that is defined calling the
49  * Localization::setInputTypes() method. This array lists all input_types whose
50  * translations are stored.
51  *
52  * @author ingo herwig <ingo@wemove.com>
53  */
54 class DefaultLocalization implements Localization {

111  /**
112  * @see Localization::getSupportedLanguages()
113  * Reads the configuration section 'languages'
114  */
115  public function getSupportedLanguages() {
116  if ($this->_supportedLanguages == null) {
117  // check if the configuration section exists
118  $config = ObjectFactory::getConfigurationInstance();
119  if (($languages = $config->getSection('languages')) !== false) {
120  $this->_supportedLanguages = $languages;
121  }
122  // if not, use the languageType
123  else {
124  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
125  $languages = $persistenceFacade->loadObjects($this->_languageType, BuildDepth::SINGLE);
126  for($i=0; $i<sizeof($languages); $i++) {
127  $curLanguage = $languages[$i];
128  $this->_supportedLanguages[$curLanguage->getCode()] = $curLanguage->getName();
129  }
130  }
131  }
132  return $this->_supportedLanguages;
133  }

145  /**
146  * @see Localization::loadTranslation()
147  */
148  public function loadTranslation($object, $lang, $useDefaults=true, $recursive=true) {
149  $translatedObject = $this->loadTranslationImpl($object, $lang, $useDefaults);
150 
151  // recurse if requested
152  if ($recursive) {
153  $mapper = $object->getMapper();
154  if ($mapper) {
155  $relations = $mapper->getRelations('child');
156  foreach ($relations as $relation) {
157  if ($relation->getOtherNavigability()) {
158  $role = $relation->getOtherRole();
159  $childValue = $object->getValue($role);
160  if ($childValue != null) {
161  $children = $relation->isMultiValued() ? $childValue : array($childValue);
162  foreach ($children as $child) {
163  // don't resolve proxies
164  if (!($child instanceof PersistentObjectProxy)) {
165  $translatedChild = $this->loadTranslationImpl($child, $lang, $useDefaults);
166  $translatedObject->addNode($translatedChild, $role);
167  }
168  }
169  }
170  }
171  }
172  }
173  }
174  return $translatedObject;
175  }

177  /**
178  * Load a translation of a single entity for a specific language.
179  * @param $object A reference to the object to load the translation into. The object
180  * is supposed to have it's values in the default language.
181  * @param $lang The language of the translation to load.
182  * @param $useDefaults Boolean whether to use the default language values
183  * for untranslated/empty values or not. Optional, default is true.
184  * @return A reference to the translated object.
185  * @throws IllegalArgumentException
186  */
187  protected function loadTranslationImpl($object, $lang, $useDefaults=true) {
188  if ($object == null) {
189  throw new IllegalArgumentException('Cannot load translation for null');
190  }
191 
192  $translatedObject = $object;
193  $oidStr = $object->getOID()->__toString();
194 
195  // load the translations and translate the object for any language
196  // different to the default language
197  if ($lang != $this->getDefaultLanguage()) {
198  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
199  $transaction = $persistenceFacade->getTransaction();
200  $translatedObject = $persistenceFacade->create($object->getType());
201  $transaction->detach($translatedObject->getOID());
202  $object->copyValues($translatedObject, false);
203 
204  $query = new ObjectQuery($this->_translationType, __CLASS__.'load_save');
205  $tpl = $query->getObjectTemplate($this->_translationType);
206  $tpl->setValue('objectid', Criteria::asValue('=', $oidStr));
207  $tpl->setValue('language', Criteria::asValue('=', $lang));
208  $translations = $query->execute(BuildDepth::SINGLE);
209 
210  // set the translated values in the object
211  $iter = new NodeValueIterator($object, false);
212  for($iter->rewind(); $iter->valid(); $iter->next()) {
213  $this->setTranslatedValue($translatedObject, $iter->key(), $translations, $useDefaults);
214  }
215  }
216  return $translatedObject;
217  }

219  /**
220  * @see Localization::loadTranslation()
221  * @note Only values whose input_type property is listed in the 'inputTypes'
222  * key in the configuration 'localization' are stored.
223  */
224  public function saveTranslation($object, $lang, $saveEmptyValues=false, $recursive=true) {
225  $this->saveTranslationImpl($object, $lang, $saveEmptyValues);
226 
227  // recurse if requested
228  if ($recursive) {
229  $iterator = new NodeIterator($object);
230  foreach($iterator as $oidStr => $obj) {
231  if ($obj->getOID() != $object->getOID()) {
232  // don't resolve proxies
233  if (!($obj instanceof PersistentObjectProxy)) {
234  $this->saveTranslationImpl($obj, $lang, $saveEmptyValues);
235  }
236  }
237  }
238  }

241  /**
242  * Save a translation of a single entity for a specific language. Only the
243  * values that have a non-empty value are considered as translations and stored.
244  * @param $object An instance of the entity type that holds the translations as values.
245  * @param $lang The language of the translation.
246  * @param $saveEmptyValues Boolean whether to save empty translations or not.
247  * Optional, default is false
248  */
249  protected function saveTranslationImpl($object, $lang, $saveEmptyValues=false) {
250  // if the requested language is the default language, do nothing
251  if ($lang == $this->getDefaultLanguage()) {
252  // nothing to do
253  }
254  // save the translations for any other language
255  else {
256  $object->beforeUpdate();
257 
258  // get the existing translations for the requested language
259  $query = new ObjectQuery($this->_translationType, __CLASS__.'load_save');
260  $tpl = $query->getObjectTemplate($this->_translationType);
261  $tpl->setValue('objectid', Criteria::asValue('=', $object->getOID()->__toString()));
262  $tpl->setValue('language', Criteria::asValue('=', $lang));
263  $translations = $query->execute(BuildDepth::SINGLE);
264 
265  // save the translations, ignore pk values
266  $pkNames = $object->getPkNames();
267  $iter = new NodeValueIterator($object, false);
268  for($iter->rewind(); $iter->valid(); $iter->next()) {
269  $valueName = $iter->key();
270  if (!in_array($valueName, $pkNames)) {
271  $curIterNode = $iter->currentNode();
272  $this->saveTranslatedValue($curIterNode, $valueName, $translations, $lang, $saveEmptyValues);
273  }
274  }
275  }
276  }

282  // if the requested language is the default language, do nothing
283  if ($lang == $this->getDefaultLanguage()) {
284  // nothing to do
285  }
286  // delete the translations for any other language
287  else {
288  // get the existing translations for the requested language or all languages
289  $query = new ObjectQuery($this->_translationType, __CLASS__.'delete_trans'.($lang != null));
290  $tpl = $query->getObjectTemplate($this->_translationType);
291  $tpl->setValue('objectid', Criteria::asValue('=', $oid->__toString()));
292  if ($lang != null) {
293  $tpl->setValue('language', Criteria::asValue('=', $lang));
294  }
295  $translations = $query->execute(BuildDepth::SINGLE);
296 
297  // delete the found tranlations
298  foreach ($translations as $curTranslation) {

327  /**
328  * Set a translated value in the given PersistentObject instance.
329  * @param $object The object to set the value on. The object
330  * is supposed to have it's values in the default language.
331  * @param $valueName The name of the value to translate
332  * @param $translations An array of translation instances for the object.
333  * @param $useDefaults Boolean whether to use the default language if no
334  * translation is found or not.
335  */
336  private function setTranslatedValue(PersistentObject $object, $valueName, array $translations, $useDefaults) {
337  $inputType = $object->getValueProperty($valueName, 'input_type');
338  if (in_array($inputType, $this->_inputTypes)) {
339  // empty the value, if the default language values should not be used
340  if (!$useDefaults) {
341  $object->setValue($valueName, null, true);
342  }
343  // translate the value
344  for ($i=0, $count=sizeof($translations); $i<$count; $i++) {
345  $curValueName = $translations[$i]->getValue('attribute');
346  if ($curValueName == $valueName) {
347  $translation = $translations[$i]->getValue('translation');
348  if (!($useDefaults && strlen($translation) == 0)) {
349  $object->setValue($valueName, $translation, true);
350  }
351  break;
352  }
353  }
354  }
355  }

357  /**
358  * Save translated values for the given object
359  * @param $object The object to save the translations on
360  * @param $valueName The name of the value to translate
361  * @param $existingTranslations An array of already existing translation
362  * instances for the object.
363  * @param $lang The language of the translations.
364  * @param $saveEmptyValues Boolean whether to also save empty translations or not.
365  */
366  private function saveTranslatedValue(PersistentObject $object, $valueName, array $existingTranslations, $lang, $saveEmptyValues) {
367  $inputType = $object->getValueProperty($valueName, 'input_type');
368  if (in_array($inputType, $this->_inputTypes)) {
369  $value = $object->getValue($valueName);
370  if ($saveEmptyValues || strlen($value) > 0) {
371  $translation = null;
372 
373  // check if a translation already exists
374  for ($i=0, $count=sizeof($existingTranslations); $i<$count; $i++) {
375  $curValueName = $existingTranslations[$i]->getValue('attribute');
376  if ($curValueName == $valueName) {
377  $translation = &$existingTranslations[$i];
378  break;
379  }
380  }
381 
382  // if not, create a new translation
383  if ($translation == null) {
384  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
385  $translation = $persistenceFacade->create($this->_translationType);
386  }
387 
388  // set all required properties
389  $translation->setValue('objectid', $object->getOID()->__toString());
390  $translation->setValue('attribute', $valueName);
391  $translation->setValue('translation', $object->getValue($valueName));
392  $translation->setValue('language', $lang);
393  }
394  }
