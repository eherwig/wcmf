144  protected function createToken($login) {
145  $jwt = (new Builder())
146  ->setIssuer($this->getTokenIssuer())
147  ->setIssuedAt(time())
148  ->setExpiration(time()+3600)

174  */
175  protected function getTokenData() {
176  $data = null;
177 
178  $request = ObjectFactory::getInstance('request');
179  $token = $request->hasHeader(self::TOKEN_HEADER) ?
180  trim(str_replace(self::AUTH_TYPE, '', $request->getHeader(self::TOKEN_HEADER))) : null;
181  if ($token !== null) {
182  $jwt = (new Parser())->parse((string)$token);
183 
184  // validate
185  $data = new ValidationData();
186  $data->setIssuer($this->getTokenIssuer());
187  if ($jwt->validate($data) && $jwt->verify($this->getTokenSigner(), $this->key)) {
188  $data = $jwt->getClaims();
189  }
190  }
191  return $data;
192  }
