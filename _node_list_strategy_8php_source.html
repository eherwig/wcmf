17 /**
18  * NodeListStrategy implements a list of entities that is retrieved
19  * from the store, where the keys are the object ids and the
20  * values are the display values.
21  * The following list definition(s) must be used in the input_type configuraton:
22  * @code
23  * node:type // list with all entities of the given type
24  * node:type1,type2,... // list with all entities of the given types
25  *
26  * node:type|type.name LIKE 'A%' ... // list with all entities of the given type that
27  * match the given query (@see StringQuery)
28  * @endcode
29  *
30  * @author ingo herwig <ingo@wemove.com>
31  */
32 class NodeListStrategy implements ListStrategy {
33 
34  /**
35  * @see ListStrategy::getList
36  */
37  public function getList($configuration, $language=null) {
38 
39  $listConfig = $this->parseConfiguration($configuration);
40  $isSingleType = sizeof($listConfig['types']) == 1;
41 
42  $list = array();
43  foreach ($listConfig['types'] as $type) {
44  $query = new StringQuery($type);
45  if ($listConfig['query']) {
46  $query->setConditionString($listConfig['query']);
47  }
48  $objects = $query->execute(BuildDepth::SINGLE);
49  foreach ($objects as $object) {
50  $id = $isSingleType ? $object->getOID()->getFirstId() : $object->getOID()->__toString();
51  $list[$id] = $object->getDisplayValue();
52  }
53  }
54 
55  return $list;
56  }
57 
58  /**
59  * Parse the given list configuration
60  * @param $configuration The configuration
61  * @return Associative array with keys 'types' and 'query'
62  */
63  protected function parseConfiguration($configuration) {
64  $query = null;
65  if (strPos($configuration, '|')) {
66  list($typeDef, $query) = explode('|', $configuration, 2);
67  }
68  else {
69  $typeDef = $configuration;
70  }
71  $types = explode(',', $typeDef);
72 
73  return array(
74  'types' => $types,
75  'query' => $query
76  );
77  }
