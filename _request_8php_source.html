42  /**
43  * Get the default request.
44  *
45  * The method tries to match the current request path against the routes
46  * defined in the configuration section 'routes' and constructs the request based on
47  * these parameters. It then adds all data contained in $_GET, $_POST, $_FILES and
48  * php://input (raw data from the request body).
49  *
50  * Examples for route definitions are:
51  * @code
52  * / = action=cms
53  * /rest/{language}/{className} = action=restAction&collection=1
54  * /rest/{language}/{className}/{id|[0-9]+} = action=restAction&collection=0
55  * @endcode
56  *
57  * @param $controller The controller to call if none is given in request parameters (optional)
58  * @param $context The context to set if none is given in request parameters (optional)
59  * @param $action The action to perform if none is given in request parameters (optional)
60  * @return Request
61  */
62  public static function getDefault($controller=null, $context=null, $action=null) {
63 
64  // get base request data from request path
65  $basePath = preg_replace('/\/?[^\/]*$/', '', $_SERVER['SCRIPT_NAME']);
66  $requestUri = preg_replace('/\?.*$/', '', $_SERVER['REQUEST_URI']);
67  $requestPath = preg_replace('/^'.StringUtil::escapeForRegex($basePath).'/', '', $requestUri);
68  if (Log::isDebugEnabled(__CLASS__)) {
69  Log::debug("Request path: ".$requestPath, __CLASS__);
70  }
71  $config = ObjectFactory::getConfigurationInstance();
72 
73  $baseRequestValues = array();
74  $defaultValuePattern = '([^/]+)';
75  if ($config->hasSection('routes')) {
76  $routes = $config->getSection('routes');
77  foreach ($routes as $pattern => $requestDef) {
78 
79  // prepare route match pattern and extract parameters
80  $params = array();
81  $pattern = preg_replace_callback('/\{([^\}]+)\}/', function ($match) use($defaultValuePattern, &$params) {
82  // a variabel may be either defined by {name} or by {name|pattern} where
83  // name is the variable's name and pattern is an optional regex pattern, the
84  // values should match
85  $paramParts = explode('|', $match[1]);
86  // add the variable name to the parameter list
87  $params[] = $paramParts[0];
88  // return the value match pattern (defaults to defaultValuePattern)
89  return sizeof($paramParts) > 1 ? '('.$paramParts[1].')' : $defaultValuePattern;
90  }, $pattern);
91  $pattern = '/^'.str_replace('/', '\/', $pattern).'\/?$/';
92 
93  // try to match the currrent request path
94  if (Log::isDebugEnabled(__CLASS__)) {
95  Log::debug("Check patther: ".$pattern, __CLASS__);
96  }
97  $matches = array();
98  if (preg_match($pattern, $requestPath, $matches)) {
99  if (Log::isDebugEnabled(__CLASS__)) {
100  Log::debug("Match", __CLASS__);
101  }
102  // set parameters from request definition
103  parse_str($requestDef, $baseRequestValues);
104  // set parameters from request path
105  for ($i=0, $count=sizeof($params); $i<$count; $i++) {
106  $baseRequestValues[$params[$i]] = isset($matches[$i+1]) ? $matches[$i+1] : null;
107  }
108  break;
109  }
110  }
111  }
112 
113  // get additional request data from parameters
114  $requestValues = array_merge($_GET, $_POST, $_FILES);
115  $rawPostBody = file_get_contents('php://input');
116  // add the raw post data if they are json encoded
117  $json = json_decode($rawPostBody, true);
118  if (is_array($json)) {
119  foreach ($json as $key => $value) {
120  $requestValues[$key] = $value;
121  }
122  }
123 
124  // get controller/context/action triple
125  $controller = isset($requestValues['controller']) ?
126  filter_var($requestValues['controller'], FILTER_SANITIZE_STRING, FILTER_FLAG_STRIP_LOW) :
127  (isset($baseRequestValues['controller']) ? $baseRequestValues['controller'] : $controller);
128 
129  $context = isset($requestValues['context']) ?
130  filter_var($requestValues['context'], FILTER_SANITIZE_STRING, FILTER_FLAG_STRIP_LOW) :
131  (isset($baseRequestValues['context']) ? $baseRequestValues['context'] : $context);
132 
133  $action = isset($requestValues['action']) ?
134  filter_var($requestValues['action'], FILTER_SANITIZE_STRING, FILTER_FLAG_STRIP_LOW) :
135  (isset($baseRequestValues['action']) ? $baseRequestValues['action'] : $action);
136 
137  // setup request
138  $request = new Request($controller, $context, $action);
139  $request->setValues(array_merge($baseRequestValues, $requestValues));
140 
141  return $request;
142  }

160  /**
161  * Get the message response format. If no explicit format is set, the
162  * format is derived from the Content-Type header value, if existing.
163  * If no format can be derived, the first format in the configuration
164  * key 'Formats' will be used.
165  * @return Format instance
166  */
167  public function getResponseFormat() {
168  if ($this->_responseFormat == null) {
169  $this->_responseFormat = self::getFormatFromMimeType($this->getHeader('Accept'));
170  }
171  return $this->_responseFormat;
172  }

174  /**
175  * Get all http headers
176  * @return Associative array
177  */
178  private static function getAllHeaders() {
179  $headers = array();
180  foreach ($_SERVER as $name => $value) {
181  if (substr($name, 0, 5) == 'HTTP_') {
182  $name = str_replace(' ', '-', ucwords(strtolower(str_replace('_', ' ', substr($name, 5)))));
183  $headers[$name] = $value;
184  }
185  else if ($name == "CONTENT_TYPE") {
186  $headers["Content-Type"] = $value;
187  }
188  else if ($name == "CONTENT_LENGTH") {
189  $headers["Content-Length"] = $value;
190  }
191  }
192  return $headers;
193  }
