20 /**
21  * ConcurrencyController is used to lock/unlock objects.
22  *
23  * The controller supports the following actions:
24  *
25  * <div class="controller-action">
26  * <div> __Action__ lock </div>
27  * <div>
28  * Lock an object.
29  * | Parameter | Description
30  * |-----------------------|-------------------------
31  * | _in_ / _out_ `oid` | The object id of the object to lock
32  * | _in_ / _out_ `type` | The lock type (_optimistic_ or _pessimistic_) (optional, default: _optimistic_)
33  * | __Response Actions__ | |
34  * | `ok` | In all cases
35  * </div>
36  * </div>
37  *
38  * <div class="controller-action">
39  * <div> __Action__ unlock </div>
40  * <div>
41  * Unlock an object.
42  * | Parameter | Description
43  * |-----------------------|-------------------------
44  * | _in_ / _out_ `oid` | The object id of the object to unlock
45  * | _in_ / _out_ `type` | The lock type (_optimistic_ or _pessimistic_) (optional, default: _optimistic_)
46  * | __Response Actions__ | |
47  * | `ok` | In all cases
48  * </div>
49  * </div>
50  *
51  * @note If the user already holds a pessimistic lock, and tries to aquire an
52  * optimistic lock, the returned lock type is still pessimistic.
53  *
54  * @author ingo herwig <ingo@wemove.com>
55  */
56 class ConcurrencyController extends Controller {
57 
58  /**
59  * @see Controller::validate()
60  */
61  function validate() {
62  $request = $this->getRequest();
63  $response = $this->getResponse();
64  $oid = ObjectId::parse($request->getValue('oid'));
65  if(!$oid) {
66  $response->addError(ApplicationError::get('OID_INVALID',
67  array('invalidOids' => array($request->getValue('oid')))));
68  return false;
69  }
70  $lockType = $request->getValue('type', Lock::TYPE_OPTIMISTIC);
71  if (!in_array($lockType, array(Lock::TYPE_OPTIMISTIC, Lock::TYPE_PESSIMISTIC))) {
72  $response->addError(ApplicationError::get('PARAMETER_INVALID',
73  array('invalidParameters' => array('type'))));
74  }
75  return true;
76  }
77 
78  /**
79  * @see Controller::doExecute()
80  */
81  function doExecute() {
82  $request = $this->getRequest();
83  $response = $this->getResponse();
84  $concurrencyManager = ObjectFactory::getInstance('concurrencyManager');
85  $oid = ObjectId::parse($request->getValue('oid'));
86  $lockType = $request->getValue('type', Lock::TYPE_OPTIMISTIC);
87 
88  // process actions
89  try {
90  if ($request->getAction() == 'lock') {
91  $concurrencyManager->aquireLock($oid, $lockType);
92  $lock = $concurrencyManager->getLock($oid);
93  $response->setValue('type', $lock->getType());
94  }
95  elseif ($request->getAction() == 'unlock') {
96  $concurrencyManager->releaseLock($oid, $lockType);
97  }
98  }
99  catch (PessimisticLockException $ex) {
100  $response->addError(ApplicationError::get('OBJECT_IS_LOCKED',
101  array('lockedOids' => array($oid->__toString()))));
102  }
103 
104  $response->setValue('oid', $oid);
105  $response->setAction('ok');
106  }
107 }
