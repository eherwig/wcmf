122  * @see PersistenceFacade::load()
123  */
124  public function load(ObjectId $oid, $buildDepth=BuildDepth::SINGLE) {
125  if ($buildDepth < 0 && !in_array($buildDepth, array(BuildDepth::INFINITE, BuildDepth::SINGLE))) {
126  throw new IllegalArgumentException("Build depth not supported: $buildDepth");
127  }
128  // check if the object is already part of the transaction
129  $transaction = $this->getTransaction();
130  $obj = $transaction->getLoaded($oid);
131 
132  // if not cached or build depth requested, load
133  if ($obj == null || $buildDepth != BuildDepth::SINGLE) {
134  $mapper = $this->getMapper($oid->getType());
135  $obj = $mapper->load($oid, $buildDepth);
136  }
137  return $obj;
138  }

199  /**
200  * @see PersistenceFacade::loadObjects()
201  */
202  public function loadObjects($typeOrTypes, $buildDepth=BuildDepth::SINGLE, $criteria=null, $orderby=null, PagingInfo $pagingInfo=null) {
203  $this->checkArrayParameter($criteria, 'criteria', 'wcmf\lib\persistence\Criteria');
204  $this->checkArrayParameter($orderby, 'orderby');
205 
206  if (!is_array($typeOrTypes)) {
207  // single type
208  $mapper = $this->getMapper($typeOrTypes);
209  $result = $mapper->loadObjects($typeOrTypes, $buildDepth, $criteria, $orderby, $pagingInfo);
210  }
211  else {
212  // multiple types
213  $numTypes = sizeof($typeOrTypes);
214  $cache = ObjectFactory::getInstance('dynamicCache');
215  $cacheSection = str_replace('\\', '.', __CLASS__);
216 
217  // get cache key for stored offsets of previous page
218  $page = $pagingInfo->getPage();
219  $prevPage = $page > 1 ? $page-1 : 1;
220  $prevPagingInfo = new PagingInfo($pagingInfo->getPageSize(), true);
221  $prevPagingInfo->setPage($prevPage);
222  $prevCacheKey = $this->getCacheKey($typeOrTypes, $buildDepth, $criteria, $orderby, $prevPagingInfo);
223 
224  // get previous offsets
225  $offsets = array_fill(0, $numTypes, 0); // default
226  if ($cache->exists($cacheSection, $prevCacheKey)) {
227  $offsets = $cache->get($cacheSection, $prevCacheKey);
228  }
229  elseif ($page > 1) {
230  // previous offsets must be generated by loading the pages for pages other than first
231  for ($i=1; $i<$page; $i++) {
232  $tmpPagingInfo = new PagingInfo($pagingInfo->getPageSize(), true);
233  $tmpPagingInfo->setPage($i);
234  $tmpCacheKey = $this->getCacheKey($typeOrTypes, $buildDepth, $criteria, $orderby, $tmpPagingInfo);
235  if (!$cache->exists($cacheSection, $tmpCacheKey)) {
236  $this->loadObjects($typeOrTypes, $buildDepth, $criteria, $orderby, $tmpPagingInfo);
237  }
238  $offsets = $cache->get($cacheSection, $tmpCacheKey);
239  }
240  }
241 
242  $tmpResult = array();
243  for ($i=0, $countI=$numTypes; $i<$countI; $i++) {
244  // collect n objects from each type
245  $type = $this->getFullyQualifiedType($typeOrTypes[$i]);
246  $mapper = $this->getMapper($type);
247  $pkNames = $mapper->getPkNames();
248 
249  // use type specific criteria
250  $typeCriteria = array();
251  if ($criteria != null) {
252  foreach ($criteria as $criterion) {
253  if ($this->getFullyQualifiedType($criterion->getType()) == $type) {
254  $typeCriteria[] = $criterion;
255  }
256  }
257  }
258 
259  // set offset condition
260  $pkOffsets = explode('|', $offsets[$i]);
261  for ($j=0, $countJ=sizeof($pkNames); $j<$countJ; $j++) {
262  $offset = isset($pkOffsets[$j]) ? $pkOffsets[$j] : 0;
263  $typeCriteria[] = new Criteria($type, $pkNames[$j], '>', $offset);
264  }
265 
266  // always start from first page
267  $tmpPagingInfo = new PagingInfo($pagingInfo->getPageSize(), true);
268 
269  $objects = $mapper->loadObjects($type, $buildDepth, $typeCriteria, $orderby, $tmpPagingInfo);
270  $tmpResult = array_merge($tmpResult, $objects);
271  }
272 
273  // sort
274  if ($orderby != null) {
275  $comparator = new NodeComparator($orderby);
276  usort($tmpResult, array($comparator, 'compare'));
277  }
278 
279  // truncate
280  $result = array_slice($tmpResult, 0, $pagingInfo->getPageSize());
281 
282  // update offsets
283  for ($i=0, $countI=$numTypes; $i<$countI; $i++) {
284  // find last object of type
285  $type = $this->getFullyQualifiedType($typeOrTypes[$i]);
286  for ($j=sizeof($result)-1; $j>=0; $j--) {
287  $object = $result[$j];
288  if ($object->getType() == $type) {
289  $mapper = $this->getMapper($type);
290  $pkNames = $mapper->getPkNames();
291  $newOffset = array();
292  foreach ($pkNames as $pkName) {
293  $newOffset[] = $object->getValue($pkName);
294  }
295  $offsets[$i] = join('|', $newOffset);
296  break;
297  }
298  }
299  }
300  $cacheKey = $this->getCacheKey($typeOrTypes, $buildDepth, $criteria, $orderby, $pagingInfo);
301  $cache->put($cacheSection, $cacheKey, $offsets);
302  }
303  return $result;
304  }

352  /**
353  * Check if the given value is either null or an array and
354  * throw an exception if not
355  * @param $param The parameter
356  * @param $name The name of the parameter (used in the exception text)
357  * @param $className Classname to match if, instances of a specific type are expected (optional)
358  */
359  private function checkArrayParameter($param, $paramName, $className=null) {
360  if ($param == null) {
361  return;
362  }
363  if (!is_array($param)) {
364  throw new IllegalArgumentException("The parameter '".$paramName.
365  "' is expected to be null or an array");
366  }
367  if ($className != null) {
368  foreach ($param as $instance) {
369  if (!($instance instanceof $className)) {
370  throw new IllegalArgumentException("The parameter '".$paramName.
371  "' is expected to contain only instances of '".$className."'");
372  }
373  }
374  }

377  /**
378  * Listen to StateChangeEvents
379  * @param $event StateChangeEvent instance
380  */
381  public function stateChanged(StateChangeEvent $event) {
382  $oldState = $event->getOldValue();
383  $newState = $event->getNewValue();
384  // store the object id in the internal registry if the object was saved after creation
385  if ($oldState == PersistentObject::STATE_NEW && $newState == PersistentObject::STATE_CLEAN) {
386  $object = $event->getObject();
387  $type = $object->getType();
388  if (!array_key_exists($type, $this->createdOIDs)) {
389  $this->createdOIDs[$type] = array();
390  }
391  $this->createdOIDs[$type][] = $object->getOID();
392  }
393  }

405  $result = is_array($typeOrTypes) ? join(',', $typeOrTypes) : $typeOrTypes;
406  $result .= ','.$buildDepth.',';
407  if ($criteriaArray != null) {

411  }
412  if ($orderArray != null) {
413  $result .= join(',', $orderArray);
414  }
415  if ($pagingInfo != null) {
416  $result .= ','.$pagingInfo->getOffset().','.$pagingInfo->getPageSize();
417  }
418  return $result;
