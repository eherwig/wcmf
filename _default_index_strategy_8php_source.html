29  $valueNames = $obj->getValueNames(true);

31  $doc->addField(Field::keyword('oid', $obj->getOID()->__toString(), 'UTF-8'));
32  $typeField = Field::keyword('type', $obj->getType(), 'UTF-8');
33  $typeField->isStored = false;
34  $doc->addField($typeField);
35  if ($language != null) {
36  $languageField = Field::keyword('lang', $language, 'UTF-8');
37  $languageField->isStored = false;

41  foreach ($valueNames as $curValueName) {
42  $inputType = $obj->getValueProperty($curValueName, 'input_type');
43  $value = $obj->getValue($curValueName);
44  if (!is_object($value) && !is_array($value)) {
45  $value = $this->encodeValue($value, $inputType);
46  if (preg_match('/^text|^f?ckeditor/', $inputType)) {
47  $value = strip_tags($value);
48  $doc->addField(Field::unStored($curValueName, $value, 'UTF-8'));
49  }
50  else {

55  }
