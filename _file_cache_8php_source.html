17 /**
18  * FileCache provides simple caching functionality.
19  *
20  * The cache is divided into different sections. All sections share
21  * the same base path.
22  *
23  * @author ingo herwig <ingo@wemove.com>
24  */
25 class FileCache implements Cache {
26 
27  private $cacheDir = null;

31  /**
32  * Constructor

34  */

42  * If not existing, the directory will be created relative to WCMF_BASE.
43  * @param $cacheDir String

49  /**
50  * @see Cache::exists()
51  */
52  public function exists($section, $key) {
53  $this->initializeCache($section);
54  return isset($this->cache[$section][$key]) &&
55  !$this->isExpired($this->cache[$section][$key][0], $this->cache[$section][$key][1]);
56  }
57 
58  /**
59  * @see Cache::getDate()
60  */
61  public function getDate($section, $key) {
62  $this->initializeCache($section);
63  if (isset($this->cache[$section][$key]) &&
64  !$this->isExpired($this->cache[$section][$key][0], $this->cache[$section][$key][1])) {
65  return (new \DateTime())->setTimeStamp($this->cache[$section][$key][0]);
66  }
67  return null;
68  }
69 
70  /**
71  * @see Cache::get()
72  */
73  public function get($section, $key) {
74  $this->initializeCache($section);
75  if (isset($this->cache[$section][$key]) &&
76  !$this->isExpired($this->cache[$section][$key][0], $this->cache[$section][$key][1])) {
77  return $this->cache[$section][$key][2];
78  }
79  return null;
80  }
81 
82  /**
83  * @see Cache::put()
84  */
85  public function put($section, $key, $value, $lifetime=null) {
86  $this->initializeCache($section);
87  $this->cache[$section][$key] = [time(), $lifetime, $value];
88  $this->saveCache($section);
89  }
90 
91  /**
92  * @see Cache::clear()
93  */
94  public function clear($section) {
95  if (preg_match('/\*$/', $section)) {
96  // handle wildcards
97  $cachBaseDir = $this->getCacheDir();
98  $directory = $cachBaseDir.dirname($section);
99  if (is_dir($directory)) {
100  $pattern = '/^'.preg_replace('/\*$/', '', $this->fileUtil->basename($section)).'/';
101  $files = $this->fileUtil->getFiles($directory, $pattern, true, true);
102  foreach ($files as $file) {
103  $this->clear(str_replace($cachBaseDir, '', $file));
104  }
105  $directories = $this->fileUtil->getDirectories($directory, $pattern, true, true);
106  foreach ($directories as $directory) {
107  $this->clear(str_replace($cachBaseDir, '', $directory).'/*');
108  @rmdir($directory);
109  }
110  }
111  }
112  else {
113  $file = $this->getCacheFile($section);
114  @unlink($file);
115  unset($this->cache[$section]);
116  }
117  }
118 
119  /**
120  * @see Cache::clearAll()
121  */
122  public function clearAll() {
123  $cacheDir = $this->getCacheDir();
124  if (is_dir($cacheDir)) {
125  $this->fileUtil->emptyDir($cacheDir);
126  }
127  $this->cache = null;
128  }
129 
130  /**
131  * Initialize the cache
132  * @param $section The caching section
133  */
134  private function initializeCache($section) {
135  if (!isset($this->cache[$section])) {
136  $file = $this->getCacheFile($section);
137  if (file_exists($file)) {
138  $this->cache[$section] = unserialize(file_get_contents($file));
139  }
140  else {
141  $this->cache[$section] = [];
142  }
143  }
144  }
145 
146  /**
147  * Check if an entry with the given creation timestamp and lifetime is expired
148  * @param $createTs The creation timestamp
149  * @param $lifetime The lifetime in seconds
150  * @return Boolean
151  */
152  private function isExpired($createTs, $lifetime) {
153  if ($lifetime === null) {
154  return false;
155  }
156  $expireDate = (new \DateTime())->setTimeStamp($createTs);
157  if (intval($lifetime)) {
158  $expireDate = $expireDate->modify('+'.$lifetime.' seconds');
159  }
160  return $expireDate < new \DateTime();
161  }
162 
163  /**
164  * Save the cache
165  * @param $section The caching section
166  */
167  private function saveCache($section) {
168  $content = serialize($this->cache[$section]);
169  $file = $this->getCacheFile($section);
170  $this->fileUtil->mkdirRec(dirname($file));
171  $fh = fopen($file, "w");
172  if ($fh !== false) {
173  fwrite($fh, $content);
174  fclose($fh);
175  }
176  else {
177  throw new ConfigurationException("The cache path is not writable: ".$file);
178  }
179  }
180 
181  /**
182  * Get the cache root directory
183  * @return String
184  */
185  private function getCacheDir() {
186  if ($this->cacheDir == null) {
187  $this->cacheDir = WCMF_BASE.'/app/cache/';
188  }
189  return $this->cacheDir;
190  }
191 
192  /**
193  * Get the cache file for the specified cache
194  * @param $section The caching section
195  * @return String
196  */
197  private function getCacheFile($section) {
198  return $this->getCacheDir().$section;
199  }
