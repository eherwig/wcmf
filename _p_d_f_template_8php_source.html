11 namespace wcmf\lib\pdf;
12 
13 use \RuntimeException;
14 use wcmf\lib\pdf\PDF;
15 use wcmf\lib\pdf\PDFPage;
16 
17 /**
18  * PDFTemplate is used to output pdf files based on a given pdf template.
19  * PDFTemplate uses FPDI/FPDF. PDFPage instances are used to render data onto
20  * the template pages.
21  *
22  * The following example shows the usage:
23  *
24  * @code
25  * // example Controller method to show a pdf download dialog
26  * // Page1 extends PDFPage and defines what is rendered onto the template
27  * function doExecute()
28  * {
29  * $template = new PDFTemplate(new MyPDF());
30  * // set the template
31  * $template->setTemplate('include/pdf/wcmf.pdf');
32  * // render Page1 on every second template page
33  * $template->setPages(array(new Page1(), null), true);
34  *
35  * // output the final page
36  * $downloadfile = 'wcmf.pdf';
37  * header("Content-disposition: attachment; filename=$downloadfile");
38  * header("Content-Type: application/force-download");
39  * header("Content-Transfer-Encoding: binary");
40  * header("Pragma: no-cache");
41  * header("Expires: 0");
42  * echo $template->output($downloadfile, 'S');
43  * }
44  * @endcode
45  *
46  * @author ingo herwig <ingo@wemove.com>
47  */
48 class PDFTemplate {
49 
50  var $_pdf = null;
51  var $_tpl = null;
52  var $_pages = array();
53  var $_cycle = false;
54  var $_data = null;
55 
56  /**
57  * Constructor
58  * @param $pdf The PDF instance to render onto, defaults to PDF created with default constructor
59  */
60  public function __construct($pdf) {
61  if (!isset($pdf) || (!($pdf instanceof PDF))) {
62  $this->_pdf = new PDF();
63  }
64  else {
65  $this->_pdf = $pdf;
66  }
67  }
68 
69  /**
70  * Set the template filename
71  * @param $filename The name of the file
72  */
73  public function setTemplate($filename) {
74  $this->_tpl = $filename;
75  }
76 
77  /**
78  * Set the PDFPage instances used to write the content to the template.
79  * The page instances will be used one after another: The 1 instance writes to the first
80  * template page, the second to the second and so on. Use the cycle parameter to cycle the
81  * instances (e.g. if the same data should be written to every template page, put one
82  * instance into the pages array and set cycle to true)
83  * @param $pages An array of PDFPage instances
84  * @param $cycle Boolean whether to cycle the PDFPage instances or not (default: _false_)
85  * @param $data An optional data object, that will passed to the PDFPage::render method (default: _null_)
86  */
87  public function setPages($pages, $cycle=false, $data=null) {
88  $this->_pages = $pages;
89  $this->_cycle = $cycle;
90  $this->_data = &$data;
91  }
92 
93  /**
94  * Output the pdf. Delegates to FPDF::Output()
95  * @see http://www.fpdf.de/funktionsreferenz/Output/
96  * @param $name The name of the pdf file
97  * @param $dest The pdf destination ('I': browser inline, 'D': browser download, 'F': filesystem, 'S': string)
98  * @return The document string in case of dest = 'S', nothing else
99  */
100  public function output($name='', $dest='') {
101 
102  if ($this->_tpl == null) {
103  throw new RuntimeException("No PDF template provided. Use PDFTemplate::setTemplate.");
104  }
105 
106  $pageIndex = 0;
107  $numPages = $this->_pdf->setSourceFile($this->_tpl);
108  for ($i=1; $i<=$numPages; $i++) {
109  // add each page
110  $tplIndex = $this->_pdf->ImportPage($i);
111  $size = $this->_pdf->getTemplatesize($tplIndex);
112  $this->_pdf->AddPage($size['h'] > $size['w'] ? 'P' : 'L');
113 
114  // render the PDFPage onto the template page
115  if ($pageIndex < sizeof($this->_pages)) {
116  $curPage = &$this->_pages[$pageIndex];
117  if ($curPage instanceof PDFPage) {
118  $this->_pdf->startPage();
119  $curPage->render($this->_pdf, $i, $this->_data);
120  $this->_pdf->endPage();
121  }
122  }
123  $pageIndex++;
124 
125  // cycle pages if required
126  if ($this->_cycle && $pageIndex == sizeof($this->_pages)) {
127  $pageIndex = 0;
128  }
129  }
130 
131  // output the pdf
132  return $this->_pdf->Output($name, $dest);
133  }
134 }
