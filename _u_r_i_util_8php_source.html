22  /**
23  * Convert an absolute URI to a relative
24  * code from http://www.webmasterworld.com/forum88/334.htm
25  * @param $abs_uri Absolute URI to convert
26  * @param $base Base URI
27  */
28  public static function makeRelative($abs_uri, $base) {
29  $abs_uri = preg_replace("{^[^:]+://[^/]+}", '', $abs_uri);
30  $base = preg_replace("{^[^:]+://[^/]+}", '', $base);
31 
32  $abs_array = explode('/', $abs_uri);
33  $base_array = explode('/', $base);
34 
35  // remove trailing file names
36  $fileName = '';
37  if (strrpos($abs_uri, '/') !== strlen($abs_uri)) {
38  $fileName = array_pop($abs_array);
39  }
40  if (strrpos($base, '/') !== strlen($base)) {
41  array_pop($base_array);
42  }
43  // ignore common path
44  while (sizeof($abs_array) > 0 && sizeof($base_array) > 0 && $abs_array[0] == $base_array[0]) {
45  array_shift($abs_array);
46  array_shift($base_array);
47  }
48 
49  // construct connecting path
50  $rel_uri = str_repeat('../', sizeof($base_array)).join('/', $abs_array).'/'.$fileName;
51  return $rel_uri;
52  }

54  /**
55  * Convert a relative URI to an absolute
56  * code from http://99webtools.com/relative-path-into-absolute-url.php
57  * @param $rel_uri Relative URI to convert
58  * @param $base Base URI
59  */
60  public static function makeAbsolute($rel_uri, $base) {
61  if(strpos($rel_uri, "//") === 0) {
62  return "http:".$rel_uri;
63  }
64  /* return if already absolute URL */
65  if (parse_url($rel_uri, PHP_URL_SCHEME) != '') {
66  return $rel_uri;
67  }
68  /* add slash to base if missing */
69  if (!preg_match('/\/$/', $base)) {
70  $base .= '/';
71  }
72  $firstChar = (strlen($rel_uri) > 0) ? substr($rel_uri, 0, 1) : '';
73  /* queries and anchors */
74  if ($firstChar == '#' || $firstChar == '?') {
75  return $base.$rel_uri;
76  }
77  /* parse base URL and convert to local variables: $scheme, $host, $path */
78  extract(parse_url($base));
79  /* remove non-directory element from path */
80  $path = preg_replace('#/[^/]*$#', '', $path);
81  /* destroy path if relative url points to root */
82  if ($firstChar == '/') {
83  $path = '';
84  }
85  /* dirty absolute URL */
86  $abs = "$host$path/$rel_uri";
87  /* replace '//' or '/./' or '/foo/../' with '/' */
88  $re = array('#(/\.?/)#', '#/(?!\.\.)[^/]+/\.\./#');
89  for ($n=1; $n>0; $abs=preg_replace($re, '/', $abs, -1, $n)) {}
90  /* absolute URL is ready! */
91  return $scheme.'://'.$abs;
92  }

94  /**
95  * Translate a relative URI from one location to the script location.
96  * For example if a file path is stored relative to location A and should be
97  * translated to the script URI (location B), use
98  * URIUtil::translate($filepathAsSeenFromA, $pathFromBtoA)
99  * @param $rel_uri Relative URI to translate as seen from base
100  * @param $base Base URI
101  * @return An associtative array with keys 'absolute' and 'relative'
102  * and the absolute and relative URI (as seen from the executed script) as values
103  */
104  public static function translate($rel_uri, $base) {
105  $self = UriUtil::getProtocolStr().$_SERVER['HTTP_HOST'].$_SERVER['SCRIPT_NAME'];
106  $path = dirname($self).'/';
107  $absUrl = self::makeAbsolute($rel_uri, $path.$base);
108  $relUrl = self::makeRelative($absUrl, $self);
109 
110  return array('absolute' => $absUrl, 'relative' => $relUrl);
111  }

113  /**
114  * Check if an url is available (HTTP-Code: 200)
115  * @note requires cURL library
116  * @param $url The url to check
117  * @param $timeout The timeout in seconds (default: 5)
118  * @return Boolean whether the url is available
119  */
120  public static function validateUrl($url, $timeout=5) {
121  $url_parts = @parse_url($url);
122  // check local relative url
123  if (empty($url_parts["host"])) {
124  $fh = @fopen($url, "r");
125  return ($fh !== false);
126  }
127 
128  // check remote url
129  $ch = curl_init();
130  curl_setopt($ch, CURLOPT_URL, $url);
131  curl_setopt($ch, CURLOPT_HEADER, true);
132  curl_setopt($ch, CURLOPT_NOBODY, true);
133  curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
134  curl_setopt($ch, CURLOPT_TIMEOUT, $timeout);
135  $r = curl_exec($ch);
136  $headers = split("\n", $r);
137  if (Log::isDebugEnabled(__CLASS__)) {
138  Log::debug("$url: HTTP-Response: ". json_encode($headers), __CLASS__);
139  }
140 
141  preg_match('/.+ ([0-9]{3}) .+/', $headers[0], $matches);
142  return (intval($matches[1]) < 400);
143  }
