79  /**
80  * Get an instance from the configuration.
81  * @param $name The name of the instance (section, where the instance is defined)
82  * @return Object
83  */
84  public static function getInstance($name) {
85  $instance = null;
86 
87  // check if the instance is registered already
88  if (!isset(self::$_instances[$name])) {
89  if (self::$_configuration == null) {
90  throw new ConfigurationException('No Configuration instance provided. Do this by calling the configure() method.');
91  }
92  // get instance configuration
93  $instanceConfig = self::$_configuration->getSection($name);
94  $instance = self::createInstance($name, $instanceConfig);
95  }
96  else {
97  $instance = self::$_instances[$name];
98  }
99  return $instance;
100  }

135  /**
136  * Create an instance from the given configuration array.
137  * @param $name The name by which the instance may be retrieved later
138  * using ObjectFactory::getInstance()
139  * @param $configuration Associative array with key value pairs for instance properties
140  * and the special keys '__class' (optional, denotes the instance class name),
141  * '__shared' (optional, if true, the instance may be reused using ObjectFactory::getInstance())
142  * @return Object
143  */
144  public static function createInstance($name, $configuration) {
145  $instance = null;
146 
147  if (isset($configuration['__class'])) {
148  // the instance belongs to the given class
149  $className = $configuration['__class'];
150 
151  // class definition must be supplied by autoloader
152  if (class_exists($className)) {
153  // create the instance
154  $obj = new $className;
155  // check against interface
156  $interface = self::getInterface($name);
157  if ($interface != null && !($obj instanceof $interface)) {
158  throw new ConfigurationException('Class \''.$className.
159  '\' is required to implement interface \''.$interface.'\'.');
160  }
161  // set the instance properties
162  foreach ($configuration as $key => $value) {
163  // exclude properties starting with __
164  if (strpos($key, '__') !== 0) {
165  // special treatments, if value is a string
166  if (is_string($value)) {
167  // replace variables denoted by a leading $
168  if (strpos($value, '$') === 0) {
169  $value = self::getInstance(preg_replace('/^\$/', '', $value));
170  }
171  else {
172  // convert booleans
173  $lower = strtolower($value);
174  if ($lower === 'true') {
175  $value = true;
176  }
177  if ($lower === 'false') {
178  $value = false;
179  }
180  }
181  }
182  // special treatments, if value is an array
183  if (is_array($value)) {
184  $result = array();
185  $containsInstance = false;
186  // check for variables
187  foreach ($value as $val) {
188  if (is_string($val) && strpos($val, '$') === 0) {
189  $result[] = self::getInstance(preg_replace('/^\$/', '', $val));
190  $containsInstance = true;
191  }
192  else {
193  $result[] = $val;
194  }
195  }
196  // only replace value, if the array containes an variable
197  if ($containsInstance) {
198  $value = $result;
199  }
200  }
201  // set the property
202  $setterName = self::getSetterName($key);
203  if (method_exists($obj, $setterName)) {
204  $obj->$setterName($value);
205  }
206  else {
207  $obj->$key = $value;
208  }
209  }
210  }
211  // register the instance if it is shared (default)
212  if (!isset($configuration['__shared']) || $configuration['__shared'] == 'true') {
213  self::registerInstance($name, $obj);
214  }
215  $instance = $obj;
216  }
217  else {
218  throw new ConfigurationException('Class \''.$className.'\' is not found.');
219  }
220  }
221  else {
222  // the instance is a map
223 
224  // get interface that map values should implement
225  $interface = self::getInterface($name);
226 
227  foreach ($configuration as $key => $value) {
228  // create instances for variables denoted by a leading $
229  if (strpos($value, '$') === 0) {
230  $obj = self::getInstance(preg_replace('/^\$/', '', $value));
231  // check against interface
232  if ($interface != null && !($obj instanceof $interface)) {
233  throw new ConfigurationException('Class of \''.$name.'.'.$key.
234  '\' is required to implement interface \''.$interface.'\'.');
235  }
236  $configuration[$key] = $obj;
237  }
238  }
239  // always register maps
240  self::registerInstance($name, $configuration);
241  $instance = $configuration;
242  }
243  return $instance;
244  }

255  return null;
256  }
