51  /**

53  * The object will be bound to the appropriate PersistenceMapper automatically,
54  * if the PersistenceFacade knows the type. The object id is needed to extract
55  * the type. If the id parameter of the object id is a dummy id, the object
56  * is supposed to be a newly created object (@see ObjectId::containsDummyIds()).
57  * @param $oid ObjectId instance (optional)
58  */

115  /**
116  * Set the object id of the PersistentObject.
117  * @param $oid The PersistentObject's oid.
118  * @param $triggerListeners Boolean, whether value CahngeListeners should be
119  * notified or not
120  */
121  protected function setOIDInternal(ObjectId $oid, $triggerListeners) {
122  $this->_type = $oid->getType();
123  $this->_oid = $oid;
124  $mapper = $this->getMapper();
125  if ($mapper != null) {
126  // update the primary key attributes
127  $ids = $oid->getId();
128  $pkNames = $mapper->getPkNames();
129  for ($i=0, $count=sizeof($pkNames); $i<$count; $i++) {
130  if ($triggerListeners) {
131  $this->setValue($pkNames[$i], $ids[$i], true);
132  }
133  else {
134  $this->setValueInternal($pkNames[$i], $ids[$i]);
135  }
136  }
137  }

184  /**
185  * Set the state of the object to one of the STATE constants.
186  */
187  public function setState($state) {
188  $oldState = $this->_state;
189  switch ($oldState) {
190  // new object must stay new when it's modified
191  case self::STATE_NEW:
192  switch ($state) {
193  case self::STATE_DIRTY:
194  $this->_state = self::STATE_NEW;
195  break;
196 
197  default:
198  $this->_state = $state;
199  }
200  break;
201 
202  // deleted object must stay deleted in every case
203  case self::STATE_DELETED:
204  $this->_state = self::STATE_DELETED;
205  break;
206 
207  default:
208  $this->_state = $state;
209  }
210  if ($oldState != $this->_state) {
211  ObjectFactory::getInstance('eventManager')->dispatch(StateChangeEvent::NAME,
212  new StateChangeEvent($this, $oldState, $this->_state));
213  }
214  }

379  /**
380  * Validate all values by calling PersistentObject::validateValue()
381  * Throws a ValidationException in case of invalid data.
382  */
383  public function validateValues() {
384  $errorMessage = '';
385  $iter = new NodeValueIterator($this, false);
386  for($iter->rewind(); $iter->valid(); $iter->next()) {
387  $curNode = $iter->currentNode();
388  try {
389  $curNode->validateValue($iter->key(), $iter->current());
390  }
391  catch (ValidationException $ex) {
392  $errorMessage .= $ex->getMessage()."\n";
393  }

413  /**
414  * Check a value's value against the validation type set on it. This method uses the
415  * validateType property of the attribute definition.
416  * Throws a ValidationException if the valud is not valid.
417  * @param $name The name of the item to set.
418  * @param $value The value of the item.
419  */
420  protected function validateValueAgainstValidateType($name, $value) {
421  $validateType = $this->getValueProperty($name, 'validate_type');
422  if (($validateType == '' || Validator::validate($value, $validateType))) {
423  return;
424  }
425  // construct the error message
426  $errorMessage = Message::get("Wrong value for %0% (%1%). ", array($name, $value));
427 
428  // use configured message if existing
429  $validateDescription = $this->getValueProperty($name, 'validate_description');
430  if (strlen($validateDescription) > 0) {
431  $errorMessage .= Message::get($validateDescription);
432  }
433  else {
434  // construct default message
435  if (strlen($validateType) > 0) {
436  $errorMessage .= Message::get("The value must match %0%.", array($validateType));
437  }
438  }
439  throw new ValidationException($errorMessage);
440  }

442  /**
443  * Set the value of a named item if it exists.
444  * @param $name The name of the item to set.
445  * @param $value The value of the item.
446  * @param $forceSet Boolean whether to set the value even if it is already set
447  * and to bypass validation (used to notify listeners) (default: _false_)
448  * @param $trackChange Boolean whether to track the change (change state, notify listeners) or not (default: _true_)
449  * Only set this false, if you know, what you are doing
450  * @return Boolean whether the operation succeeds or not
451  */
452  public function setValue($name, $value, $forceSet=false, $trackChange=true) {
453  if (!$forceSet) {
454  try {
455  $this->validateValue($name, $value);
456  }
457  catch(ValidationException $ex) {
458  $msg = "Invalid value (".$value.") for ".$this->getOID().".".$name.": ".$ex->getMessage();
459  throw new ValidationException($msg);
460  }
461  }
462  $oldValue = self::getValue($name);
463  if ($forceSet || $oldValue !== $value) {
464  $this->setValueInternal($name, $value);
465  $mapper = $this->getMapper();
466  if ($mapper != null && in_array($name, $mapper->getPKNames())) {
467  $this->updateOID();
468  }
469  if ($trackChange) {
470  self::setState(self::STATE_DIRTY);
471  $this->_changedAttributes[$name] = true;
472  ObjectFactory::getInstance('eventManager')->dispatch(ValueChangeEvent::NAME,
473  new ValueChangeEvent($this, $name, $oldValue, $value));
474  }
475  return true;
476  }
477  return false;
478  }
