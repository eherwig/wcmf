92  /**
93  * @see Request::initialize()
94  *
95  * The method tries to match the current request path against the routes
96  * defined in the configuration section 'routes' and constructs the request based on
97  * these parameters. It then adds all data contained in $_GET, $_POST, $_FILES and
98  * php://input (raw data from the request body).
99  *
100  * Examples for route definitions are:
101  * @code
102  * GET/ = action=cms
103  * GET,POST,PUT,DELETE/rest/{language}/{className} = action=restAction&collection=1
104  * GET,POST,PUT,DELETE/rest/{language}/{className}/{id|[0-9]+} = action=restAction&collection=0
105  * @endcode
106  */
107  public function initialize($controller=null, $context=null, $action=null) {
108  // get base request data from request path
109  $basePath = preg_replace('/\/?[^\/]*$/', '', $_SERVER['SCRIPT_NAME']);
110  $requestUri = preg_replace('/\?.*$/', '', $_SERVER['REQUEST_URI']);
111  $requestPath = preg_replace('/^'.StringUtil::escapeForRegex($basePath).'/', '', $requestUri);
112  $requestMethod = $this->getMethod();
113  if (self::$logger->isDebugEnabled()) {
114  self::$logger->debug("Request: ".$requestMethod." ".$requestPath);
115  }
116 
117  // get all routes from the configuration that match the request path
118  $matchingRoutes = $this->getMatchingRoutes($requestPath);
119  if (self::$logger->isDebugEnabled()) {
120  self::$logger->debug("Matching routes:");
121  self::$logger->debug($matchingRoutes);
122  }
123 
124  // get client info error for logging
125  $clientInfo = array('ip' => $_SERVER['REMOTE_ADDR'],
126  'agent' => $_SERVER['HTTP_USER_AGENT'],
127  'referrer' => $_SERVER['HTTP_REFERER']);
128 
129  // check if the requested route matches any configured route
130  if (sizeof($matchingRoutes) == 0) {
131  throw new ApplicationException($this, $this->getResponse(),
132  ApplicationError::get('ROUTE_NOT_FOUND', array_merge(
133  $clientInfo, array('route' => $requestPath))));
134  }
135 
136  // get the best matching route
137  $route = $this->getBestRoute($matchingRoutes);
138  if (self::$logger->isDebugEnabled()) {
139  self::$logger->debug("Best route:");
140  self::$logger->debug($route);
141  }
142 
143  // check if method is allowed
144  $allowedMethods = $route['methods'];
145  if ($allowedMethods != null && !in_array($requestMethod, $allowedMethods)) {
146  throw new ApplicationException($this, $this->getResponse(),
147  ApplicationError::get('METHOD_NOT_ALLOWED', array_merge(
148  $clientInfo, array('method' => $requestMethod, 'route' => $requestPath))));
149  }
150 
151  // get request parameters from route
152  $pathRequestData = $route['parameters'];
153 
154  // get other request data
155  $requestData = array();
156  switch ($requestMethod) {
157  case 'GET':
158  $requestData = $_GET;
159  break;
160  case 'POST':
161  case 'PUT':
162  $requestData = $_POST;
163  break;
164  }
165 
166  // get controller/context/action triple
167  $controller = isset($requestData['controller']) ?
168  filter_var($requestData['controller'], FILTER_SANITIZE_STRING, FILTER_FLAG_STRIP_LOW) :
169  (isset($pathRequestData['controller']) ? $pathRequestData['controller'] : $controller);
170 
171  $context = isset($requestData['context']) ?
172  filter_var($requestData['context'], FILTER_SANITIZE_STRING, FILTER_FLAG_STRIP_LOW) :
173  (isset($pathRequestData['context']) ? $pathRequestData['context'] : $context);
174 
175  $action = isset($requestData['action']) ?
176  filter_var($requestData['action'], FILTER_SANITIZE_STRING, FILTER_FLAG_STRIP_LOW) :
177  (isset($pathRequestData['action']) ? $pathRequestData['action'] : $action);
178 
179  // setup request
180  $this->setSender($controller);
181  $this->setContext($context);
182  $this->setAction($action);
183  $this->setValues(array_merge($pathRequestData, $requestData));
184  }

221  /**
222  * Get all routes from the Routes configuration section that match the given
223  * request path
224  * @param $requestPath
225  * @return Array of arrays with keys 'numPathParameters', 'parameters', 'methods'
226  */
227  protected function getMatchingRoutes($requestPath) {
228  $matchingRoutes = array();
229  $defaultValuePattern = '([^/]+)';
230 
231  $config = ObjectFactory::getInstance('configuration');
232  if ($config->hasSection('routes')) {
233  $routes = $config->getSection('routes');
234  foreach ($routes as $route => $requestDef) {
235  // extract allowed http methods
236  $allowedMethods = null;
237  if (strpos($route, '/') !== false) {
238  list($methodStr, $route) = explode('/', $route, 2);
239  $allowedMethods = preg_split('/\s*,\s*/', trim($methodStr));
240  $route = '/'.trim($route);
241  }
242 
243  // extract parameters from route definition and prepare as regex pattern
244  $params = array();
245  $routePattern = preg_replace_callback('/\{([^\}]+)\}/', function ($match)
246  use($defaultValuePattern, &$params) {
247  // a variable may be either defined by {name} or by {name|pattern} where
248  // name is the variable's name and pattern is an optional regex pattern, the
249  // values should match
250  $paramParts = explode('|', $match[1], 2);
251  // add the variable name to the parameter list
252  $params[] = $paramParts[0];
253  // return the value match pattern (defaults to defaultValuePattern)
254  return sizeof($paramParts) > 1 ? '('.$paramParts[1].')' : $defaultValuePattern;
255  }, $route);
256 
257  // replace wildcard character and slashes
258  $routePattern = str_replace(array('*', '/'), array('.*', '\/'), $routePattern);
259 
260  // try to match the current request path
261  if (self::$logger->isDebugEnabled()) {
262  self::$logger->debug("Check path: ".$route." -> ".$routePattern);
263  }
264  $matches = array();
265  if (preg_match('/^'.$routePattern.'\/?$/', $requestPath, $matches)) {
266  if (self::$logger->isDebugEnabled()) {
267  self::$logger->debug("Match");
268  }
269  // ignore first match
270  array_shift($matches);
271 
272  // collect request variables
273  $requestParameters = array();
274 
275  // 1. path variables
276  for ($i=0, $count=sizeof($params); $i<$count; $i++) {
277  $requestParameters[$params[$i]] = isset($matches[$i]) ? $matches[$i] : null;
278  }
279 
280  // 2. parameters from route configuration (overriding path parameters)
281  $requestDefData = array();
282  parse_str($requestDef, $requestDefData);
283  $requestParameters = array_merge($requestParameters, $requestDefData);
284 
285  // store match
286  $matchingRoutes[] = array(
287  'route' => $route,
288  'numPathParameters' => (preg_match('/\*/', $route) ? PHP_INT_MAX : sizeof($params)),
289  'parameters' => $requestParameters,
290  'methods' => $allowedMethods
291  );
292  }
293  }
294  }
295  return $matchingRoutes;
296  }

298  /**
299  * Get the best matching route from the given list of routes
300  * @param $routes Array of route definitions as returned by getMatchingRoutes()
301  * @return Array with keys 'numPathParameters', 'parameters', 'methods'
302  */
303  protected function getBestRoute($routes) {
304  // order matching routes by number of parameters
305  $method = $this->getMethod();
306  usort($routes, function($a, $b) use ($method) {
307  $numParamsA = $a['numPathParameters'];
308  $numParamsB = $b['numPathParameters'];
309  if ($numParamsA == $numParamsB) {
310  $hasMethodA = in_array($method, $a['methods']);
311  $hasMethodB = in_array($method, $b['methods']);
312  return ($hasMethodA && !$hasMethodB) ? -1 :
313  ((!$hasMethodA && $hasMethodB) ? 1 : 0);
314  }
315  return ($numParamsA > $numParamsB) ? 1 : -1;
316  });
317 
318  if (self::$logger->isDebugEnabled()) {
319  self::$logger->debug("Ordered routes:");
320  self::$logger->debug($routes);
321  }
322  // return most specific route (minimum number of parameters)
323  return array_shift($routes);
324  }

326  /**
327  * Get all http headers
328  * @return Associative array
329  */
330  private static function getAllHeaders() {
331  $headers = array();
332  foreach ($_SERVER as $name => $value) {
333  if (substr($name, 0, 5) == 'HTTP_') {
334  $name = str_replace(' ', '-', ucwords(strtolower(str_replace('_', ' ', substr($name, 5)))));
335  $headers[$name] = $value;
336  }
337  else if ($name == "CONTENT_TYPE") {
338  $headers["Content-Type"] = $value;
339  }
340  else if ($name == "CONTENT_LENGTH") {
341  $headers["Content-Length"] = $value;
342  }
343  }
344  return $headers;
345  }

347  /**
348  * Fix request parameters (e.g. PHP replaces dots by underscore)
349  * Code from http://stackoverflow.com/questions/68651/get-php-to-stop-replacing-characters-in-get-or-post-arrays/18163411#18163411
350  * @param $target
351  * @param $source
352  * @param $keep
353  */
354  private static function fix(&$target, $source, $keep=false) {
355  if (!$source) {
356  return;
357  }
358  $keys = array();
359 
360  $source = preg_replace_callback(
361  '/
362  # Match at start of string or &
363  (?:^|(?<=&))
364  # Exclude cases where the period is in brackets, e.g. foo[bar.blarg]
365  [^=&\[]*
366  # Affected cases: periods and spaces
367  (?:\.|%20)
368  # Keep matching until assignment, next variable, end of string or
369  # start of an array
370  [^=&\[]*
371  /x',
372  function ($key) use (&$keys) {
373  $keys[] = $key = base64_encode(urldecode($key[0]));
374  return urlencode($key);
375  },
376  $source
377  );
378 
379  if (!$keep) {
380  $target = array();
381  }
382 
383  parse_str($source, $data);
384  foreach ($data as $key => $val) {
385  // Only unprocess encoded keys
386  if (!in_array($key, $keys)) {
387  $target[$key] = $val;
388  continue;
389  }
390 
391  $key = base64_decode($key);
392  $target[$key] = $val;
393 
394  if ($keep) {
395  // Keep a copy in the underscore key version
396  $key = preg_replace('/(\.| )/', '_', $key);
397  $target[$key] = $val;
398  }
399  }
400  }
