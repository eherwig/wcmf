73  /**
74  * @see PermissionManager::authorize()
75  */
76  public function authorize($resource, $context, $action, $login=null) {
77  // get authenticated user, if no user is given
78  if ($login == null) {
79  $login = $this->session->getAuthUser();
80  }
81  if (self::$logger->isDebugEnabled()) {
82  self::$logger->debug("Checking authorization for: '$resource?$context?$action' and user '".$login."'");
83  }
84 
85  // normalize resource to string
86  $resourceStr = ($resource instanceof ObjectId) ? $resource->__toString() : $resource;
87 
88  // determine the resource type and set entity type, oid and property if applicable
89  $resourceDesc = $this->parseResource($resourceStr);
90  $resourceType = $resourceDesc['resourceType'];
91  $oid = $resourceDesc['oid'];
92  $type = $resourceDesc['type'];
93  $oidProperty = $resourceDesc['oidProperty'];
94  $typeProperty = $resourceDesc['typeProperty'];
95  if (self::$logger->isDebugEnabled()) {
96  self::$logger->debug("Resource type: ".$resourceType);
97  }
98 
99  // proceed by authorizing type depending resource
100  // always start checking from most specific
101  switch ($resourceType) {
102  case (self::RESOURCE_TYPE_ENTITY_INSTANCE_PROPERTY):
103  $authorized = $this->authorizeAction($oidProperty, $oidProperty, $context, $action, $login);
104  if ($authorized === null) {
105  $authorized = $this->authorizeAction($oidProperty, $typeProperty, $context, $action, $login);
106  if ($authorized === null) {
107  $authorized = $this->authorizeAction($oidProperty, $oid, $context, $action, $login);
108  if ($authorized === null) {
109  $authorized = $this->authorizeAction($oidProperty, $type, $context, $action, $login);
110  }
111  }
112  }
113  break;
114 
115  case (self::RESOURCE_TYPE_ENTITY_INSTANCE):
116  $authorized = $this->authorizeAction($oid, $oid, $context, $action, $login);
117  if ($authorized === null) {
118  $authorized = $this->authorizeAction($oid, $type, $context, $action, $login);
119  }
120  break;
121 
122  case (self::RESOURCE_TYPE_ENTITY_TYPE_PROPERTY):
123  $authorized = $this->authorizeAction($typeProperty, $typeProperty, $context, $action, $login);
124  if ($authorized === null) {
125  $authorized = $this->authorizeAction($typeProperty, $type, $context, $action, $login);
126  }
127  break;
128 
129  case (self::RESOURCE_TYPE_ENTITY_TYPE_PROPERTY):
130  $authorized = $this->authorizeAction($type, $type, $context, $action, $login);
131  break;
132 
133  default:
134  $authorized = $this->authorizeAction($resourceStr, $resourceStr, $context, $action, $login);
135  break;
136  }
137 
138  // check parent entities in composite relations
139  if ($authorized === null && $resourceType == self::RESOURCE_TYPE_ENTITY_INSTANCE) {
140  if (self::$logger->isDebugEnabled()) {
141  self::$logger->debug("Check parent objects");
142  }
143  $mapper = $this->persistenceFacade->getMapper($type);
144  $parentRelations = $mapper->getRelations('parent');
145  if (sizeof($parentRelations) > 0) {
146 
147  $oidObj = ObjectId::parse($oid);
148  $tmpPerm = $this->addTempPermission($oidObj, $context, PersistenceAction::READ);
149  $object = $this->persistenceFacade->load($oidObj);
150  $this->removeTempPermission($tmpPerm);
151 
152  if ($object != null) {
153  foreach ($parentRelations as $parentRelation) {
154  if ($parentRelation->getThisAggregationKind() == 'composite') {
155  $parentType = $parentRelation->getOtherType();
156 
157  $tmpPerm = $this->addTempPermission($parentType, $context, PersistenceAction::READ);
158  $parents = $object->getValue($parentRelation->getOtherRole());
159  $this->removeTempPermission($tmpPerm);
160 
161  if ($parents != null) {
162  if (!$parentRelation->isMultiValued()) {
163  $parents = [$parents];
164  }
165  foreach ($parents as $parent) {
166  $authorized = $this->authorize($parent->getOID(), $context, $action);
167  if (!$authorized) {
168  break;
169  }
170  }
171  }
172  }
173  }
174  }
175  }
176  }
177 
178  if ($authorized === null) {
179  $authorized = $this->getDefaultPolicy($login);
180  }
181  if (self::$logger->isDebugEnabled()) {
182  self::$logger->debug("Result for $resource?$context?$action: ".(!$authorized ? "not " : "")."authorized");
183  }
184 
185  return $authorized;
186  }

188  /**
189  * Authorize a resource, context, action triple by using the permissions set
190  * on another resource (e.g. authorize an action on an entity instance base
191  * on the permissions defined for it's type).
192  * @param $requestedResource The resource string to authorize.
193  * @param $permissionResource The resource string to use for selecting permissions.
194  * @param $context The context in which the action takes place.
195  * @param $action The action to process.
196  * @param $login The login of the user to use for authorization
197  * @return Boolean or null if undefined
198  */
199  protected function authorizeAction($requestedResource, $permissionResource,
200  $context, $action, $login) {
201  if (self::$logger->isDebugEnabled()) {
202  self::$logger->debug("Authorizing $requestedResource?$context?$action ".
203  "using permissions of $permissionResource?$context?$action");
204  }
205  $authorized = null;
206 
207  // check temporary permissions
208  if ($this->hasTempPermission($permissionResource, $context, $action)) {
209  if (self::$logger->isDebugEnabled()) {
210  self::$logger->debug("Has temporary permission");
211  }
212  $authorized = true;
213  }
214  else {
215  // check other permissions
216  $permissions = $this->getPermissions($permissionResource, $context, $action);
217  if (self::$logger->isDebugEnabled()) {
218  self::$logger->debug("Permissions: ".StringUtil::getDump($permissions));
219  }
220  if ($permissions != null) {
221  // matching permissions found, check user roles
222  $authorized = $this->matchRoles($requestedResource, $permissions, $login);
223  }
224  }
225  if (self::$logger->isDebugEnabled()) {
226  self::$logger->debug("Result: ".(is_bool($authorized) ? ((!$authorized ? "not " : "")."authorized") : "not defined"));
227  }
228  return $authorized;
229  }

241  /**
242  * Get the resource type and parameters (as applicable) from a resource
243  * @param $resource The resource represented as string
244  * @return Associative array with keys
245  * 'resourceType' (one of the RESOURCE_TYPE_ constants),
246  * 'oid' (object id),
247  * 'type' (entity type),
248  * 'oidProperty' (object id with instance property),
249  * 'typeProperty' (type id with entity property)
250  */
251  protected function parseResource($resource) {
252  $resourceType = null;
253  $oid = null;
254  $type = null;
255  $oidProperty = null;
256  $typeProperty = null;
257  $extensionRemoved = preg_replace('/\.[^\.]*?$/', '', $resource);
258  if (($oidObj = ObjectId::parse($resource)) !== null) {
259  $resourceType = self::RESOURCE_TYPE_ENTITY_INSTANCE;
260  $oid = $resource;
261  $type = $oidObj->getType();
262  }
263  elseif (($oidObj = ObjectId::parse($extensionRemoved)) !== null) {
264  $resourceType = self::RESOURCE_TYPE_ENTITY_INSTANCE_PROPERTY;
265  $oid = $extensionRemoved;
266  $type = $oidObj->getType();
267  $oidProperty = $resource;
268  $typeProperty = $type.substr($resource, strlen($extensionRemoved));
269  }
270  elseif ($this->persistenceFacade->isKnownType($resource)) {
271  $resourceType = self::RESOURCE_TYPE_ENTITY_TYPE;
272  $type = $resource;
273  }
274  elseif ($this->persistenceFacade->isKnownType($extensionRemoved)) {
275  $resourceType = self::RESOURCE_TYPE_ENTITY_TYPE_PROPERTY;
276  $type = $extensionRemoved;
277  $typeProperty = $resource;
278  }
279  else {
280  // defaults to other
281  $resourceType = self::RESOURCE_TYPE_OTHER;
282  }
283  return [
284  'resourceType' => $resourceType,
285  'oid' => $oid,
286  'type' => $type,
287  'oidProperty' => $oidProperty,
288  'typeProperty' => $typeProperty
289  ];
290  }

292  /**
293  * Parse a permissions string and return an associative array with the keys
294  * 'default', 'allow', 'deny', where 'allow', 'deny' are arrays itself holding roles
295  * and 'default' is a boolean value derived from the wildcard policy (+* or -*).
296  * @param $value A role string (+*, +administrators, -guest, entries without '+' or '-'
297  * prefix default to allow rules).
298  * @return Associative array containing the permissions as an associative array with the keys
299  * 'default', 'allow', 'deny' or null, if value is empty
300  */
301  protected function deserializePermissions($value) {
302  if (strlen($value) == 0) {
303  return null;
304  }
305  $result = [
306  'default' => null,
307  'allow' => [],
308  'deny' => [],
309  ];
310 
311  $roleValues = explode(" ", $value);
312  foreach ($roleValues as $roleValue) {
313  $roleValue = trim($roleValue);
314  $matches = [];
315  preg_match('/^([+-]?)(.+)$/', $roleValue, $matches);
316  if (sizeof($matches) > 0) {
317  $prefix = $matches[1];
318  $role = $matches[2];
319  if ($role === '*') {
320  $result['default'] = $prefix == '-' ? false : true;
321  }
322  else {
323  if ($prefix === '-') {
324  $result['deny'][] = $role;
325  }
326  else {
327  // entries without '+' or '-' prefix default to allow rules
328  $result['allow'][] = $role;
329  }
330  }
331  }
332  }
333  // if no wildcard policy is defined, set default to false
334  if (!isset($result['default'])) {
335  $result['default'] = false;
336  }
337  return $result;
338  }

340  /**
341  * Convert an associative permissions array with keys 'default', 'allow', 'deny'
342  * into a string.
343  * @param $permissions Associative array with keys 'default', 'allow', 'deny',
344  * where 'allow', 'deny' are arrays itself holding roles and 'default' is a
345  * boolean value derived from the wildcard policy (+* or -*).
346  * @return A role string (+*, +administrators, -guest, entries without '+' or '-'
347  * prefix default to allow rules).
348  */
349  protected function serializePermissions($permissions) {
350  $result = $permissions['default'] === true ? PermissionManager::PERMISSION_MODIFIER_ALLOW.'* ' :
351  PermissionManager::PERMISSION_MODIFIER_DENY.'* ';
352  if (isset($permissions['allow'])) {
353  foreach ($permissions['allow'] as $role) {
354  $result .= PermissionManager::PERMISSION_MODIFIER_ALLOW.$role.' ';
355  }
356  }
357  if (isset($permissions['deny'])) {
358  foreach ($permissions['deny'] as $role) {
359  $result .= PermissionManager::PERMISSION_MODIFIER_DENY.$role.' ';
360  }
361  }
362  return trim($result);
363  }

365  /**
366  * Matches the roles of the user and the roles in the given permissions
367  * @param $resource The resource string to authorize.
368  * @param $permissions An array containing permissions as an associative array
369  * with the keys 'default', 'allow', 'deny', where 'allow', 'deny' are arrays
370  * itself holding roles and 'default' is a boolean value derived from the
371  * wildcard policy (+* or -*). 'allow' overwrites 'deny' overwrites 'default'
372  * @param $login the login of the user to match the roles for
373  * @return Boolean whether the user is authorized according to the permissions
374  */
375  protected function matchRoles($resource, $permissions, $login) {
376  if (self::$logger->isDebugEnabled()) {
377  self::$logger->debug("Matching roles for ".$login);
378  }
379  $user = $this->principalFactory->getUser($login, true);
380  if ($user != null) {
381  foreach (['allow' => true, 'deny' => false] as $key => $result) {
382  if (isset($permissions[$key])) {
383  foreach ($permissions[$key] as $role) {
384  if ($this->matchRole($user, $role, $resource)) {
385  if (self::$logger->isDebugEnabled()) {
386  self::$logger->debug($key." because of role ".$role);
387  }
388  return $result;
389  }
390  }
391  }
392  }
393  }
394  if (self::$logger->isDebugEnabled()) {
395  self::$logger->debug("Check default ".$permissions['default']);
396  }
397  return (isset($permissions['default']) ? $permissions['default'] : false);
398  }

400  /**
401  * Check if a user matches the role for a resource
402  * @param $user The user instance.
403  * @param $role The role name.
404  * @param $resource The resource string to authorize.
405  * @return Boolean
406  */
407  protected function matchRole(User $user, $role, $resource) {
408  $isDynamicRole = isset($this->dynamicRoles[$role]);
409  return (($isDynamicRole && $this->dynamicRoles[$role]->match($user, $resource) === true)
410  || (!$isDynamicRole && $user->hasRole($role)));
411  }

437  /**
438  * @see PermissionManager::hasTempPermission()
439  */
440  public function hasTempPermission($resource, $context, $action) {
441  if (sizeof($this->tempPermissions) == 0) {
442  return false;
443  }
444 
445  // check if the resource has a direct permission
446  $permissions = array_flip($this->tempPermissions);
447  $actionKey = ActionKey::createKey($resource, $context, $action);
448  if (!isset($permissions[$actionKey])) {
449  // if not and the resource belongs to an entity instance,
450  // we might have a permission for the type
451  $resourceDesc = $this->parseResource($resource);
452  switch ($resourceDesc['resourceType']) {
453  case self::RESOURCE_TYPE_ENTITY_INSTANCE:
454  $typeResource = $resourceDesc['type'];
455  break;
456  case self::RESOURCE_TYPE_ENTITY_INSTANCE_PROPERTY:
457  $typeResource = $resourceDesc['typeProperty'];
458  break;
459  default:
460  $typeResource = null;
461  }
462  // set alternative action key
463  if ($typeResource != null) {
464  $actionKey = ActionKey::createKey($typeResource, $context, $action);
465  }
466  }
467  return isset($permissions[$actionKey]);
468  }
