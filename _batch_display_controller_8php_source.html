27 /**
28  * BatchDisplayController is used to load a tree of Node instances recursivly and
29  * return them in lists of a given size. The reconstruction of the tree must be
30  * handled by the client.
31  *
32  * The controller supports the following actions:
33  *
34  * <div class="controller-action">
35  * <div> __Action__ _default_ </div>
36  * <div>
37  * Load the Nodes.
38  * | Parameter | Description
39  * |---------------------------|-------------------------
40  * | _in_ / _out_ `oid` | The object id of the Node to start loading from
41  * | _in_ `translateValues` | Boolean whether list values should be translated to their display values (optional, default: _true_)
42  * | _in_ `nodesPerCall` | The number of Node instances to load in one call (default: 50)
43  * | _out_ `list` | Array of Node instances
44  * </div>
45  * </div>
46  *
47  * For additional actions and parameters see [BatchController actions](@ref BatchController).
48  *
49  * @author ingo herwig <ingo@wemove.com>
50  */
51 class BatchDisplayController extends BatchController {
52 
53  // session name constants
54  private $REQUEST = 'BatchDisplayController.request';
55  private $REGISTRY = 'BatchDisplayController.registry';
56  private $ITERATOR_ID = 'BatchDisplayController.iteratorid';
57 
58  // default values, maybe overriden by corresponding request values (see above)
59  private $_NODES_PER_CALL = 50;
60 
61  /**
62  * @see Controller::initialize()
63  */
64  public function initialize(Request $request, Response $response) {
65  parent::initialize($request, $response);
66 
67  // initialize controller
68  if ($request->getAction() != 'continue') {
69  $session = ObjectFactory::getInstance('session');
70 
71  // set defaults
72  if (!$request->hasValue('nodesPerCall')) {
73  $request->setValue('nodesPerCall', $this->_NODES_PER_CALL);
74  }
75  if (!$request->hasValue('translateValues')) {
76  $request->setValue('translateValues', true);
77  }
78 
79  // store request in session
80  $session->set($this->REQUEST, $request);
81  $reg = array();
82  $session->set($this->REGISTRY, $reg);
83  }
84  }

86  /**
87  * @see Controller::validate()
88  */
89  protected function validate() {
90  $request = $this->getRequest();
91  $response = $this->getResponse();
92  if ($request->getAction() != 'continue') {
93  $oid = ObjectId::parse($request->getValue('oid'));
94  if(!$oid) {
95  $response->addError(ApplicationError::get('OID_INVALID',
96  array('invalidOids' => array($request->getValue('oid')))));
97  return false;
98  }
99  if (!$this->checkLanguageParameter()) {
100  return false;
101  }
102  }
103  // do default validation
104  return parent::validate();
105  }

107  /**
108  * @see BatchController::getWorkPackage()
109  */
110  protected function getWorkPackage($number) {
111  if ($number == 0) {
112  return array('name' => Message::get('Loading'), 'size' => 1, 'oids' => array(1), 'callback' => 'startProcess');
113  }
114  else {
115  return null;
116  }
117  }

119  /**
120  * Initialize the iterator (oids parameter will be ignored)
121  * @param $oids The oids to process
122  */
123  protected function startProcess($oids) {
124  $session = ObjectFactory::getInstance('session');
125 
126  // restore the request from session
127  $request = $session->get($this->REQUEST);
128  $nodeOID = ObjectId::parse($request->getValue('oid'));
129 
130  // do the action
131  $iterator = new PersistentIterator($nodeOID);
132  $iteratorID = $iterator->save();
133  $session->set($this->ITERATOR_ID, $iteratorID);
134 
135  // display the first node in order to reduce the number of calls
136  $this->loadNode($iterator->current());
137 
138  $iterator->next();
139 
140  // proceed if nodes are left
141  if ($iterator->valid()) {
142  $iteratorID = $iterator->save();
143  $session->set($this->ITERATOR_ID, $iteratorID);
144 
145  $name = Message::get('Loading tree: continue with %0%', array($iterator->current()));
146  $this->addWorkPackage($name, 1, array(null), 'loadNodes');
147  }
148  else {
149  // set the result and finish
150  $this->endProcess();
151  }
152  }
153 
154  /**
155  * Load nodes provided by the persisted iterator (oids parameter will be ignored)
156  * @param $oids The oids to process
157  */
158  protected function loadNodes($oids) {
159  $session = ObjectFactory::getInstance('session');
160 
161  // restore the request from session
162  $request = $session->get($this->REQUEST);
163 
164  // check for iterator in session
165  $iterator = null;
166  $iteratorID = $session->get($this->ITERATOR_ID);
167  if ($iteratorID != null) {
168  $iterator = PersistentIterator::load($iteratorID);
169  }
170 
171  // no iterator, finish
172  if ($iterator == null) {
173  // set the result and finish
174  $this->endProcess();
175  }
176 
177  // process _NODES_PER_CALL nodes
178  $counter = 0;
179  while ($iterator->valid() && $counter < $request->getValue('nodesPerCall')) {
180  $currentOID = $iterator->current();
181  $this->loadNode($currentOID);
182 
183  $iterator->next();
184  $counter++;
185  }
186 
187  // decide what to do next
188  if ($iterator->valid()) {
189  // proceed with current iterator
190  $iteratorID = $iterator->save();
191  $session->set($this->ITERATOR_ID, $iteratorID);
192 
193  $name = Message::get('Loading tree: continue with %0%', array($iterator->current()));
194  $this->addWorkPackage($name, 1, array(null), 'loadNodes');
195  }
196  else {
197  // set the result and finish
198  $this->endProcess();
199  }
200  }
201 
202  /**
203  * Finish the process and set the result
204  */
205  protected function endProcess() {
206  $session = ObjectFactory::getInstance('session');
207 
208  // clear session variables
209  $tmp = null;
210  $session->set($this->REQUEST, $tmp);
211  $session->set($this->REGISTRY, $tmp);
212  $session->set($this->ITERATOR_ID, $tmp);
213  }
214 
215  /**
216  * Load the node with the given object id and assign it to the response.
217  * @param $oid The oid of the node to copy
218  */
219  protected function loadNode(ObjectId $oid) {
220  // check if we already loaded the node
221  if ($this->isRegistered($oid)) {
222  return;
223  }
224  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
225  $session = ObjectFactory::getInstance('session');
226 
227  // restore the request from session
228  $request = $session->get($this->REQUEST);
229 
230  // load the node
231  $node = $persistenceFacade->load($oid);
232  if ($node == null) {
233  throw new PersistenceException("Can't load node '".$oid."'");
234  }
235 
236  // translate all nodes to the requested language if requested
237  if ($this->isLocalizedRequest()) {
238  $localization = ObjectFactory::getInstance('localization');
239  $node = $localization->loadTranslation($node, $request->getValue('language'), true, true);
240  }
241 
242  // translate values if requested
243  if ($request->getBooleanValue('translateValues')) {
244  $nodes = array($node);
245  if ($this->isLocalizedRequest()) {
246  NodeUtil::translateValues($nodes, $request->getValue('language'));
247  }
248  else {
249  NodeUtil::translateValues($nodes);
250  }
251  }
252 
253  // assign it to the response
254  $this->addNodeToResponse($node);
255 
256  $this->register($oid);
257 
258  if (Log::isInfoEnabled(__CLASS__)) {
259  Log::info("Loaded: ".$node->getOID(), __CLASS__);
260  }
261  if (Log::isDebugEnabled(__CLASS__)) {
262  Log::debug($node->toString(), __CLASS__);
263  }
264  }
265 
266  /**
267  * Register an object id in the registry
268  * @param $oid The object id to register
269  */
270  protected function register(ObjectId $oid) {
271  $session = ObjectFactory::getInstance('session');
272  $registry = $session->get($this->REGISTRY);
273  $registry[] = $oid;
274  $session->set($this->REGISTRY, $registry);
275  }

277  /**
278  * Check if an object id is registered in the registry
279  * @param $oid The object id to check
280  * @return Boolean whether the oid is registered or not
281  */
282  protected function isRegistered(ObjectId $oid) {
283  $session = ObjectFactory::getInstance('session');
284  $registry = $session->get($this->REGISTRY);
285 
286  return in_array($oid, $registry);
287  }

289  /**
290  * Add a given node to the list variable of the response
291  * @param $node A reference to the node to add
292  */
293  protected function addNodeToResponse(Node $node) {
294  $response = $this->getResponse();
295  if (!$response->hasValue('list')) {
296  $objects = array();
297  $response->setValue('list', $objects);
298  }
299 
300  $objects = $response->getValue('list');
301  $objects[] = $node;
302  $response->setValue('list', $objects);
303  }
