18 /**
19  * JsonFormat realizes the JSON request/response format. All data will
20  * be serialized using the json_encode method except for Nodes.
21  * Nodes are serialized into an array before encoding (see JsonFormat::serializeValue)
22  * using the NodeSerializer class.
23  * On serialization the data will be outputted directly using the print command.
24  *
25  * JsonFormat collects the response data from all executed controllers
26  * into one response array and returns it all at once at the end of
27  * script execution. This prevents from having multiple junks of json
28  * from each controller response that can't be decoded by clients.
29  *
30  * @author ingo herwig <ingo@wemove.com>
31  */
32 class JsonFormat extends HierarchicalFormat {
33 
34  private static $_jsonData = array();
35  private static $_jsonUsed = false;
36 
37  protected $_serializer = null;
38 
39  public static function printJSONResult() {
40  if (self::$_jsonUsed) {
41  $data = self::$_jsonData;
42  if ($data !== null) {
43  $encoded = json_encode($data);
44  if (Log::isDebugEnabled('JsonFormat')) {
45  Log::debug($data, 'JsonFormat');
46  Log::debug($encoded, 'JsonFormat');
47  }
48  print($encoded);
49  }
50  }
51  }
52 
53  /**
54  * @see Format::getMimeType()
55  */
56  public function getMimeType() {
57  return 'application/json';
58  }
59 
60  /**
61  * Set the NodeSerializer instance to use
62  * @param $serializer NodeSerializer
63  */
64  public function setSerializer(NodeSerializer $serializer) {
65  $this->_serializer = $serializer;
66  }
67 
68  /**
69  * @see HierarchicalFormat::afterSerialize()
70  */
71  protected function afterSerialize($values) {
72  // TODO: check if merging is required for multiple actions
73  /*
74  // merge data into global data array
75  // new values override old
76  self::$_jsonData = array_merge(self::$_jsonData, $data);
77  */
78  self::$_jsonData = $values;
79  self::$_jsonUsed = true;
80  return $values;
81  }
82

85  */
86  protected function isSerializedNode($value) {
87  if ($this->_serializer == null) {
88  throw new ConfigurationException("The serializer is not set.");
89  }
90  return $this->_serializer->isSerializedNode($value);
91  }
92 
93  /**
94  * @see HierarchicalFormat::serializeNode()
95  */
96  protected function serializeNode($value) {
97  if ($this->_serializer == null) {
98  throw new ConfigurationException("The serializer is not set.");
99  }
100  $node = $this->_serializer->serializeNode($value);
101  return $node;
102  }
103 
104  /**
105  * @see HierarchicalFormat::deserializeNode()
106  */
107  protected function deserializeNode($value) {
108  if ($this->_serializer == null) {
109  throw new ConfigurationException("The serializer is not set.");
110  }
111  $result = $this->_serializer->deserializeNode($value);
112  return $result;
113  }
114 }
