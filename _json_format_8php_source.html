18 /**
19  * JsonFormat realizes the JSON request/response format. All data will
20  * be serialized using the json_encode method except for Nodes.
21  * Nodes are serialized into an array before encoding (see JsonFormat::serializeValue)
22  * using the NodeSerializer class.
23  * On serialization the data will be outputted directly using the print command.
24  *
25  * JsonFormat collects the response data from all executed controllers
26  * into one response array and returns it all at once at the end of
27  * script execution. This prevents from having multiple junks of json
28  * from each controller response that can't be decoded by clients.
29  *
30  * @author ingo herwig <ingo@wemove.com>
31  */
32 class JsonFormat extends HierarchicalFormat {

51  public static function printJSONResult() {
52  if (self::$jsonUsed) {
53  $data = self::$jsonData;
54  if ($data !== null) {
55  $encoded = json_encode($data);
56  if (self::$logger->isDebugEnabled(__CLASS__)) {
57  self::$logger->debug($data);

60  print($encoded);
61  }
62  }
63  }
