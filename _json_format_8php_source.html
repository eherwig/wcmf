19 /**
20  * JsonFormat implements the JSON request/response format. All data will
21  * be serialized using the json_encode method except for Nodes.
22  * Nodes are serialized into an array before encoding (see JsonFormat::serializeValue)
23  * using the NodeSerializer class.
24  * On serialization the data will be outputted directly using the print command.
25  *
26  * JsonFormat collects the response data from all executed controllers
27  * into one response array and returns it all at once at the end of
28  * script execution. This prevents from having multiple chunks of JSON
29  * from each controller response that can't be decoded by clients.
30  *
31  * @author ingo herwig <ingo@wemove.com>
32  */
33 class JsonFormat extends HierarchicalFormat {

90  */
91  protected function afterSerialize(Response $response) {
92  $values = $response->getValues();
93 
94  $cacheId = $response->getCacheId();
95  $caching = strlen($cacheId) > 0;
96  if (!$caching || !$this->isCached($response)) {
97  // encode data
98  $encoded = json_encode($values);
99  if (self::$logger->isDebugEnabled()) {
100  self::$logger->debug($values);
101  self::$logger->debug($encoded);
102  }
103 
104  // cache result
105  if ($caching) {
106  $this->cache->put($this->getCacheSection($response), $cacheId, $encoded, $response->getCacheLifetime());
107  }
108  }
109  else {
110  $encoded = $this->cache->get($this->getCacheSection($response), $cacheId);
111  }
112 
113  // render
114  print($encoded);
115 
116  return $values;
117  }
