69  /**
70  * @see Controller::doExecute()
71  */
72  protected function doExecute() {
73  $session = $this->getSession();
74  $permissionManager = $this->getPermissionManager();
75  $persistenceFacade = $this->getPersistenceFacade();
76  $request = $this->getRequest();
77  $response = $this->getResponse();
78 
79  // change password
80 
81  // load model
82  $authUser = $this->principalFactory->getUser($session->getAuthUser());
83  if ($authUser) {
84  // add permissions for this operation
85  $oidStr = $authUser->getOID()->__toString();
86  $tmpPerm1 = $permissionManager->addTempPermission($oidStr, '', PersistenceAction::READ);
87  $tmpPerm2 = $permissionManager->addTempPermission($oidStr, '', PersistenceAction::UPDATE);
88 
89  // start the persistence transaction
90  $transaction = $persistenceFacade->getTransaction();
91  $transaction->begin();
92  try {
93  $this->changePassword($authUser, $request->getValue('oldpassword'),
94  $request->getValue('newpassword1'), $request->getValue('newpassword2'));
95  $transaction->commit();
96  }
97  catch(\Exception $ex) {
98  $response->addError(ApplicationError::fromException($ex));
99  $transaction->rollback();
100  }
101  // remove temporary permissions
102  $permissionManager->removeTempPermission($tmpPerm1);
103  $permissionManager->removeTempPermission($tmpPerm2);
104  }
105  // success
106  $response->setAction('ok');
107  }
108 
109  /**
110  * Change a users password.
111  * @param $user The User instance
112  * @param $oldPassword The old password of the user
113  * @param $newPassword The new password for the user
114  * @param $newPasswordRepeated The new password of the user again
115  */
116  protected function changePassword(User $user, $oldPassword, $newPassword, $newPasswordRepeated) {
117  $message = $this->getMessage();
118  // check old password
119  if (!$user->verifyPassword($oldPassword)) {
120  throw new IllegalArgumentException($message->getText("The old password is incorrect"));
121  }
122  if (strlen($newPassword) == 0) {
123  throw new IllegalArgumentException($message->getText("The password can't be empty"));
124  }
125  if ($newPassword != $newPasswordRepeated) {
126  throw new IllegalArgumentException($message->getText("The given passwords don't match"));
127  }
128  // set password
129  $user->setPassword($newPassword);
130  }
