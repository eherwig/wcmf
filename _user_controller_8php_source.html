44  /**
45  * @see Controller::doExecute()
46  */
47  protected function doExecute() {
48  $permissionManager = ObjectFactory::getInstance('permissionManager');
49  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
50  $request = $this->getRequest();
51  $response = $this->getResponse();
52 
53  // change password
54 
55  // load model
56  $authUser = $permissionManager->getAuthUser();
57 
58  // add permissions for this operation
59  $oidStr = $authUser->getOID()->__toString();
60  $permissionManager->addTempPermission($oidStr, '', PersistenceAction::READ);
61  $permissionManager->addTempPermission($oidStr, '', PersistenceAction::UPDATE);
62 
63  // start the persistence transaction
64  $transaction = $persistenceFacade->getTransaction();
65  $transaction->begin();
66  try {
67  $this->changePassword($authUser, $request->getValue('oldpassword'),
68  $request->getValue('newpassword1'), $request->getValue('newpassword2'));
69  $transaction->commit();
70  }
71  catch(\Exception $ex) {
72  $response->addError(ApplicationError::fromException($ex));
73  $transaction->rollback();
74  }
75  // remove temporary permissions
76  $permissionManager->clearTempPermissions();
77 
78  // success
79  $response->setAction('ok');
80  }
81 
82  /**
83  * Change a users password.
84  * @param $user The User instance
85  * @param $oldPassword The old password of the user
86  * @param $newPassword The new password for the user
87  * @param $newPasswordRepeated The new password of the user again
88  */
89  public function changePassword(User $user, $oldPassword, $newPassword, $newPasswordRepeated) {
90  // check old password
91  if (!$user->verifyPassword($oldPassword, $user->getPassword())) {
92  throw new IllegalArgumentException(Message::get("The old password is incorrect"));
93  }
94  if (strlen($newPassword) == 0) {
95  throw new IllegalArgumentException(Message::get("The password can't be empty"));
96  }
97  if ($newPassword != $newPasswordRepeated) {
98  throw new IllegalArgumentException(Message::get("The given passwords don't match"));
99  }
100  // set password
101  $user->setPassword($newPassword);
102  }
103 }
