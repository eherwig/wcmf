25 /**
26  * Default ActionMapper implementation.
27  *
28  * @author ingo herwig <ingo@wemove.com>
29  */
30 class DefaultActionMapper implements ActionMapper {
31 
32  private $_lastResponses = array();
33 
34  /**
35  * @see ActionMapper::processAction()
36  */
37  public function processAction(Request $request) {
38  $isDebugEnabled = Log::isDebugEnabled(__CLASS__);
39 
40  $eventManager = ObjectFactory::getInstance('eventManager');
41  $eventManager->dispatch(ApplicationEvent::NAME, new ApplicationEvent(
42  ApplicationEvent::BEFORE_ROUTE_ACTION, $request));
43  $actionKeyProvider = new ConfigActionKeyProvider();
44  $actionKeyProvider->setConfigSection('actionmapping');
45 
46  $referrer = $request->getSender();
47  $context = $request->getContext();
48  $action = $request->getAction();
49  $response = ObjectFactory::getInstance('response');
50  $response->setSender($referrer);
51  $response->setContext($context);
52  $response->setFormat($request->getResponseFormat());
53 
54  $config = ObjectFactory::getConfigurationInstance();
55  $permissionManager = ObjectFactory::getInstance('permissionManager');
56 
57  // check authorization for controller/context/action triple
58  if (!$permissionManager->authorize($referrer, $context, $action)) {
59  $authUser = $permissionManager->getAuthUser();
60  if ($authUser instanceof AnonymousUser) {
61  Log::error("Session invalid. The request was: ".$request->__toString(), __CLASS__);
62  throw new ApplicationException($request, $response, ApplicationError::get('SESSION_INVALID'));
63  }
64  else {
65  $login = $authUser->getLogin();
66  Log::error("Authorization failed for '".$referrer.'?'.$context.'?'.$action."' user '".$login."'", __CLASS__);
67  throw new ApplicationException($request, $response, ApplicationError::get('PERMISSION_DENIED'));
68  }
69  }
70 
71  // get best matching action key from inifile
72  $actionKey = ActionKey::getBestMatch($actionKeyProvider, $referrer, $context, $action);
73 
74  if ($isDebugEnabled) {
75  Log::debug($referrer."?".$context."?".$action.' -> '.$actionKey, __CLASS__);
76  }
77 
78  $controllerClass = null;
79  if (strlen($actionKey) == 0) {
80  // return last response, if action key is not defined
81  $lastResponse = array_pop($this->_lastResponses);
82  return $lastResponse;
83  }
84  else {
85  // get next controller
86  $controllerClass = $config->getValue($actionKey, 'actionmapping');
87  }
88  if (strlen($controllerClass) == 0) {
89  throw new ApplicationException($request, $response, "No controller found for best action key ".$actionKey.". Request was $referrer?$context?$action");
90  }
91 
92  // instantiate controller
93  $controllerObj = new $controllerClass();
94 
95  // everything is right in place, start processing
96  if ($isDebugEnabled) {
97  Log::debug("Request: ".$request->__toString(), __CLASS__);
98  }
99  Formatter::deserialize($request);
100 
101  // initialize controller
102  if ($isDebugEnabled) {
103  Log::debug("Execute ".$controllerClass." with request: ".$request->__toString(), __CLASS__);
104  }
105  $eventManager->dispatch(ApplicationEvent::NAME, new ApplicationEvent(
106  ApplicationEvent::BEFORE_INITIALIZE_CONTROLLER, $request, $response, $controllerObj));
107  $controllerObj->initialize($request, $response);
108 
109  // execute controller
110  $eventManager->dispatch(ApplicationEvent::NAME, new ApplicationEvent(
111  ApplicationEvent::BEFORE_EXECUTE_CONTROLLER, $request, $response, $controllerObj));
112  $controllerObj->execute();
113  $eventManager->dispatch(ApplicationEvent::NAME, new ApplicationEvent(
114  ApplicationEvent::AFTER_EXECUTE_CONTROLLER, $request, $response, $controllerObj));
115 
116  Formatter::serialize($response);
117 
118  // check if an action key exists for the return action
119  $nextActionKey = ActionKey::getBestMatch($actionKeyProvider, $controllerClass,
120  $response->getContext(), $response->getAction());
121  if ($isDebugEnabled) {
122  Log::debug("Next action key: ".$nextActionKey, __CLASS__);
123  }
124 
125  if (strlen($nextActionKey) == 0) {
126  if ($isDebugEnabled) {
127  Log::debug("Terminate", __CLASS__);
128  }
129  // stop processing
130  return $response;
131  }
132  else {
133  if ($isDebugEnabled) {
134  Log::debug("Processing next action", __CLASS__);
135  }
136  // store last response
137  $this->_lastResponses[] = $response;
138 
139  // proceed based on the result
140  $nextRequest = ObjectFactory::getInstance('request');
141  $nextRequest->setSender($controllerClass);
142  $nextRequest->setContext($response->getContext());
143  $nextRequest->setAction($response->getAction());
144  $nextRequest->setFormat($response->getFormat());
145  $nextRequest->setValues($response->getValues());
146  $nextRequest->setErrors($response->getErrors());
147  $nextRequest->setResponseFormat($request->getResponseFormat());
148  $this->processAction($nextRequest);
149  }
150  }
151 
152  /**
153  * @see ActionMapper::reset()
154  */
155  public function reset() {
156  $this->_lastResponses = array();
157  }
158 }
