23 /**
24  * CSVExportController exports instances of one type into a CSV file. It uses
25  * the `fputcsv` function of PHP with the default values for delimiter, enclosing
26  * and escape character.
27  *
28  * The controller supports the following actions:
29  *
30  * <div class="controller-action">
31  * <div> __Action__ _default_ </div>
32  * <div>
33  * Initiate the export.
34  * | Parameter | Description
35  * |------------------------|-------------------------
36  * | _in_ `docFile` | The name of the file to write to (path relative to script main location) (default: 'export.csv')
37  * | _in_ `className` | The entity type to export instances of
38  * | _in_ `sortFieldName` | The field name to sort the list by. Must be one of the fields of the type selected by the className parameter. If omitted, the sorting is undefined (optional)
39  * | _in_ `sortDirection` | The direction to sort the list. Must be either _asc_ for ascending or _desc_ for descending (optional, default: _asc_)
40  * | _in_ `query` | A query condition encoded in RQL to be used with StringQuery::setRQLConditionString()
41  * | _in_ `nodesPerCall` | The number of nodes to process in one call (default: 50)
42  * </div>
43  * </div>
44  *
45  * For additional actions and parameters see [BatchController actions](@ref BatchController).
46  *
47  * @author ingo herwig <ingo@wemove.com>
48  */
49 class CSVExportController extends BatchController {

51  // default values, maybe overriden by corresponding request values (see above)
52  private $DOCFILE = "export.csv";
53  private $NODES_PER_CALL = 50;

55  /**
56  * @see Controller::initialize()
57  */
58  public function initialize(Request $request, Response $response) {
59  // initialize controller
60  if ($request->getAction() != 'continue') {
61  // set defaults (will be stored with first request)
62  if (!$request->hasValue('docFile')) {
63  $request->setValue('docFile', $this->DOCFILE);
64  }
65  if (!$request->hasValue('nodesPerCall')) {
66  $request->setValue('nodesPerCall', $this->NODES_PER_CALL);
67  }
68 
69  // set the cache section and directory for the download file
70  $config = $this->getConfiguration();
71  $cacheBaseDir = WCMF_BASE.$config->getValue('cacheDir', 'StaticCache');
72  $cacheSection = 'csv-export-'.uniqid().'/cache';
73  $downloadDir = $cacheBaseDir.dirname($cacheSection).'/';
74  FileUtil::mkdirRec($downloadDir);
75  $request->setValue('cacheSection', $cacheSection);
76  $request->setValue('downloadFile', $downloadDir.$request->getValue('docFile'));
77  }
78  // initialize parent controller after default request values are set
79  parent::initialize($request, $response);
80  }

82  /**
83  * @see Controller::validate()
84  */
85  protected function validate() {
86  $request = $this->getRequest();
87  $response = $this->getResponse();
88  if ($request->getAction() != 'continue') {
89  if (!$request->hasValue('className') ||
90  !$this->getPersistenceFacade()->isKnownType($request->getValue('className')))
91  {
92  $response->addError(ApplicationError::get('PARAMETER_INVALID',
93  ['invalidParameters' => ['className']]));
94  return false;
95  }
96  // check for permission to read instances of className
97  if (!$this->getPermissionManager()->authorize($request->getValue('className'), '', PersistenceAction::READ)) {
98  $response->addError(ApplicationError::get('PERMISSION_DENIED'));
99  return false;
100  }
101  if ($request->hasValue('sortFieldName') &&
102  !$this->getPersistenceFacade()->getMapper($request->getValue('className'))->hasAttribute($request->hasValue('sortFieldName'))) {
103  $response->addError(ApplicationError::get('SORT_FIELD_UNKNOWN'));
104  return false;
105  }
106  if ($request->hasValue('sortDirection')) {
107  $sortDirection = $request->getValue('sortDirection');
108  if (strtolower($sortDirection) != 'asc' && strtolower($sortDirection) != 'desc') {
109  $response->addError(ApplicationError::get('SORT_DIRECTION_UNKNOWN'));
110  return false;
111  }
112  }
113  }
114  // do default validation
115  return parent::validate();
116  }

135  /**
136  * Initialize the CSV export (object ids parameter will be ignored)
137  * @param $oids The object ids to process
138  * @note This is a callback method called on a matching work package, see BatchController::addWorkPackage()
139  */
140  protected function initExport($oids) {
141  $persistenceFacade = $this->getPersistenceFacade();
142  $message = $this->getMessage();
143 
144  // get document definition
145  $docFile = $this->getDownloadFile();
146  $type = $this->getRequestValue('className');
147 
148  // delete export file
149  if (file_exists($docFile)) {
150  unlink($docFile);
151  }
152 
153  // get the query
154  $queryTerm = urldecode($this->getRequestValue('query'));
155 
156  // add sort term
157  $sortArray = null;
158  $orderBy = $this->getRequestValue('sortFieldName');
159  if (strlen($orderBy) > 0) {
160  $sortArray = [$orderBy." ".$this->getRequestValue('sortDirection')];
161  }
162 
163  // get object ids of all nodes to export
164  $query = new StringQuery($type);
165  $query->setRQLConditionString($queryTerm);
166  $oids = $query->execute(false, $sortArray);
167 
168  // get csv columns
169  $names = [];
170  $mapper = $persistenceFacade->getMapper($type);
171  foreach($mapper->getAttributes() as $attribute) {
172  $names[] = $attribute->getName();
173  }
174 
175  // initialize export file
176  $fileHandle = fopen($docFile, "a");
177  fputcsv($fileHandle, $names);
178  fclose($fileHandle);
179 
180  // create work packages for nodes
181  $nodesPerCall = $this->getRequestValue('nodesPerCall');
182  $this->addWorkPackage(
183  $message->getText('Exporting %0%', [$type]),
184  $nodesPerCall, $oids, 'exportNodes');
185  }
186 
187  /**
188  * Serialize all Nodes with given object ids to CSV
189  * @param $oids The object ids to process
190  * @note This is a callback method called on a matching work package, see BatchController::addWorkPackage()
191  */
192  protected function exportNodes($oids) {
193  $persistenceFacade = $this->getPersistenceFacade();
194  $permissionManager = $this->getPermissionManager();
195 
196  // get document definition
197  $docFile = $this->getDownloadFile();
198  $type = $this->getRequestValue('className');
199 
200  $mapper = $persistenceFacade->getMapper($type);
201  $attributes = $mapper->getAttributes();
202 
203  // process nodes
204  $fileHandle = fopen($docFile, "a");
205  foreach ($oids as $oid) {
206  if ($permissionManager->authorize($oid, '', PersistenceAction::READ)) {
207  $node = $persistenceFacade->load($oid);
208  $values = [];
209  foreach ($attributes as $attribute) {
210  $inputType = $attribute->getInputType();
211  $values[] = ValueListProvider::translateValue(
212  $node->getValue($attribute->getName()), $inputType);
213  }
214  fputcsv($fileHandle, $values);
215  }
216  }
217  fclose($fileHandle);
218  }
219 
220  /**
221  * @see BatchController::cleanup()
222  */
223  protected function cleanup() {
224  $downloadDir = dirname($this->getRequestValue('downloadFile'));
225  FileUtil::emptyDir($downloadDir);
226  rmdir($downloadDir);
227  parent::cleanup();
228  }
