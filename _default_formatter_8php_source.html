50  foreach ($this->formats as $name => $instance) {
51  $firstFormat = $firstFormat == null ? $name : $firstFormat;
52  $formatMimeType = $instance->getMimeType();
53  if (strlen($formatMimeType) > 0 && strpos($mimeType, $formatMimeType) !== false) {
54  return $name;
55  }
56  }

77  /**
78  * @see Formatter::serialize()
79  */
80  public function serialize(Response $response) {
81  // handle caching
82  $responseSent = false;
83  $cacheId = $response->getCacheId();
84  if ($cacheId != null) {
85  // get the caching request headers
86  $request = $response->getRequest();
87  $ifModifiedSinceHeader = $request->hasHeader("If-Modified-Since") ?
88  $request->getHeader("If-Modified-Since") : false;
89  $etagHeader = $request->hasHeader("If-None-Match") ?
90  trim($request->getHeader("If-None-Match")) : false;
91 
92  // get caching parameters from response
93  $cacheDate = $response->getCacheDate();
94  $lastModified = $cacheDate !== null ? $cacheDate : new \DateTime();
95  $lastModifiedTs = $lastModified->getTimestamp();
96  $etag = md5($lastModifiedTs.$cacheId);
97  $maxAge = $response->getCacheLifetime() !== null ? $response->getCacheLifetime() : 31536000;
98 
99  // send caching headers
100  $response->setHeader("Cache-Control", "public, max-age=".$maxAge);
101  $response->setHeader("Last-Modified", $lastModified->format("D, d M Y H:i:s")." GMT");
102  $response->setHeader("ETag", $etag);
103 
104  // check if page has changed and send 304 if not
105  if (($ifModifiedSinceHeader !== false && strtotime($ifModifiedSinceHeader) == $lastModifiedTs) ||
106  $etagHeader == $etag) {
107  // client has current response already
108  $response->setStatus(304);
109  $responseSent = true;
110  }
111  }
112 
113  // remove unwanted headers
114  $response->setHeader('X-Powered-By', null);
115  $response->setHeader('Expires', null);
116 
117  // delegate serialization to the response format
118  if (!$responseSent) {
119  $formatName = $response->getFormat();
120  if (strlen($formatName) == 0) {
121  // the format must be given!
122  throw new ConfigurationException("No response format defined for ".$response->__toString());
123  }
124  $format = $this->getFormat($formatName);
125 
126  // send headers
127  if (!headers_sent()) {
128  foreach ($format->getResponseHeaders($response) as $name => $value) {
129  if (strlen($value) > 0) {
130  header($name.": ".$value);
131  }
132  else {
133  header_remove($name);
134  }
135  }
136  }
137 
138  // send body
139  $format->serialize($response);
140  }
141 
142  http_response_code($response->getStatus());
143  }
