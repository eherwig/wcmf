59  protected function transformRequest($request) {
60  // transform range header into paging parameters
61  if ($request->hasHeader('Range')) {
62  if (preg_match('/^items=([\-]?[0-9]+)-([\-]?[0-9]+)$/', $request->getHeader('Range'), $matches)) {
63  $offset = intval($matches[1]);
64  $limit = intval($matches[2])-$offset+1;
65  $request->setValue('offset', $offset);
66  $request->setValue('limit', $limit);
67  }
68  }
69 
70  // transform position header into reference oid parameter for sorting
71  if ($request->hasHeader('Position')) {
72  $position = $request->getHeader('Position');
73  if ($position == 'last') {
74  $referenceOid = 'ORDER_BOTTOM';
75  }
76  else {
77  list($ignore, $orderReferenceIdStr) = preg_split('/ /', $position);
78  if ($request->hasValue('relation') && $request->hasValue('sourceOid')) {
79  // sort in relation
80  $sourceOid = ObjectId::parse($request->getValue('sourceOid'));
81  $relatedType = $this->getRelatedType($sourceOid, $request->getValue('relation'));
82  $referenceOid = new ObjectId($relatedType, $orderReferenceIdStr);
83  }
84  else {
85  // sort in root
86  $referenceOid = new ObjectId($request->getValue('className'), $orderReferenceIdStr);
87  }
88  }
89  $request->setValue('referenceOid', $referenceOid);
90  }
91 
92  // parse get parameters
93  foreach ($request->getValues() as $key => $value) {
94  // sort definition
95  if (preg_match('/^sort\(([^\)]+)\)$|sortBy=([.]+)$/', $key, $matches)) {
96  $sortDefs = preg_split('/,/', $matches[1]);
97  // ListController allows only one sortfield
98  $sortDef = $sortDefs[0];
99  $sortFieldName = substr($sortDef, 1);
100  $sortDirection = preg_match('/^-/', $sortDef) ? 'desc' : 'asc';
101  $request->setValue('sortFieldName', $sortFieldName);
102  $request->setValue('sortDirection', $sortDirection);
103  }
104  // limit
105  if (preg_match('/^limit\(([^\)]+)\)$/', $key, $matches)) {
106  $rangeDefs = preg_split('/,/', $matches[1]);
107  $limit = intval($rangeDefs[0]);
108  $offset = sizeof($rangeDefs) > 0 ? intval($rangeDefs[1]) : 0;
109  $request->setValue('offset', $offset);
110  $request->setValue('limit', $limit);
111  }
112  }
113 
114  // create query from optional GET values encoded in RQL
115  // (https://github.com/persvr/rql)
116  if ($request->hasValue('className') && $request->hasValue('query')) {
117  $type = $request->getValue('className');
118  $stringQuery = StringQuery::fromRql($type, urldecode($request->getValue('query')));
119  $query = $stringQuery->getQueryCondition();
120  $request->setValue('query', $query);
121  }
122  }

124  protected function transformResponse($request, $response) {
125  // set content-range header
126  if ($response->hasValue('list')) {
127  $objects = $response->getValue('list');
128  $size = sizeof($objects);
129 
130  // set response range header
131  $offset = $request->getValue('offset');
132  $limit = $size == 0 ? $offset : $offset+$size-1;
133  $total = $response->getValue('totalCount');
134  $response->setHeader('Content-Range', 'items '.$offset.'-'.$limit.'/'.$total);
135  }
136  }
