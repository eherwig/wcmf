25 /**
26  * DisplayController is used to read a Node instance.
27  *
28  * The controller supports the following actions:
29  *
30  * <div class="controller-action">
31  * <div> __Action__ _default_ </div>
32  * <div>
33  * Load the given Node instance.
34  * | Parameter | Description
35  * |------------------------|-------------------------
36  * | _in_ `oid` | The object id of the Node to read
37  * | _in_ `depth` | The number of levels referenced Node must be returned as complete objects. Below this level, Nodes are returned as references. The value -1 has the special meaning of unlimited depth (optional, default: 1)
38  * | _in_ `translateValues` | Boolean whether list values should be translated to their display values (optional, default: _true_)
39  * | _out_ `object` | The Node to read
40  * | __Response Actions__ | |
41  * | `ok` | In all cases
42  * </div>
43  * </div>
44  *
45  * @author ingo herwig <ingo@wemove.com>
46  */
47 class DisplayController extends Controller {

49  /**
50  * @see Controller::validate()
51  */
52  protected function validate() {
53  $request = $this->getRequest();
54  $response = $this->getResponse();
55  $oid = ObjectId::parse($request->getValue('oid'));
56  if(!$oid) {
57  $response->addError(ApplicationError::get('OID_INVALID',
58  array('invalidOids' => array($request->getValue('oid')))));
59  return false;
60  }
61  if($request->hasValue('depth')) {
62  $depth = intval($request->getValue('depth'));
63  if ($depth < -1) {
64  $response->addError(ApplicationError::get('DEPTH_INVALID'));
65  }
66  }
67  if (!$this->checkLanguageParameter()) {
68  return false;
69  }
70  // do default validation
71  return parent::validate();
72  }
73 
74  /**
75  * @see Controller::doExecute()
76  */
77  protected function doExecute() {
78  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
79  $permissionManager = ObjectFactory::getInstance('permissionManager');
80  $request = $this->getRequest();
81  $response = $this->getResponse();
82 
83  // load model
84  $oid = ObjectId::parse($request->getValue('oid'));
85  if ($oid && $permissionManager->authorize($oid, '', PersistenceAction::READ)) {
86  // determine the builddepth
87  $buildDepth = BuildDepth::SINGLE;
88  if ($request->hasValue('depth')) {
89  $buildDepth = $request->getValue('depth');
90  }
91  $node = $persistenceFacade->load($oid, $buildDepth);
92  if ($node == null) {
93  throw new IllegalArgumentException(Message::get("The object with oid '%0%' does not exist.", array($oid)));
94  }
95 
96  // translate all nodes to the requested language if requested
97  if ($this->isLocalizedRequest()) {
98  $localization = ObjectFactory::getInstance('localization');
99  $node = $localization->loadTranslation($node, $request->getValue('language'), true, true);
100  }
101 
102  if (Log::isDebugEnabled(__CLASS__)) {
103  Log::debug(nl2br($node->__toString()), __CLASS__);
104  }
105 
106  // translate values if requested
107  if ($request->getBooleanValue('translateValues')) {
108  $nodes = array($node);
109  if ($this->isLocalizedRequest()) {
110  NodeUtil::translateValues($nodes, $request->getValue('language'));
111  }
112  else {
113  NodeUtil::translateValues($nodes);
114  }
115  }
116 
117  // assign node data
118  $response->setValue('object', $node);
119  }
120  else {
121  throw new AuthorizationException(Message::get("Authorization failed for action '%0%' on '%1%'.",
122  array(Message::get('read'), $oid)));
123  }
124  // success
125  $response->setAction('ok');
126  }
127 }
