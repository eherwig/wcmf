21 /**
22  * BatchController is used to process complex, longer running actions, that need
23  * to be divided into several requests to overcome resource limits and provide
24  * progress information to the user.
25  *
26  * Conceptionally the process is divided into subactions (_work packages_),
27  * that are called sequentially. Depending on the progress, the controller sets
28  * different actions on the response as result of one execution.
29  *
30  * BatchController only sets up the infrastructure, the concrete process is defined
31  * by creating a subclass and implementing the abstract methods (mainly
32  * BatchController::getWorkPackage()).
33  *
34  * The controller supports the following actions:
35  *
36  * <div class="controller-action">
37  * <div> __Action__ _default_ </div>
38  * <div>
39  * Initialize the work packages and process the first action.
40  * | Parameter | Description
41  * |-----------------------|-------------------------
42  * | _in_ `oneCall` | Boolean whether to accomplish the task in one call (optional, default: _false_)
43  * | _out_ `stepNumber` | The current step starting with 1, ending with _numberOfSteps_+1
44  * | _out_ `numberOfSteps` | Total number of steps
45  * | _out_ `displayText` | The display text for the current step
46  * | __Response Actions__ | |
47  * | `next` | The process is not finished and `continue` should be called as next action
48  * | `download` | The process is finished and the next call to `continue` will trigger the file download
49  * | `done` | The process is finished
50  * </div>
51  * </div>
52  *
53  * <div class="controller-action">
54  * <div> __Action__ continue </div>
55  * <div>
56  * Continue to process the next action.
57  * | Parameter | Description
58  * |-----------------------|-------------------------
59  * | _out_ `stepNumber` | The current step starting with 1, ending with _numberOfSteps_+1
60  * | _out_ `numberOfSteps` | Total number of steps
61  * | _out_ `displayText` | The display text for the current step
62  * | __Response Actions__ | |
63  * | `next` | The process is not finished and `continue` should be called as next action
64  * | `download` | The process is finished and the next call to `continue` will trigger the file download
65  * | `done` | The process is finished
66  * </div>
67  * </div>
68  *
69  * @author ingo herwig <ingo@wemove.com>
70  */

83  /**
84  * @see Controller::initialize()
85  */
86  public function initialize(Request $request, Response $response) {
87  parent::initialize($request, $response);
88 
89  $session = ObjectFactory::getInstance('session');
90  if ($request->getAction() == 'continue') {
91  // get step for current call from session
92  if ($session->exist(self::STEP_SESSION_VARNAME)) {
93  $this->_curStep = $session->get(self::STEP_SESSION_VARNAME);
94  }
95  else {
96  throw new ApplicationException($request, $response, ApplicationError::getGeneral("Current step undefined."));
97  }
98  // get workpackage definition for current call from session
99  if ($session->exist(self::WORK_PACKAGES_VARNAME)) {
100  $this->_workPackages = $session->get(self::WORK_PACKAGES_VARNAME);
101  }
102  else {
103  throw new ApplicationException($request, $response, ApplicationError::getGeneral("Work packages undefined."));
104  }
105  }
106  else {
107  // first call, initialize step session variable
108  $this->_curStep = 1;
109  $session->set(self::ONE_CALL_SESSION_VARNAME, $request->getBooleanValue('oneCall', false));
110 
111  $tmpArray = array();
112  $session->set(self::WORK_PACKAGES_VARNAME, $tmpArray);
113  $session->set(self::DOWNLOAD_STEP, false);
114 
115  // define work packages
116  $number = 0;
117  while (($workPackage = $this->getWorkPackage($number)) !== null) {
118  if (!isset($workPackage['name']) || !isset($workPackage['size']) ||
119  !isset($workPackage['oids']) || !isset($workPackage['callback'])) {
120  throw new ApplicationException($request, $response, ApplicationError::getGeneral("Incomplete work package description."));
121  }
122  else {
123  $args = isset($workPackage['args']) ? $workPackage['args'] : null;
124  $this->addWorkPackage($workPackage['name'], $workPackage['size'], $workPackage['oids'], $workPackage['callback'], $args);
125  $number++;
126  }
127  }
128  if ($number == 0) {
129  throw new ApplicationException($request, $response, ApplicationError::getGeneral("No work packages."));
130  }
131  }
132  $nextStep = $this->_curStep+1;
133  $session->set(self::STEP_SESSION_VARNAME, $nextStep);
134  }
135 
136  /**
137  * @see Controller::doExecute()
138  */
139  protected function doExecute() {
140  $session = ObjectFactory::getInstance('session');
141  $response = $this->getResponse();
142 
143  // check if a download was triggered in the last step
144  if ($session->get(self::DOWNLOAD_STEP) == true) {
145  $file = $this->getDownloadFile();
146  header("Content-Type: application/force-download");
147  header("Content-Type: application/octet-stream");
148  header("Content-Type: application/download");
149  header('Content-Disposition: attachment; filename="'.basename($file).'"');
150  echo file_get_contents($file);
151  exit;
152  }
153 
154  // continue processing
155  $curStep = $this->getStepNumber();
156  $numberOfSteps = $this->getNumberOfSteps();
157  if ($curStep <= $numberOfSteps) {
158  $this->processPart();
159 
160  $response->setValue('stepNumber', $curStep);
161  $response->setValue('numberOfSteps', $numberOfSteps);
162  $response->setValue('displayText', $this->getDisplayText($curStep));
163  }
164 
165  // check if we are finished or should continue
166  // (number of packages may have changed while processing)
167  $numberOfSteps = $this->getNumberOfSteps();
168  if ($curStep >= $numberOfSteps || $session->get(self::ONE_CALL_SESSION_VARNAME) == true) {
169  // finished -> check for download
170  $file = $this->getDownloadFile();
171  if ($file) {
172  $response->setAction('download');
173  $session->set(self::DOWNLOAD_STEP, true);
174  }
175  else {
176  $response->setAction('done');
177  }
178  }
179  else {
180  // proceed
181  $response->setAction('next');
182  }
183  }

185  /**
186  * Get the number of the current step (1..number of steps).
187  * @return The number of the current step
188  */
189  protected function getStepNumber() {
190  // since we actally call processPart() in the second step,
191  // return the real step number reduced by one
192  return $this->_curStep;
193  }

195  /**
196  * Add a work package to session. This package will be devided into sub packages of given size.
197  * @param $name Display name of the package (will be supplemented by startNumber-endNumber, e.g. '1-7', '8-14', ...)
198  * @param $size Size of one sub package. This defines how many of the oids will be passed to the callback in one call (e.g. '7' means pass 7 oids per call)
199  * @param $oids An array of object ids (or other application specific package identifiers) with _at least one value_ that will be distributed into sub packages of given size
200  * @param $callback The name of method to call for this package type.
201  * The callback method must accept the following parameters:
202  * 1. array parameter (the object ids to process in the current call)
203  * 2. optionally array parameter (the additional arguments)
204  * @param $args Assoziative array of additional callback arguments (application specific) (default: _null_)
205  */
206  protected function addWorkPackage($name, $size, $oids, $callback, $args=null) {
207  $request = $this->getRequest();
208  $response = $this->getResponse();
209  if ($size < 1) {
210  throw new ApplicationException($request, $response,
211  ApplicationError::getGeneral("Wrong work package description '".$name."': Size must be at least 1."));
212  }
213  if (sizeOf($oids) == 0) {
214  throw new ApplicationException($request, $response,
215  ApplicationError::getGeneral("Wrong work package description '".$name."': No oids given."));
216  }
217  if (strlen($callback) == 0) {
218  throw new ApplicationException($request, $response,
219  ApplicationError::getGeneral("Wrong work package description '".$name."': No callback given."));
220  }
221 
222  $session = ObjectFactory::getInstance('session');
223  $workPackages = $session->get(self::WORK_PACKAGES_VARNAME);
224 
225  $counter = 1;
226  $total = sizeOf($oids);
227  while(sizeOf($oids) > 0) {
228  $items = array();
229  for($i=0; $i<$size; $i++) {
230  $nextItem = array_shift($oids);
231  if($nextItem !== null) {
232  $items[] = $nextItem;
233  }
234  }
235 
236  // define status text
237  $start = $counter;
238  $end = ($counter+sizeOf($items)-1);
239  $stepsText = $counter;
240  if ($start != $end) {
241  $stepsText .= '-'.($counter+sizeOf($items)-1);
242  }
243  $statusText = "";
244  if ($total > 1) {
245  $statusText = $stepsText.'/'.$total;
246  }
247 
248  $curWorkPackage = array('name' => $name.' '.$statusText,
249  'oids' => $items,
250  'callback' => $callback,
251  'args' => $args);
252  $workPackages[] = $curWorkPackage;
253  $counter += $size;
254  }
255  $session->set(self::WORK_PACKAGES_VARNAME, $workPackages);
256  $session->set(self::NUM_STEPS_VARNAME, sizeOf($workPackages));
257 
258  $this->_workPackages = $workPackages;
259  }
260 
261  /**
262  * Process the next step.
263  */
264  protected function processPart() {
265  $curWorkPackageDef = $this->_workPackages[$this->getStepNumber()-1];
266  if (strlen($curWorkPackageDef['callback']) == 0) {
267  throw new ApplicationException($request, $response, ApplicationError::getGeneral("Empty callback name."));
268  }
269  else {
270  if (!method_exists($this, $curWorkPackageDef['callback'])) {
271  throw new ApplicationException($request, $response,
272  ApplicationError::getGeneral("Method '".$curWorkPackageDef['callback']."' must be implemented by ".get_class($this)));
273  }
274  else {
275  call_user_func(array($this, $curWorkPackageDef['callback']), $curWorkPackageDef['oids'], $curWorkPackageDef['args']);
276  }
277  }
278  }

301  protected function getDownloadFile() {
302  return null;
303  }

305  /**
306  * Get definitions of work packages.
307  * @param $number The number of the work package (first number is 0, number is incremented on every call)
308  * @note This function gets called on first initialization run as often until it returns null.
309  * This allows to define different static work packages. If you would like to add work packages dynamically on

313  */
314  protected abstract function getWorkPackage($number);
