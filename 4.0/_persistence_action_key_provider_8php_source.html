56  /**
57  * @see ActionKeyProvider::getKeyValue()
58  */
59  public function getKeyValue($actionKey) {
60  $query = new ObjectQuery($this->_entityType, __CLASS__.__METHOD__);
61  $tpl = $query->getObjectTemplate($this->_entityType);
62  $actionKeyParams = ActionKey::parseKey($actionKey);
63  $tpl->setValue($this->_valueMap['resource'], Criteria::asValue('=', $actionKeyParams['resource']));
64  $tpl->setValue($this->_valueMap['context'], Criteria::asValue('=', $actionKeyParams['context']));
65  $tpl->setValue($this->_valueMap['action'], Criteria::asValue('=', $actionKeyParams['action']));
66  $keys = $query->execute(BuildDepth::SINGLE);
67  if (sizeof($keys) > 0) {
68  return $keys[0]->getValue($this->_valueMap['value']);
69  }
70  return null;
71  }
