33  /**
34  * Truncate a string up to a number of characters while preserving whole words and HTML tags
35  * code from: http://alanwhipple.com/2011/05/25/php-truncate-string-preserving-html-tags-words/
36  * @param $string String to truncate.
37  * @param $length Length of returned string, including ellipsis.
38  * @param $ending Ending to be appended to the trimmed string.
39  * @param $exact If false, $text will not be cut mid-word
40  * @param $considerHtml If true, HTML tags would be handled correctly
41  * @return String
42  */
43  public static function cropString($string, $length=100, $ending='...', $exact=false, $considerHtml=true) {
44  if ($considerHtml) {
45  // if the plain text is shorter than the maximum length, return the whole text
46  if (strlen(preg_replace('/<.*?>/', '', $string)) <= $length) {
47  return $string;
48  }
49  // splits all html-tags to scanable lines
50  preg_match_all('/(<.+?>)?([^<>]*)/s', $string, $lines, PREG_SET_ORDER);
51  $total_length = strlen($ending);
52  $open_tags = array();
53  $truncate = '';
54  foreach ($lines as $line_matchings) {
55  // if there is any html-tag in this line, handle it and add it (uncounted) to the output
56  if (!empty($line_matchings[1])) {
57  // if it's an "empty element" with or without xhtml-conform closing slash
58  if (preg_match('/^<(\s*.+?\/\s*|\s*(img|br|input|hr|area|base|basefont|col|frame|isindex|link|meta|param)(\s.+?)?)>$/is', $line_matchings[1])) {
59  // do nothing
60  // if tag is a closing tag
61  } else if (preg_match('/^<\s*\/([^\s]+?)\s*>$/s', $line_matchings[1], $tag_matchings)) {
62  // delete tag from $open_tags list
63  $pos = array_search($tag_matchings[1], $open_tags);
64  if ($pos !== false) {
65  unset($open_tags[$pos]);
66  }
67  // if tag is an opening tag
68  } else if (preg_match('/^<\s*([^\s>!]+).*?>$/s', $line_matchings[1], $tag_matchings)) {
69  // add tag to the beginning of $open_tags list
70  array_unshift($open_tags, strtolower($tag_matchings[1]));
71  }
72  // add html-tag to $truncate'd text
73  $truncate .= $line_matchings[1];
74  }
75  // calculate the length of the plain text part of the line; handle entities as one character
76  $content_length = strlen(preg_replace('/&[0-9a-z]{2,8};|&#[0-9]{1,7};|[0-9a-f]{1,6};/i', ' ', $line_matchings[2]));
77  if ($total_length+$content_length> $length) {
78  // the number of characters which are left
79  $left = $length - $total_length;
80  $entities_length = 0;
81  // search for html entities
82  if (preg_match_all('/&[0-9a-z]{2,8};|&#[0-9]{1,7};|[0-9a-f]{1,6};/i', $line_matchings[2], $entities, PREG_OFFSET_CAPTURE)) {
83  // calculate the real length of all entities in the legal range
84  foreach ($entities[0] as $entity) {
85  if ($entity[1]+1-$entities_length <= $left) {
86  $left--;
87  $entities_length += strlen($entity[0]);
88  } else {
89  // no more characters left
90  break;
91  }
92  }
93  }
94  $truncate .= substr($line_matchings[2], 0, $left+$entities_length);
95  // maximum lenght is reached, so get off the loop
96  break;
97  } else {
98  $truncate .= $line_matchings[2];
99  $total_length += $content_length;
100  }
101  // if the maximum length is reached, get off the loop
102  if($total_length>= $length) {
103  break;
104  }
105  }
106  } else {
107  if (strlen($string) <= $length) {
108  return $string;
109  } else {
110  $truncate = substr($string, 0, $length - strlen($ending));
111  }
112  }
113  // if the words shouldn't be cut in the middle...
114  if (!$exact) {
115  // ...search the last occurance of a space...
116  $spacepos = strrpos($truncate, ' ');
117  if (isset($spacepos)) {
118  // ...and cut the text in this position
119  $truncate = substr($truncate, 0, $spacepos);
120  }
121  }
122  // add the defined ending to the text
123  $truncate .= $ending;
124  if($considerHtml) {
125  // close all unclosed html-tags
126  foreach ($open_tags as $tag) {
127  $truncate .= '</' . $tag . '>';
128  }
129  }
130  return $truncate;
131  }

133  /**
134  * Create an excerpt from the given text around the given phrase
135  * code based on: http://stackoverflow.com/questions/1292121/how-to-generate-the-snippet-like-generated-by-google-with-php-and-mysql
136  * @param $string
137  * @param $phrase
138  * @param $radius
139  */
140  public static function excerpt($string, $phrase, $radius = 100) {
141  if ($radius > strlen($string)) {
142  return $string;
143  }
144  $phraseLen = strlen($phrase);
145  if ($radius < $phraseLen) {
146  $radius = $phraseLen;
147  }
148  $pos = strpos(strtolower($string), strtolower($phrase));
149 
150  $startPos = 0;
151  if ($pos > $radius) {
152  $startPos = $pos - $radius;
153  }
154  $textLen = strlen($string);
155 
156  $endPos = $pos + $phraseLen + $radius;
157  if ($endPos >= $textLen) {
158  $endPos = $textLen;
159  }
160 
161  // make sure to cut at spaces
162  $firstSpacePos = strpos($string, " ", $startPos);
163  $lastSpacePos = strrpos($string, " ", -(strlen($string)-$endPos));
164 
165  $excerpt1 = substr($string, $firstSpacePos, $lastSpacePos-$firstSpacePos);
166 
167  // remove open tags
168  $excerpt = preg_replace('/^[^<]*?>|<[^>]*?$/', '', $excerpt1);
169  return $excerpt;
170  }

172  /**
173  * Extraxt urls from a string.
174  * @param $string The string to search in
175  * @return An array with urls
176  * @note This method searches for occurences of <a..href="xxx"..>, <img..src="xxx"..>,
177  * <input..src="xxx"..> or <form..action="xxx"..> and extracts xxx.
178  */
179  public static function getUrls($string) {
180  preg_match_all("/<a[^>]+href=\"([^\">]+)/i", $string, $links);
181 
182  // find urls in javascript popup links
183  for ($i=0; $i<sizeof($links[1]); $i++) {
184  if (preg_match_all("/javascript:.*window.open[\(]*'([^']+)/i", $links[1][$i], $popups)) {
185  $links[1][$i] = $popups[1][0];
186  }
187  }
188  // remove mailto links
189  for ($i=0; $i<sizeof($links[1]); $i++) {
190  if (preg_match("/^mailto:/i", $links[1][$i])) {
191  unset($links[1][$i]);
192  }
193  }
194  preg_match_all("/<img[^>]+src=\"([^\">]+)/i", $string, $images);
195  preg_match_all("/<input[^>]+src=\"([^\">]+)/i", $string, $buttons);
196  preg_match_all("/<form[^>]+action=\"([^\">]+)/i", $string, $actions);
197  return array_merge($links[1], $images[1], $buttons[1], $actions[1]);
198  }

200  /**
201  * Split a quoted string
202  * code from: http://php3.de/manual/de/function.split.php
203  * @code
204  * $string = '"hello, world", "say \"hello\"", 123, unquotedtext';
205  * $result = quotsplit($string);
206  *
207  * // results in:
208  * // ['hello, world'] [say "hello"] [123] [unquotedtext]
209  *
210  * @endcode
211  *
212  * @param $string The string to split
213  * @return An array of strings
214  */
215  public static function quotesplit($string) {
216  $r = Array();
217  $p = 0;
218  $l = strlen($string);
219  while ($p < $l) {
220  while (($p < $l) && (strpos(" \r\t\n",$string[$p]) !== false)) $p++;
221  if ($string[$p] == '"') {
222  $p++;
223  $q = $p;
224  while (($p < $l) && ($string[$p] != '"')) {
225  if ($string[$p] == '\\') { $p+=2; continue; }
226  $p++;
227  }
228  $r[] = stripslashes(substr($string, $q, $p-$q));
229  $p++;
230  while (($p < $l) && (strpos(" \r\t\n",$string[$p]) !== false)) $p++;
231  $p++;
232  }
233  else if ($string[$p] == "'") {
234  $p++;
235  $q = $p;
236  while (($p < $l) && ($string[$p] != "'")) {
237  if ($string[$p] == '\\') { $p+=2; continue; }
238  $p++;
239  }
240  $r[] = stripslashes(substr($string, $q, $p-$q));
241  $p++;
242  while (($p < $l) && (strpos(" \r\t\n",$string[$p]) !== false)) $p++;
243  $p++;
244  }
245  else {
246  $q = $p;
247  while (($p < $l) && (strpos(",;",$string[$p]) === false)) {
248  $p++;
249  }
250  $r[] = stripslashes(trim(substr($string, $q, $p-$q)));
251  while (($p < $l) && (strpos(" \r\t\n",$string[$p]) !== false)) $p++;
252  $p++;
253  }
254  }
255  return $r;
256  }

258  /**
259  * Split string preserving quoted strings
260  * code based on: http://www.php.net/manual/en/function.explode.php#94024
261  * @param $string String to split
262  * @param $delim Regexp to use in preg_split
263  * @param $quoteChr Quote character
264  * @param $preserve Boolean whether to preserve the quote character or not
265  * @return Array
266  */
267  public static function splitQuoted($string, $delim='/ /', $quoteChr='"', $preserve=false){
268  $resArr = array();
269  $n = 0;
270  $expEncArr = explode($quoteChr, $string);
271  foreach($expEncArr as $encItem) {
272  if ($n++%2) {
273  $resArr[] = array_pop($resArr) . ($preserve?$quoteChr:'') . $encItem.($preserve?$quoteChr:'');
274  }
275  else {
276  $expDelArr = preg_split($delim, $encItem);
277  $resArr[] = array_pop($resArr) . array_shift($expDelArr);
278  $resArr = array_merge($resArr, $expDelArr);
279  }
280  }
281  return $resArr;
282  }
