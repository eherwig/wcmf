31  /**
32  * @see AuthUser::login()
33  */
34  public function login($login, $password) {
35  $config = ObjectFactory::getConfigurationInstance();
36  $userTypeInst = ObjectFactory::getInstance('User');
37 
38  // because there is no authorized user already, we have to add a temporary permission to the
39  // PermissionManager for this operation to allow user retrieval from the persistent storage
40  $permissionManager = ObjectFactory::getInstance('permissionManager');
41  $permissionManager->addTempPermission($userTypeInst->getType(), '', 'read');
42  // try to receive the user with given credentials
43  $user = $userTypeInst::getByLogin($login);
44  // remove the temporary permission
45  $permissionManager->removeTempPermission($userTypeInst->getType(), '', 'read');
46 
47  // check if user exists
48  $loginOk = false;
49  if ($user != null) {
50  // check password
51  $loginOk = $user->verifyPassword($password, $user->getPassword());
52  if ($loginOk) {
53  // login succeeded, store the user instance
54  $this->_user = clone $user;
55 
56  // load user config initially
57  $userConfig = $this->getConfig();
58  if (strlen($userConfig) > 0) {
59  $config->addConfiguration($userConfig);
60  }
61 
62  // add policies
63  $policies = $config->getSection('authorization');
64  $this->addPolicies($policies);
65  $this->_login_time = strftime("%c", time());
66  }
67  }
68 
69  return $loginOk;
70  }

103  /**
104  * Adds one ore more policies to the policy repository of the user.
105  * @param $policies An associative array with the policy information
106  * (key=action, value=policy string).
107  * @note A policy string looks like this "+*, -guest, +admin"
108  */
109  protected function addPolicies(array $policies) {
110  foreach ($policies AS $key => $value) {
111  if (!isset($this->_policies[$key])) {
112  $parsedPolicies = Policy::parse($value);
113  $this->_policies[$key] = $parsedPolicies;
114  }
115  }
116  }

118  /**
119  * Matches the roles of the user and the roles for a certain key
120  * @param $val An array containing policy information as an associative array
121  * with the keys ('default', 'allow', 'deny'). Where 'allow', 'deny' are arrays
122  * itselves holding roles. 'allow' overwrites 'deny' overwrites 'default'
123  * @return Boolean whether the user has access right according to this policy.
124  */
125  protected function matchRoles($val) {
126  if (isset($val['allow'])) {
127  foreach ($val['allow'] as $value) {
128  if ($this->hasRole($value)) {
129  return true;
130  }
131  }
132  }
133  if (isset($val['deny'])) {
134  foreach ($val['deny'] as $value) {
135  if ($this->hasRole($value)) {
136  return false;
137  }
138  }
139  }
140  return isset($val['default']) ? $val['default'] : false;
141  }
