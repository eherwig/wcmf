18 /**
19  * Controller is the base class of all controllers.
20  *
21  * Error Handling:
22  * - throw an Exception or use response action _failure_ to signal fatal errors
23  * (calls wcmf::application::controller::FailureController)
24  * - add an ApplicationError to the response to signal non fatal errors
25  *
26  * The following default request/response parameters are defined:
27  *
28  * | Parameter | Description
29  * |-------------------------|-------------------------
30  * | _in_ / _out_ `action` | The action to be executed
31  * | _in_ / _out_ `context` | The context of the action
32  * | _in_ `language` | The language of the requested data (optional)
33  * | _out_ `controller` | The name of the executed controller
34  * | _out_ `success` | Boolean whether the action completed successfully or not (depends on existence of error messages)
35  * | _out_ `errorMessage` | An error message which is displayed to the user
36  * | _out_ `errorCode` | An error code, describing the type of error
37  * | _out_ `errorData` | Some error codes require to transmit further information to the client
38  *
39  * @author ingo herwig <ingo@wemove.com>
40  */

46  /**
47  * Initialize the Controller with request/response data. Which data is required is defined by the Controller.
48  * The base class method just stores the parameters in a member variable. Specialized Controllers may overide
49  * this behaviour for further initialization.
50  * @attention It lies in its responsibility to fail or do some default action if some data is missing.
51  * @param $request A reference to the Request sent to the Controller. The sender attribute of the Request is the
52  * last controller's name, the context is the current context and the action is the requested one.
53  * All data sent from the last controller are accessible using the Request::getValue method. The request is
54  * supposed to be read-only. It will not be used any more after beeing passed to the controller.
55  * @param $response A reference to the Response that will be modified by the Controller. The initial values for
56  * context and action are the same as in the request parameter and are meant to be modified according to the
57  * performed action. The sender attribute of the response is set to the current controller. Initially there
58  * are no data stored in the response.
59  */
60  public function initialize(Request $request, Response $response) {
61  // set sender on response
62  $response->setSender(get_class($this));
63 
64  // restore the error messages of a previous call
65  if ($request->hasErrors()) {
66  foreach ($request->getErrors() as $error) {
67  $response->addError($error);
68  }
69  }
70 
71  $this->_request = $request;
72  $this->_response = $response;
73  }
74 
75  /**
76  * Check if the data given by initialize() meet the requirements of the Controller.
77  * Subclasses will override this method to validate against their special requirements.
78  * @return Boolean whether the data are ok or not.
79  */
80  protected function validate() {
81  return true;
82  }

84  /**
85  * Execute the Controller resulting in its action processed. The actual
86  * processing is done in Controller::doExecute(), which is implemented
87  * by concrete Controller subclasses.
88  */
89  public function execute() {
90  $isDebugEnabled = Log::isDebugEnabled(__CLASS__);
91  if ($isDebugEnabled) {
92  Log::debug('Executing: '.get_class($this), __CLASS__);
93  Log::debug('Request: '.$this->_request, __CLASS__);
94  }
95 
96  // validate controller data
97  $validationFailed = false;
98  if (!$this->validate()) {
99  $validationFailed = true;
100  }
101 
102  // execute controller logic
103  if (!$validationFailed) {
104  $this->doExecute();
105  }
106 
107  // prepare the response
108  $this->assignResponseDefaults();
109  if ($isDebugEnabled) {
110  Log::debug('Response: '.$this->_response, __CLASS__);
111  }
112 
113  // log errors
114  $errors = $this->_response->getErrors();
115  for ($i=0,$count=sizeof($errors); $i<$count; $i++) {
116  Log::error($errors[$i]->__toString(), __CLASS__);
117  }
118  }

120  /**
121  * Process the request.

126  /**
127  * Delegate the current request to another action. The context is the same as
128  * the current context and the source controller will be set to this.
129  * The request and response format will be NullFormat
130  * which means that all request values should be passed in the application internal
131  * format and all response values will have that format.
132  * @param $action The name of the action to execute
133  * @return Response instance
134  */
135  protected function executeSubAction($action) {
136  $curRequest = $this->getRequest();
137  $subRequest = new Request(get_class($this), $curRequest->getContext(), $action);
138  $subRequest->setHeaders($curRequest->getHeaders());
139  $subRequest->setValues($curRequest->getValues());
140  $formats = ObjectFactory::getInstance('formats');
141  $nullFormat = $formats['null'];
142  $subRequest->setFormat($nullFormat);
143  $subRequest->setResponseFormat($nullFormat);
144  $response = ObjectFactory::getInstance('actionMapper')->processAction($subRequest);
145  return $response;
146  }

164  /**
165  * Assign default variables to the response. This method is called after Controller execution.
166  * This method may be used by derived controller classes for convenient response setup.
167  */
168  protected function assignResponseDefaults() {
169  // return the first error
170  $errors = $this->_response->getErrors();
171  if (sizeof($errors) > 0) {
172  $error = $errors[0];
173  $this->_response->setValue('errorCode', $error->getCode());
174  $this->_response->setValue('errorMessage', $error->getMessage());
175  $this->_response->setValue('errorData', $error->getData());
176  $this->_response->setStatus(Response::STATUS_400);
177  }
178  // set the success flag
179  if (sizeof($errors) > 0) {
180  $this->_response->setValue('success', false);
181  }
182  else {
183  $this->_response->setValue('success', true);
184  }
185 
186  // set wCMF specific values
187  $this->_response->setValue('controller', get_class($this));
188  $this->_response->setValue('context', $this->_response->getContext());
189  $this->_response->setValue('action', $this->_response->getAction());
190  }
191 
192  /**
193  * Check if the current request is localized. This is true,
194  * if it has a language parameter that is not equal to Localization::getDefaultLanguage().
195  * Throws an exception if a language is given which is not supported
196  * @return Boolean whether the request is localized or not
197  */
198  protected function isLocalizedRequest() {
199  if ($this->_request->hasValue('language')) {
200  $language = $this->_request->getValue('language');
201  $localization = ObjectFactory::getInstance('localization');
202  if ($language != $localization->getDefaultLanguage()) {
203  return true;
204  }
205  }
206  return false;
207  }
208 
209  /**
210  * Checks the language request parameter and adds an respons error,
211  * if it is not contained in the Localization::getSupportedLanguages() list.
212  * @return Boolean
213  */
214  protected function checkLanguageParameter() {
215  if ($this->_request->hasValue('language')) {
216  $language = $this->_request->getValue('language');
217  $localization = ObjectFactory::getInstance('localization');
218  if (!in_array($language, array_keys($localization->getSupportedLanguages()))) {
219  $this->_response->addError(ApplicationError::get('PARAMETER_INVALID',
220  array('invalidParameters' => array('language'))));
221  return false;
222  }
223  }
224  return true;
225  }
