28  if (!defined('WCMF_BASE')) {

30  . "Please define this constant to point to "

41  /**
42  * Search a class definition in any subfolder of WCMF_BASE
43  * Code from: http://php.net/manual/en/language.oop5.autoload.php
44  *
45  * @param $className The name of the class
46  * @param $sub The start directory (optional, default: '/')
47  * @return The directory name
48  */
49  function searchClass($className, $sub="/") {
50  if(file_exists(WCMF_BASE.$sub.getFileName($className))) {
51  return WCMF_BASE.$sub;
52  }
53 
54  $dir = dir(WCMF_BASE.$sub);
55  while(false !== ($folder = $dir->read())) {
56  if($folder != "." && $folder != "..") {
57  if(is_dir(WCMF_BASE.$sub.$folder)) {
58  $subFolder = searchClass($className, $sub.$folder."/");
59  if($subFolder) {
60  return $subFolder;
61  }
62  }
63  }
64  }
65  $dir->close();
66  return false;
67  }
