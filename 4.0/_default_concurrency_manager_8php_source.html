96  /**
97  * @see ConcurrencyManager::checkPersist()
98  */
99  public function checkPersist(PersistentObject $object) {
100  $oid = $object->getOID();
101  $lock = $this->_lockHandler->getLock($oid);
102  if ($lock != null) {
103  $type = $lock->getType();
104 
105  // if there is a pessimistic lock on the object and it's not
106  // owned by the current user, throw a PessimisticLockException
107  if ($type == Lock::TYPE_PESSIMISTIC) {
108  $permissionManager = ObjectFactory::getInstance('permissionManager');
109  $currentUser = $permissionManager->getAuthUser();
110  if ($lock->getLogin() != $currentUser->getLogin()) {
111  throw new PessimisticLockException($lock);
112  }
113  }
114 
115  // if there is an optimistic lock on the object and the object was updated
116  // in the meantime, throw a OptimisticLockException
117  if ($type == Lock::TYPE_OPTIMISTIC) {
118  $originalState = $lock->getCurrentState();
119  // temporarily detach the object from the transaction in order to get
120  // the latest version from the store
121  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
122  $transaction = $persistenceFacade->getTransaction();
123  $transaction->detach($object->getOID());
124  $currentState = $persistenceFacade->load($oid, BuildDepth::SINGLE);
125  // check for deletion
126  if ($currentState == null) {
127  throw new OptimisticLockException(null);
128  }
129  // check for modifications
130  $mapper = $persistenceFacade->getMapper($object->getType());
131  $it = new NodeValueIterator($originalState, false);
132  foreach($it as $valueName => $originalValue) {
133  $attribute = $mapper->getAttribute($valueName);
134  // ignore references
135  if (!($attribute instanceof ReferenceDescription)) {
136  $currentValue = $currentState->getValue($valueName);
137  if (strval($currentValue) != strval($originalValue)) {
138  if (Log::isDebugEnabled(__CLASS__)) {
139  Log::debug("Current state is different to original state: ".$object->getOID()."-".$valueName.": current[".
140  serialize($currentValue)."], original[".serialize($originalValue)."]", __CLASS__);
141  }
142  throw new OptimisticLockException($currentState);
143  }
144  }
145  }
146  // if there was no concurrent update, attach the object again
147  if ($object->getState() == PersistentObject::STATE_DIRTY) {
148  $transaction->registerDirty($object);
149  }
150  elseif ($object->getState() == PersistentObject::STATE_DELETED) {
151  $transaction->registerDeleted($object);
152  }
153  }
154  }
155  // everything is ok
156  }
