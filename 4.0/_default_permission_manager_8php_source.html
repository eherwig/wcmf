17 /**
18  * DefaultPermissionManager retrieves authorization rules from the
19  * application configuration.
20  *
21  * @author ingo herwig <ingo@wemove.com>
22  */
23 class DefaultPermissionManager extends AbstractPermissionManager implements PermissionManager {
24 
25  const AUTHORIZATION_SECTION = 'authorization';
26 
27  /**
28  * @see PermissionManager::getPermissions()
29  */
30  public function getPermissions($resource, $context, $action) {
31  $actionKey = ActionKey::getBestMatch(self::AUTHORIZATION_SECTION, $resource, $context, $action);
32  if (strlen($actionKey) > 0) {
33  $config = ObjectFactory::getConfigurationInstance();
34  return $this->parsePermissions($config->getValue($actionKey, self::AUTHORIZATION_SECTION));
35  }
36  return null;
37  }
38 
39  /**
40  * @see PermissionManager::createPermission()
41  */
42  public function createPermission($resource, $context, $action, $role, $modifier) {
43  return self::modifyPermission($resource, $context, $action, $role, $modifier);
44  }
45 
46  /**
47  * @see PermissionManager::removePermission()
48  */
49  public function removePermission($resource, $context, $action, $role) {
50  return self::modifyPermission($resource, $context, $action, $role, null);
51  }
52 
53  /**
54  * Modify a permission for the given role.
55  * @param $resource The resource (e.g. class name of the Controller or OID).
56  * @param $context The context in which the action takes place.
57  * @param $action The action to process.
58  * @param $role The role to authorize.
59  * @param $modifier One of the PERMISSION_MODIFIER_ constants, null, if the permission
60  * should be removed.
61  * @return boolean
62  */
63  protected function modifyPermission($resource, $context, $action, $role, $modifier) {
64  // get config file to modify
65  $appConfig = ObjectFactory::getConfigurationInstance();
66  $configFiles = $appConfig->getConfigurations();
67  if (sizeof($configFiles) == 0) {
68  return false;
69  }
70 
71  // create a writable configuration and modify the permission
72  $mainConfig = $configFiles[0];
73  $newConfig = new IniFileConfiguration(dirname($mainConfig));
74  $newConfig->addConfiguration(basename($mainConfig));
75 
76  $permDef = ActionKey::createKey($resource, $context, $action);
77  $permVal = '';
78  if ($modifier != null) {
79  $permVal = $modifier.$role;
80  }
81  if ($newConfig->getValue($permDef, self::AUTHORIZATION_SECTION) === false && $modifier != null) {
82  $newConfig->setValue($permDef, $permVal, self::AUTHORIZATION_SECTION, true);
83  }
84  else {
85  $value = $newConfig->getValue($permDef, self::AUTHORIZATION_SECTION);
86  // remove role from value
87  $newValue = str_replace(array(PermissionManager::PERMISSION_MODIFIER_ALLOW.$role,
88  PermissionManager::PERMISSION_MODIFIER_DENY.$role), "", $value);
89  if ($newValue != '') {
90  $newConfig->setValue($permDef, $newValue." ".$permVal, self::AUTHORIZATION_SECTION, false);
91  }
92  else {
93  $newConfig->removeKey($permDef, self::AUTHORIZATION_SECTION);
94  }
95  }
96 
97  $newConfig->writeConfiguration(basename($mainConfig));
98  return true;
99  }
100 }
