61  /**
62  * Create the test database, if sqlite is configured
63  * @return Associative array with connection parameters and key 'connection'
64  */
65  public static function createDatabase() {
66  // get connection from first entity type
67  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
68  $types = $persistenceFacade->getKnownTypes();
69  $mapper = $persistenceFacade->getMapper($types[0]);
70  $pdo = $mapper->getConnection();
71 
72  // create db
73  $params = $mapper->getConnectionParams();
74  switch($params['dbType']) {
75  case 'sqlite':
76  $numTables = $pdo->query('SELECT count(*) FROM sqlite_master WHERE type = "table"')->fetchColumn();
77  if ($numTables == 0) {
78  $schema = file_get_contents(WCMF_BASE.'install/tables_sqlite.sql');
79  $pdo->exec($schema);
80  }
81  break;
82  case 'mysql':
83  $numTables = $pdo->query('SELECT COUNT(*) FROM information_schema.tables WHERE table_schema = "'.$params['dbName'].'"')->fetchColumn();
84  if ($numTables == 0) {
85  $schema = file_get_contents(WCMF_BASE.'install/tables_mysql.sql');
86  $pdo->exec($schema);
87  }
88  break;
89  }
90  $params['connection'] = $pdo;
91  return $params;
92  }

94  /**
95  * Start the built-in webserver
96  * @param $documentRoot Document root directory
97  * @param $router Router script filename (optional)
98  * @param $killOnExit Boolean, whether to kill the server process after script execution or not (optional, default: false)
99  */
100  public static function startServer($documentRoot, $router='', $killOnExit=false) {
101  if (!is_dir($documentRoot)) {
102  throw new \Exception('Document root '.$documentRoot.' does not exist');
103  }
104  if (!defined('SERVER_HOST')) {
105  define('SERVER_HOST', 'localhost');
106  define('SERVER_PORT', 8500);
107  }
108  $cmd = sprintf('php -S %s:%d -t %s %s', SERVER_HOST, SERVER_PORT, $documentRoot, $router);
109 
110  $descriptorspec = [
111  0 => ['pipe', 'r'], // stdin
112  1 => ['pipe', 'w'], // stdout
113  2 => ['pipe', 'a'] // stderr
114  ];
115  $pipes = null;
116  if (self::isWindows()) {
117  $resource = proc_open("start /B ".$cmd, $descriptorspec, $pipes);
118  }
119  else {
120  $resource = proc_open("nohup ".$cmd, $descriptorspec, $pipes);
121  }
122  if (!is_resource($resource)) {
123  throw new \Exception("Failed to execute ".$cmd);
124  }
125 
126  if ($killOnExit) {
127  // kill the web server when the process ends
128  register_shutdown_function(function() use ($resource) {
129  $status = proc_get_status($resource);
130  $pid = $status['pid'];
131  if (self::isWindows()) {
132  $output = array_filter(explode(" ", shell_exec("wmic process get parentprocessid,processid | find \"$pid\"")));
133  array_pop($output);
134  $pid = end($output);
135  exec("taskkill /F /T /PID $pid");
136  }
137  else {
138  $pid = $pid+1;
139  exec("kill -9 $pid");
140  }
141  });
142  }
143 
144  sleep(1);
145  }

164  /**
165  * Start a session. This is useful for simulateRequest calls
166  * @param $user The name of the user
167  * @param $password The password of the user
168  * @return The session id. Use this as data['sid'] parameter for
169  * subsequent simulateRequest calls
170  */
171  public static function startSession($user, $password) {
172  $session = ObjectFactory::getInstance('session');
173  $authManager = ObjectFactory::getInstance('authenticationManager');
174  $authUser = $authManager->login([
175  'login' => $user,
176  'password' => $password
177  ]);
178  if ($authUser) {
179  $session->clear();
180  $session->setAuthUser($authUser->getLogin());
181  }
182  else {
183  throw new \RuntimeException("Session could not be started for user '$user'");
184  }
185  return $session->getID();

206  /**
207  * Call a protected/private method of an instance (PHP >= 5.3.2)
208  * @param $instance The instance
209  * @param $methodName The method name
210  * @param $args An array of method arguments
211  */
212  public static function callProtectedMethod($instance, $methodName, $args=null) {
213  $className = get_class($instance);
214  $class = new \ReflectionClass($className);
215  $method = $class->getMethod($methodName);
216  $method->setAccessible(true);
217 
218  if ($args != null) {
219  return $method->invokeArgs($instance, $args);
220  }
221  else {
222  return $method->invoke($instance);
223  }
224  }
