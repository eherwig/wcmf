17 /**
18  * ConfigListStrategy implements list of key value pairs that is retrieved
19  * from an configuration section.
20  * The following list definition(s) must be used in the input_type configuraton:
21  * @code
22  * config:section // where section is the name of a configuration section
23  * @endcode
24  *
25  * @author ingo herwig <ingo@wemove.com>
26  */
27 class ConfigListStrategy implements ListStrategy {
28 
29  private $_lists = array();
30 
31  /**
32  * @see ListStrategy::getList
33  */
34  public function getList($configuration, $language=null) {
35  $listKey = $configuration.$language;
36  if (!isset($this->_lists[$listKey])) {
37  $config = ObjectFactory::getConfigurationInstance();
38  $map = $config->getSection($configuration);
39  $result = array();
40  foreach ($map as $key => $value) {
41  $result[$key] = Message::get($value, null, $language);
42  }
43  $this->_lists[$listKey] = $result;
44  }
45  return $this->_lists[$listKey];
46  }
47 
48  /**
49  * @see ListStrategy::isStatic
50  */
51  public function isStatic($configuration) {
52  return true;
53  }
54 }
