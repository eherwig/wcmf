16 /**
17  * PersistentIterator is used to iterate over a tree/list build of oids
18  * using a Depth-First-Algorithm. To persist its state use the PersistentIterator::save() method,
19  * to restore its state use the static PersistentIterator::load() method, which returns the loaded instance.
20  * States are identified by an unique id, which is provided after saving.
21  *
22  * @author ingo herwig <ingo@wemove.com>
23  */
24 class PersistentIterator implements \Iterator {
25 
26  protected $_end; // indicates if the iteration is ended
27  protected $_oidList; // the list of oids to process
28  protected $_processedOidList; // the list of all seen object ids
29  protected $_currentOid; // the oid the iterator points to
30  protected $_startOid; // the oid the iterator started with
31  protected $_currentDepth; // the depth in the tree of the oid the iterator points to
32  protected $_aggregationKinds; // array of aggregation kind values to follow (empty: all)
33 
34  /**
35  * Constructor.
36  * @param $oid The oid to start from.
37  * @param $aggregationKinds Array of aggregation kind values of relations to follow
38  * possible values: 'none', 'shared', 'composite'. Empty array means all (default: empty)
39  */
40  public function __construct(ObjectId $oid, $aggregationKinds=array()) {
41  $this->_end = false;
42  $this->_oidList = array();
43  $this->_processedOidList = array();
44  $this->_currentOid = $oid;
45  $this->_startOid = $oid;
46  $this->_currentDepth = 0;
47  $this->_aggregationKinds = $aggregationKinds;
48  }
49 
50  /**
51  * Save the iterator state to the session
52  * @return A unique id to provide for load, see PersistentIterator::load()
53  */
54  public function save() {
55  $session = ObjectFactory::getInstance('session');
56 
57  $uid = md5(uniqid(""));
58  $state = array('end' => $this->_end, 'oidList' => $this->_oidList, 'processedOidList' => $this->_processedOidList,
59  'currentOID' => $this->_currentOid, 'currentDepth' => $this->_currentDepth, 'aggregationKinds' => $this->_aggregationKinds);
60  $session->set('PersistentIterator.'.$uid, $state);
61  return $uid;
62  }
63 
64  /**
65  * Load an iterator state from the session
66  * @param $uid The unique id returned from the save method, see PersistentIterator::save()
67  * @return PersistentIterator instance holding the saved state or null if unique id is not found
68  */
69  public static function load($uid) {
70  // get state from session
71  $session = ObjectFactory::getInstance('session');
72  $state = $session->get('PersistentIterator.'.$uid);
73  if ($state == null) {
74  return null;
75  }
76  // create instance
77  $instance = new PersistentIterator($state['currentOID']);
78  $instance->_end = $state['end'];
79  $instance->_oidList = $state['oidList'];
80  $instance->_processedOidList = $state['processedOidList'];
81  $instance->_currentDepth = $state['currentDepth'];
82  $instance->_aggregationKinds = $state['aggregationKinds'];
83  return $instance;
84  }
85 
86  /**
87  * Return the current element
88  * @return ObjectId, the current object id
89  */
90  public function current() {
91  return $this->_currentOid;
92  }
93 
94  /**
95  * Return the key of the current element
96  * @return Number, the current depth
97  */
98  public function key() {
99  return $this->_currentDepth;
100  }

102  /**
103  * Move forward to next element
104  */
105  public function next() {
106  // the current oid was processed
107  $this->_processedOidList[] = $this->_currentOid->__toString();
108 
109  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
110  $node = $persistenceFacade->load($this->_currentOid);
111 
112  // collect navigable children for the given aggregation kinds
113  $childOIDs = array();
114  $mapper = $node->getMapper();
115  $relations = $mapper->getRelations('child');
116  $followAll = sizeof($this->_aggregationKinds) == 0;
117  foreach ($relations as $relation) {
118  $aggregationKind = $relation->getOtherAggregationKind();
119  if ($relation->getOtherNavigability() && ($followAll || in_array($aggregationKind, $this->_aggregationKinds))) {
120  $childValue = $node->getValue($relation->getOtherRole());
121  if ($childValue != null) {
122  $children = $relation->isMultiValued() ? $childValue : array($childValue);
123  foreach ($children as $child) {
124  $childOIDs[] = $child->getOID();
125  }
126  }
127  }
128  }
129  $this->addToQueue($childOIDs, ++$this->_currentDepth);
130 
131  // set current node
132  if (sizeof($this->_oidList) != 0) {
133  list($oid, $depth) = array_pop($this->_oidList);
134  $oidStr = $oid->__toString();
135  // not the last node -> search for unprocessed nodes
136  while (sizeof($this->_oidList) > 0 && in_array($oidStr, $this->_processedOidList)) {
137  list($oid, $depth) = array_pop($this->_oidList);
138  $oidStr = $oid->__toString();
139  }
140  // last node found, but it was processed already
141  if (sizeof($this->_oidList) == 0 && in_array($oidStr, $this->_processedOidList)) {
142  $this->_end = true;
143  }
144  $this->_currentOid = $oid;
145  $this->_currentDepth = $depth;
146  }
147  else {
148  $this->_end = true;
149  }
150  return $this;
151  }
152 
153  /**
154  * Rewind the Iterator to the first element
155  */
156  public function rewind() {
157  $this->_end = false;
158  $this->_oidList= array();
159  $this->_processedOidList = array();
160  $this->_currentOid = $this->_startOid;
161  $this->_currentDepth = 0;
162  }
163 
164  /**
165  * Checks if current position is valid
166  */
167  public function valid() {
168  return !$this->_end;
169  }
170 
171  /**
172  * Add oids to the processing queue.
173  * @param $oidList An array of oids.
174  * @param $depth The depth of the oids in the tree.
175  */
176  protected function addToQueue($oidList, $depth) {
177  for ($i=sizeOf($oidList)-1; $i>=0; $i--) {
178  $this->_oidList[] = array($oidList[$i], $depth);
179  }
180  }
181 }
