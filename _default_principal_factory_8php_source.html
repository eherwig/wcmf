73  public function getUserRoles(User $user, $useTempPermission=false) {
74  $permissionManager = ObjectFactory::getInstance('permissionManager');
75  if ($useTempPermission) {
76  $permissionManager->addTempPermission($this->_roleType, '', PersistenceAction::READ);
77  }
78 
79  // initialize role relation definition
80  if ($this->_roleRelationNames == null) {
81  $this->_roleRelationNames = array();
82  $mapper = $user->getMapper();
83  foreach ($mapper->getRelationsByType($this->_roleType) as $relation) {
84  $this->_roleRelationNames[] = $relation->getOtherRole();
85  }
86  }
87 
88  foreach ($this->_roleRelationNames as $roleName) {
89  $user->loadChildren($roleName);
90  }
91 
92  if ($useTempPermission) {

94  }
95 
96  // TODO add role nodes from addedNodes array
