44  /**
45  * @see Controller::initialize()
46  */
47  public function initialize(Request $request, Response $response) {
48  // construct oid from className and id
49  if ($request->hasValue('className') && $request->hasValue('id')) {
50  $oid = new ObjectId($request->getValue('className'), $request->getValue('id'));
51  $request->setValue('oid', $oid->__toString());
52  }
53  // construct sourceOid from className and sourceId
54  if ($request->hasValue('className') && $request->hasValue('sourceId')) {
55  $sourceOid = new ObjectId($request->getValue('className'), $request->getValue('sourceId'));
56  $request->setValue('sourceOid', $sourceOid->__toString());
57  }
58  // construct oid, targetOid from sourceOid, relation and targetId
59  if ($request->hasValue('sourceOid') && $request->hasValue('relation') && $request->hasValue('targetId')) {
60  $sourceOid = ObjectId::parse($request->getValue('sourceOid'));
61  $relatedType = $this->getRelatedType($sourceOid, $request->getValue('relation'));
62  $targetOid = new ObjectId($relatedType, $request->getValue('targetId'));
63  $request->setValue('targetOid', $targetOid->__toString());
64  // non-collection requests
65  $request->setValue('oid', $targetOid->__toString());
66  }
67  parent::initialize($request, $response);
68  }

70  /**
71  * @see Controller::validate()
72  */
73  protected function validate() {
74  $request = $this->getRequest();
75  $response = $this->getResponse();
76  if ($request->hasValue('className') &&
77  !$this->getPersistenceFacade()->isKnownType($request->getValue('className')))
78  {
79  $response->addError(ApplicationError::get('PARAMETER_INVALID',
80  array('invalidParameters' => array('className'))));
81  return false;
82  }
83  if ($request->hasHeader('Position')) {
84  $position = $request->getHeader('Position');
85  if (!preg_match('/^before /', $position) && !preg_match('/^last$/', $position)) {
86  $response->addError(ApplicationError::get('PARAMETER_INVALID',
87  array('invalidParameters' => array('Position'))));
88  return false;
89  }
90  }
91  // do default validation
92  return parent::validate();
93  }

129  /**
130  * Read objects that are related to another object
131  *
132  * | Parameter | Description
133  * |------------------|-------------------------
134  * | _in_ `language` | The language of the returned objects
135  * | _in_ `sourceId` | Id of the object to which the returned objects are related (determines the object id together with _className_)
136  * | _in_ `className` | The type of the object defined by _sourceId_
137  * | _in_ `relation` | Name of the relation to the object defined by _sourceId_ (determines the type of the returned objects)
138  * | _in_ `sortBy` | _?sortBy=+foo_ for sorting the list by foo ascending or
139  * | _in_ `sort` | _?sort(+foo)_ for sorting the list by foo ascending
140  * | _in_ `limit` | _?limit(10,25)_ for loading 10 objects starting from position 25
141  * | _out_ | List of objects
142  */
143  public function readInRelation() {
144  $request = $this->getRequest();
145 
146  // rewrite query if querying for a relation
147  $relationName = $request->getValue('relation');
148 
149  // set the query
150  $sourceOid = ObjectId::parse($request->getValue('sourceOid'));
151  $persistenceFacade = $this->getPersistenceFacade();
152  $sourceNode = $persistenceFacade->load($sourceOid);
153  if ($sourceNode) {
154  $query = NodeUtil::getRelationQueryCondition($sourceNode, $relationName);
155  $request->setValue('query', $query);
156 
157  // set the class name
158  $mapper = $sourceNode->getMapper();
159  $relation = $mapper->getRelation($relationName);
160  $otherType = $relation->getOtherType();
161  $request->setValue('className', $otherType);
162 
163  // set default order
164  if (!$request->hasValue('sortFieldName')) {
165  $otherMapper = $persistenceFacade->getMapper($otherType);
166  $sortkeyDef = $otherMapper->getSortkey($relation->getThisRole());
167  if ($sortkeyDef != null) {
168  $request->setValue('sortFieldName', $sortkeyDef['sortFieldName']);
169  $request->setValue('sortDirection', $sortkeyDef['sortDirection']);
170  }
171  }
172  }
173 
174  // delegate to ListController
175  $subResponse = $this->executeSubAction('list');
176  $this->handleSubResponse($subResponse);
177  }

199  /**
200  * Create an object of a given type in the given relation
201  *
202  * | Parameter | Description
203  * |------------------|-------------------------
204  * | _in_ `language` | The language of the object
205  * | _in_ `className` | Type of object to create
206  * | _in_ `sourceId` | Id of the object to which the created objects are added (determines the object id together with _className_)
207  * | _in_ `relation` | Name of the relation to the object defined by _sourceId_ (determines the type of the created/added object)
208  * | _out_ | Created object data
209  *
210  * The object data is contained in POST content. If an existing object
211  * should be added, an `oid` parameter in the object data is sufficient.
212  */
213  public function createInRelation() {
214  $request = $this->getRequest();
215 
216  // create new object
217  $sourceOid = ObjectId::parse($request->getValue('sourceOid'));
218  $relatedType = $this->getRelatedType($sourceOid, $request->getValue('relation'));
219  $request->setValue('className', $relatedType);
220  $subResponseCreate = $this->executeSubAction('create');
221 
222  $targetOid = $subResponseCreate->getValue('oid');
223  $targetOidStr = $targetOid->__toString();
224 
225  // add new object to relation
226  $request->setValue('targetOid', $targetOidStr);
227  $request->setValue('role', $request->getValue('relation'));
228  $subResponse = $this->executeSubAction('associate');
229 
230  // add related object to subresponse similar to default update action
231  $persistenceFacade = $this->getPersistenceFacade();
232  $targetObj = $persistenceFacade->load($targetOid);
233  $subResponse->setValue('oid', $targetOid);
234  $subResponse->setValue($targetOidStr, $targetObj);
235  $subResponse->setStatus($subResponseCreate->getStatus());
236 
237  // in case of success, return object only
238  $oidStr = $subResponse->hasValue('oid') ? $subResponse->getValue('oid')->__toString() : '';
239  $this->handleSubResponse($subResponse, $oidStr);
240  }

254  public function update() {
255  $request = $this->getRequest();
256 
257  $oidStr = $this->getFirstRequestOid();
258  $isOrderRequest = $request->hasValue('referenceOid');
259  if ($isOrderRequest) {
260  // change order in all objects of the same type
261  $request->setValue('insertOid', $this->getFirstRequestOid());
262 
263  // delegate to SortController
264  $subResponse = $this->executeSubAction('moveBefore');
265 
266  // add sorted object to subresponse similar to default update action
267  if ($subResponse->getStatus() == 200) {
268  $oid = ObjectId::parse($oidStr);
269  $persistenceFacade = $this->getPersistenceFacade();
270  $object = $persistenceFacade->load($oid);
271  $subResponse->setValue('oid', $oid);
272  $subResponse->setValue($oidStr, $object);
273  }
274  }
275  else {
276  // delegate to SaveController
277  $subResponse = $this->executeSubAction('update');
278  }
279 
280  $this->handleSubResponse($subResponse, $oidStr);

292  * | _in_ `sourceId` | Id of the object to which the added object is related (determines the object id together with _className_)
293  * | _in_ `targetId` | Id of the object to be added to the relation (determines the object id together with _relation_)

296  * The object data is contained in POST content.

298  public function updateInRelation() {
299  $request = $this->getRequest();
300 
301  $isOrderRequest = $request->hasValue('referenceOid');
302  $request->setValue('role', $request->getValue('relation'));
303  if ($isOrderRequest) {
304  // change order in a relation
305  $request->setValue('containerOid', $request->getValue('sourceOid'));
306  $request->setValue('insertOid', $request->getValue('targetOid'));
307 
308  // delegate to SortController
309  $subResponse = $this->executeSubAction('insertBefore');
310  }
311  else {
312  // add existing object to relation
313  // delegate to AssociateController
314  $subResponse = $this->executeSubAction('associate');
315  if ($subResponse->getStatus() == 200) {
316  // and update object
317  // delegate to SaveController
318  $subResponse = $this->executeSubAction('update');
319  }
320  }
321 
322  // add related object to subresponse similar to default update action
323  if ($subResponse->getStatus() == 200) {
324  $targetOidStr = $request->getValue('targetOid');
325  $targetOid = ObjectId::parse($targetOidStr);
326  $persistenceFacade = $this->getPersistenceFacade();
327  $targetObj = $persistenceFacade->load($targetOid);
328  $subResponse->setValue('oid', $targetOid);
329  $subResponse->setValue($targetOidStr, $targetObj);
330  }
331 
332  $this->handleSubResponse($subResponse, $targetOidStr);
333  }

373  /**
374  * Create the actual response from the response resulting from delegating to
375  * another controller
376  * @param $subResponse The response returned from the other controller
377  * @param $oidStr Serialized object id of the object to return (optional)
378  */
379  protected function handleSubResponse(Response $subResponse, $oidStr=null) {
380  $response = $this->getResponse();
381  if (!$subResponse->hasErrors()) {
382  $response->clearValues();
383  if ($subResponse->hasValue('object')) {
384  $object = $subResponse->getValue('object');
385  if ($object != null) {
386  $response->setValue($object->getOID()->__toString(), $object);
387  }
388  }
389  if ($subResponse->hasValue('list')) {
390  $objects = $subResponse->getValue('list');
391  $response->setValues($objects);
392  }
393  if ($oidStr != null && $subResponse->hasValue($oidStr)) {
394  $object = $subResponse->getValue($oidStr);
395  $response->setValue($oidStr, $object);
396  if ($subResponse->getStatus() == 201) {
397  $this->setLocationHeaderFromOid($oidStr);
398  }
399  }
400  $response->setStatus($subResponse->getStatus());
401  }
402  else {
403  // in case of error, return default response
404  $response->setValues($subResponse->getValues());
405  $response->setStatus(400);
406  }
407  }
