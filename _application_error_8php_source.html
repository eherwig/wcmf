18 /**
19  * ApplicationError is used to signal errors that occur
20  * while processing a request.
21  *
22  * This class allows to use predefined errors by using the ApplicationError::get()
23  * method. Errors are defined in the following way:
24 
25  * @code
26  * $message = ObjectFactory::getInstance('message');
27  * define('GENERAL_ERROR', serialize(array('GENERAL_ERROR', ApplicationError::LEVEL_ERROR, 400,
28  * $message->getText('An unspecified error occured.')
29  * )));
30  * @endcode
31  *
32  * To use the error in the code:
33  *
34  * @code
35  * $error = ApplicationError::get('GENERAL_ERROR', $mySpecificErrorData);
36  * @endcode
37  *
38  * @author ingo herwig <ingo@wemove.com>
39  */
40 class ApplicationError {

126  */
127  public function __toString() {
128  $str = strtoupper($this->level).": ".$this->code." (".$this->statusCode.

136  /**
137  * Factory method for retrieving a predefined error instance.
138  * @param $code An error code
139  * @param $data Some error codes required to transmit further information
140  * to the client (optional, default: _null_)
141  * @return ApplicationError
142  */
143  public static function get($code, $data=null) {
144  self::predefine();
145  if (defined($code)) {
146  $def = unserialize(constant($code));
147  $error = new ApplicationError($def[0], $def[1], $def[2], $def[3], $data);
148  return $error;
149  }
150  else {
151  throw new IllegalArgumentException("The error code '".$code."' is not defined");
152  }
153  }
154 
155  /**
156  * Factory method for creating a general error instance.
157  * @param $message Error message
158  * @param $statusCode HTTP status code (optional, default: _DEFAULT_ERROR_STATUS_)
159  * @return ApplicationError
160  */
161  public static function getGeneral($message, $statusCode=self::DEFAULT_ERROR_STATUS) {
162  $error = new ApplicationError('GENERAL_ERROR', ApplicationError::LEVEL_ERROR, $statusCode, $message);
163  return $error;
164  }

166  /**
167  * Factory method for transforming an exception into an ApplicationError instance.
168  * @param $ex Exception
169  * @return ApplicationError
170  */
171  public static function fromException(\Exception $ex) {
172  if ($ex instanceof ApplicationException) {
173  return $ex->getError();
174  }
175  return self::getGeneral($ex->getMessage());
176  }
