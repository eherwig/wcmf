11 namespace wcmf\lib\i18n;
12 
13 use wcmf\lib\core\ObjectFactory;
14 
15 /**
16  * Message is used to get localized messages to be used in the user interface.
17  * The localization directory must be given in the configuration value
18  * 'localeDir' in section 'application'.
19  * Inside this directory there must be a messages_$lang.php files for each
20  * language defining the translation for each message in an associative array
21  * called $messages.
22  *
23  * For example the messages_de file could have the following content:
24  * @code
25  * $messages = array(
26  * 'up' => 'hoch',
27  * 'down' => 'runter',
28  * ...
29  * );
30  * @endcode
31  * @note The language is determined in one of 3 ways (in this order):
32  * -# use the value of the lang parameter passed to Message::get()
33  * -# use the value of the configuration value 'language' in section 'application'
34  * -# use the value of the global variable $_SERVER['HTTP_ACCEPT_LANGUAGE']
35  *
36  * @author ingo herwig <ingo@wemove.com>
37  */
38 class Message {
39 
40  private static $localeDir;
41  private static $language;
42  private static $initialized = false;
43 
44  /**
45  * Cache for loaded translations
46  */
47  private static $translations = array();
48 
49  /**
50  * Initialize static members
51  */
52  private static function initialize() {
53  $config = ObjectFactory::getConfigurationInstance();
54  self::$localeDir = $config->getDirectoryValue('localeDir', 'application');
55  if ($config->hasValue('language', 'application')) {
56  self::$language = $config->getValue('language', 'application');
57  }
58  else if (isset($_SERVER['HTTP_ACCEPT_LANGUAGE'])) {
59  self::$language = $_SERVER['HTTP_ACCEPT_LANGUAGE'];
60  }
61  self::$initialized = true;
62  }
63 
64  /**
65  * Get a localized string.
66  * @note It is not recommended to use this method with concatenated strings because this
67  * restricts the positions of words in translations. E.g. 'She was born in %0% on %1%'
68  * translates to the german sentence 'Sie wurde am %1% in %0% geboren' with the variables
69  * flipped.
70  * @param $message The message to translate (%0%, %1%, ... will be replaced by given parameters).
71  * @param $parameters An array of values for parameter substitution in the message.
72  * @param $lang The language (optional, default: '')
73  * @return The localized string
74  */
75  public static function get($message, $parameters=null, $lang='') {
76  if (!self::$initialized) {
77  self::initialize();
78  }
79 
80  // get the translations
81  $lang = strlen($lang) == 0 ? self::$language : $lang;
82  $translations = self::getTranslations($lang);
83  $localizedMessage = '';
84  if (isset($translations[$message])) {
85  $localizedMessage = $translations[$message];
86  }
87  if (strlen($localizedMessage) == 0) {
88  $localizedMessage = $message;
89  }
90 
91  // replace parameters
92  preg_match_all("/%([0-9]+)%/", $localizedMessage, $matches);
93  $matches = $matches[1];
94  for ($i=0; $i<sizeof($matches); $i++) {
95  $matches[$i] = '/\%'.$matches[$i].'\%/';
96  }
97  sort($matches);
98  if (sizeof($matches) > 0 && is_array($parameters)) {
99  $localizedMessage = preg_replace($matches, $parameters, $localizedMessage);
100  }
101  return $localizedMessage;
102  }
103 
104  /**
105  * The getAll() method is used to get a localized list of all defined strings.
106  * See Message::get() for more information.
107  * @param $lang The language (optional, default: '')
108  * @return An array of localized string
109  */
110  public static function getAll($lang='') {
111  if (!self::$initialized) {
112  self::initialize();
113  }
114  // get the translations
115  $translations = self::getTranslations($lang);
116  return $translations;
117  }
118 
119  /**
120  * Get all translations for a language.
121  * @param $lang The language (optional, default: '')
122  * @return The translations as associative array
123  */
124  private static function getTranslations($lang) {
125  if (!isset(self::$translations[$lang])) {
126  $messageFile = self::$localeDir."/messages_".$lang.".php";
127  if (file_exists($messageFile)) {
128  require_once($messageFile);
129  // store for later reference
130  self::$translations[$lang] = ${"messages_$lang"};
131  }
132  else {
133  self::$translations[$lang] = array();
134  }
135  }
136  return self::$translations[$lang];
137  }
138 }
