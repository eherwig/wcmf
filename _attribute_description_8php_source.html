56  /**
57  * Check if this attribute is tagged with the given application specific tags
58  * @param $tags An array of tags that the attribute should match. Empty array results in true the given matchMode (default: empty array)
59  * @param $matchMode One of 'all', 'none', 'any', defines how the attribute's tags should match the given tags (default: 'all')
60  * @return True if the attribute has all data types, false else
61  */
62  public function matchTags(array $tags=array(), $matchMode='all') {
63  $numGivenTags = sizeof($tags);
64  if (sizeof($numGivenTags) == 0) {
65  return true;
66  }
67  $result = true;
68  $diff = sizeof(array_diff($tags, $this->tags));
69  switch ($matchMode) {
70  case 'all':
71  $result = ($diff == 0);
72  break;
73  case 'none':
74  $result = ($diff == $numGivenTags);
75  break;
76  case 'any':
77  $result = ($diff < $numGivenTags);
78  break;
79  }
80  return $result;
81  }
