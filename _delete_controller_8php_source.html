22 /**
23  * DeleteController is used to delete Node instances.
24  *
25  * The controller supports the following actions:
26  *
27  * <div class="controller-action">
28  * <div> __Action__ _default_ </div>
29  * <div>
30  * Delete the given Node instance.
31  * | Parameter | Description
32  * |-----------------------|-------------------------
33  * | _in_ / _out_ `oid` | The object id of the Node to delete
34  * | __Response Actions__ | |
35  * | `ok` | In all cases
36  * </div>
37  * </div>
38  *
39  * @author ingo herwig <ingo@wemove.com>
40  */
41 class DeleteController extends Controller {
42 
43  /**
44  * @see Controller::validate()
45  */
46  protected function validate() {
47  if (!$this->checkLanguageParameter()) {
48  return false;
49  }
50  // do default validation
51  return parent::validate();
52  }
53 
54  /**
55  * @see Controller::doExecute()
56  */
57  protected function doExecute() {
58  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
59  $request = $this->getRequest();
60  $response = $this->getResponse();
61 
62  $oid = ObjectId::parse($request->getValue('oid'));
63 
64  // start the transaction
65  $transaction = $persistenceFacade->getTransaction();
66  $transaction->begin();
67  try {
68  // load the doomed node
69  if ($oid) {
70  $doomedNode = $persistenceFacade->load($oid, BuildDepth::SINGLE);
71  if ($doomedNode == null) {
72  Log::warn("An object with oid ".$oid." is does not exist.", __CLASS__);
73  }
74  else {
75  if($this->confirmDelete($doomedNode)) {
76  // commit changes
77  $localization = ObjectFactory::getInstance('localization');
78  if ($this->isLocalizedRequest()) {
79  // delete the translations for the requested language
80  $localization->deleteTranslation($doomedNode->getOID(), $request->getValue('language'));
81  }
82  else {
83  // delete the real object data and all translations
84  $localization->deleteTranslation($doomedNode->getOID());
85  $doomedNode->delete();
86  }
87  // after delete
88  $this->afterDelete($oid);
89  }
90  }
91  }
92  $transaction->commit();
93  }
94  catch (PessimisticLockException $ex) {
95  $response->addError(ApplicationError::get('OBJECT_IS_LOCKED',
96  array('lockedOids' => array($oid->__toString()))));
97  $transaction->rollback();
98  }
99  catch (Exception $ex) {
100  $response->addError(ApplicationError::fromException($ex));
101  $transaction->rollback();
102  }
103 
104  $response->setValue('oid', $oid);
105  $response->setAction('ok');
106  }
107 
108  /**
109  * Confirm delete action on given Node.

111  * @param $node A reference to the Node to confirm.
112  * @return Boolean whether the Node should be deleted (default: _true_)
113  */
114  protected function confirmDelete($node) {
115  return true;
116  }

123  protected function afterDelete($oid) {}
