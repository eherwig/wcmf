22 /**
23  * StringQuery executes queries from a string representation. Queries are
24  * constructed in two possible ways:
25  * 1. Like WHERE clauses in SQL
26  * 2. As RQL queries (https://github.com/persvr/rql)
27  *
28  * Note the following rules:
29  * - Foreign key relations between different types do not need to be included in
30  * the query string.
31  * - Attributes have to be prepended with the type name (or in case of ambiguity
32  * the role name), e.g. Author.name instead of name.
33  *
34  * @note: The query does not search in objects, that are created inside the current transaction.
35  *
36  * The following example shows the usage:
37  *
38  * @code
39  * $queryStr = "Author.name LIKE '%ingo%' AND (Recipe.name LIKE '%Salat%' OR Recipe.portions = 4)";
40  * $query = new StringQuery('Author');
41  * $query->setConditionString($queryStr);
42  * $authorOIDs = $query->execute(false);
43  *
44  * $queryStr = "(Profile.keyword1=in=8,1|Profile.keyword2=in=8,2|Profile.keywords3=in=8,3)&Profile.salary=gte=1000";
45  * $query = new StringQuery('Profile');
46  * $query->setConditionString($queryStr);
47  * $authorOIDs = $query->execute(false);
48  * @endcode
49  *
50  * @author ingo herwig <ingo@wemove.com>
51  */

80  /**
81  * @see AbstractQuery::buildQuery()
82  */
83  protected function buildQuery($buildDepth, $orderby=null, PagingInfo $pagingInfo=null) {
84  $queryType = $this->getQueryType();
85  $mapper = self::getMapper($queryType);
86 
87  // create the attribute string (use the default select from the mapper,
88  // since we are only interested in the attributes)
89  $attributes = $buildDepth === false ? $mapper->getPkNames() : null;
90  $selectStmt = $mapper->getSelectSQL(null, null, $attributes, null, $pagingInfo, $this->getId());
91  if (!$selectStmt->isCached()) {
92  // initialize the statement
93  $selectStmt->quantifier(SelectStatement::QUANTIFIER_DISTINCT);
94 
95  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
96  $quoteIdentifierSymbol = $mapper->getQuoteIdentifierSymbol();
97  // get all referenced types/roles from the condition and translate
98  // attributes to column names
99  $conditionString = $this->condition;
100  $otherRoles = [];
101  $tokens = StringUtil::splitQuoted($conditionString, "/[\s=<>()!]+/", "'", true);
102  $operators = ['and', 'or', 'not', 'like', 'regexp', 'is', 'null', 'in'];
103  foreach ($tokens as $token) {
104  if (strlen($token) > 0) {
105  if (!in_array(strtolower($token), $operators)) {
106  // three possibilities left: token is
107  // 1. type or attribute (not allowed)
108  // 2. type.attribute
109  // 3. searchterm
110  if (!preg_match('/^\'|^[0-9]/', $token)) {
111  // token is no searchterm (does not start with a quote or a number)
112  $token = str_replace($quoteIdentifierSymbol, '', $token);
113  $pos = strpos($token, '.');
114  if ($pos > 0) {
115  // token is type/role.attribute
116  list($typeOrRole, $attribute) = explode('.', $token, 2);
117  // check if the token is a type
118  $fqType = $persistenceFacade->isKnownType($typeOrRole) ?
119  $persistenceFacade->getFullyQualifiedType($typeOrRole) : null;
120  if ($fqType == null || $fqType != $queryType) {
121  // find connection if the token does not match the queryType
122  if (!isset($otherRoles[$typeOrRole])) {
123  // find the path from the queryType to the other type/role
124  // (role is preferred)
125  $paths = NodeUtil::getConnections($queryType, $typeOrRole, null);
126  if (sizeof($paths) == 0) {
127  // fallback: search for type
128  $paths = NodeUtil::getConnections($queryType, null, $typeOrRole);
129  }
130  if (sizeof($paths) == 0) {
131  // no connection found
132  throw new IllegalArgumentException("There is no connection between '".$queryType."' and '".$typeOrRole."'.");
133  }
134  elseif (sizeof($paths) > 1) {
135  // more than one connection found
136  throw new IllegalArgumentException("There is more than one connection between '".$queryType."' and '".$typeOrRole."'. ".
137  "Try to use a role name for the target end.");
138  }
139  // exactly one connection (store it for later reference)
140  $otherRoles[$typeOrRole] = $paths[0];
141  }
142  // find the type of the referenced node
143  $path = $otherRoles[$typeOrRole];
144  $type = $path->getEndType();
145  }
146  else {
147  $type = $queryType;
148  }
149 
150  // map the attributes to columns
151  list($table, $column) = self::mapToDatabase($type, $attribute);
152  $conditionString = str_replace($quoteIdentifierSymbol.$attribute.$quoteIdentifierSymbol,
153  $quoteIdentifierSymbol.$column.$quoteIdentifierSymbol, $conditionString);
154  }
155  else {
156  throw new IllegalArgumentException("Please specify the type/role to that the attribute '".$token."' belongs: e.g. Author.name.");
157  }
158  }
159  }
160  }
161  }
162  if (strlen($conditionString)) {
163  $selectStmt->where($conditionString);
164  }
165 
166  // get relation conditions
167  // NOTE: temporarily created objects must be detached from the transaction
168  $tx = $persistenceFacade->getTransaction();
169  $rootNode = $persistenceFacade->create($queryType);
170  $tx->detach($rootNode->getOID());
171  foreach ($otherRoles as $typeOrRole => $pathDescription) {
172  $relationDescriptions = $pathDescription->getPath();
173  $parent = $rootNode;
174  foreach ($relationDescriptions as $relationDescription) {
175  $node = $persistenceFacade->create($relationDescription->getOtherType());
176  $tx->detach($node->getOID());
177  $parent->addNode($node, $relationDescription->getOtherRole());
178  $parent = $node;
179  }
180  // set the table name of the target node to the name that is
181  // referenced in the query condition
182  $node->setProperty(self::PROPERTY_TABLE_NAME, $typeOrRole);
183  }
184  $this->processObjectTemplate($rootNode, $selectStmt);
185  }
186 
187  // set orderby after all involved tables are known in order to
188  // prefix the correct table name
189  $this->processOrderBy($orderby, $selectStmt);
190 
191  // reset internal variables
192  $this->resetInternals();
193 
194  return $selectStmt;
195  }

197  /**
198  * Parse the given query encoded in RQL (https://github.com/persvr/rql), e.g.
199  * (Profile.keyword1=in=8,1|Profile.keyword2=in=8,2|Profile.keywords3=in=8,3)&Profile.yearlySalary=gte=1000
200  * @param $query RQL query string
201  * @return String
202  */
203  protected function parseRQL($query) {
204  if (strlen($query) > 0) {
205  // replace rql operators
206  $operatorMap = ['eq' => '=', 'ne' => '!=', 'lt' => '<', 'lte' => '<=',
207  'gt' => '>', 'gte' => '>=', 'in' => 'in', 'match' => 'regexp', '=' => '='];
208  $combineMap = ['|' => 'OR', '&' => 'AND'];
209  $queryType = $this->getQueryType();
210  $mapper = self::getMapper($queryType);
211  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
212  foreach ($operatorMap as $rqlOp => $sqlOp) {
213  // add '=' separator for letter-only operators
214  $operator = preg_match('/^[a-zA-Z]+$/', $rqlOp) ? '='.$rqlOp.'=' : $rqlOp;
215  $query = preg_replace_callback('/([^ <>!=|&\(\)]+) ?'.$operator.' ?([^ <>!=|&\(\)]+)/', function ($match)
216  use($rqlOp, $sqlOp, $mapper, $persistenceFacade) {
217  $typeAttr = $match[1];
218  $value = preg_replace('/^[\'"]|[\'"]$/', '', $match[2]);
219 
220  // try to determine value type
221  list($typeOrRole, $attribute) = explode('.', preg_replace('/[`"\'\(\)]/', '', $typeAttr), 2);
222  $attributeDef = null;
223  if ($persistenceFacade->isKnownType($typeOrRole) &&
224  $persistenceFacade->getFullyQualifiedType($typeOrRole) == $mapper->getType() && $mapper->hasAttribute($attribute)) {
225  $attributeDef = $mapper->getAttribute($attribute);
226  }
227  elseif ($mapper->hasRelation($typeOrRole)) {
228  $otherMapper = self::getMapper($mapper->getRelation($typeOrRole)->getOtherType());
229  $attributeDef = $otherMapper->hasAttribute($attribute) ? $otherMapper->getAttribute($attribute) : null;
230  }
231  $attributeType = $attributeDef != null ? strtolower($attributeDef->getType()) : null;
232  $isNumber = $attributeType == 'integer' || $attributeType == 'float';
233 
234  // strip slashes from regexp
235  if ($sqlOp === 'regexp') {
236  $value = preg_replace('/^\/|\/[a-zA-Z]*$/', '', $value);
237  }
238 
239  // handle null values
240  if ($value === 'null') {
241  $sqlOp = $sqlOp == '=' ? 'is' : 'is not';
242  $replace = $typeAttr.' '.$sqlOp.' null';
243  }
244  else {
245  // quote value according to type
246  $replace = $typeAttr.' '.$sqlOp.' ';
247  if ($rqlOp == 'in') {
248  $replace .= '('.($isNumber ? $value : join(',', array_map([$mapper, 'quoteValue'], explode(',', $value)))).')';
249  }
250  else {
251  $replace .= $isNumber ? $value : $mapper->quoteValue($value);
252  }
253  }
254 
255  return $replace;
256  }, $query);
257  }
258  foreach ($combineMap as $rqlOp => $sqlOp) {
259  $query = str_replace($rqlOp, ' '.$sqlOp.' ', $query);
260  }
261  }
262  return $query;
263  }
264 
265  /**
266  * Map a application type and value name to the appropriate database names
267  * @param $type The type to map
268  * @param $valueName The name of the value to map
269  * @return An array with the table and column name or null if no mapper is found
270  */
271  protected static function mapToDatabase($type, $valueName) {
272  $mapper = self::getMapper($type);
273  $attributeDescription = $mapper->getAttribute($valueName);
274 
275  $table = $mapper->getRealTableName();
276  $column = $attributeDescription->getColumn();
277  return [$table, $column];
278  }
