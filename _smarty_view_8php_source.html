174  /**
175  * @see View::getTemplate()
176  */
177  public static function getTemplate($controller, $context, $action) {
178  if (self::$_actionKeyProvider == null) {
179  self::$_actionKeyProvider = new ConfigActionKeyProvider();
180  self::$_actionKeyProvider->setConfigSection('views');
181  }
182 
183  $actionKey = ActionKey::getBestMatch(self::$_actionKeyProvider, $controller, $context, $action);
184  if (Log::isDebugEnabled(__CLASS__)) {
185  Log::debug('SmartyView::getTemplate: '.$controller."?".$context."?".$action.' -> '.$actionKey, __CLASS__);
186  }
187  // get corresponding view
188  $config = ObjectFactory::getConfigurationInstance();
189  try {
190  $view = $config->getValue($actionKey, 'views', false);
191  } catch (\Exception $ex) {
192  throw new ConfigurationException("No view defined for ".$controller."?".$context."?".$action);
193  }
194  return $view;
195  }
