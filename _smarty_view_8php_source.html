159  public function render($tplFile, $cacheId=null, $cacheLifetime=null, $display=true) {
160  $oldCacheLifetime = $this->view->cache_lifetime;
161  if ($cacheLifetime !== null) {

164  if ($display) {

166  }

169  }

172  }

210  /**

213  public static function getTemplate($controller, $context, $action) {
214  $config = ObjectFactory::getInstance('configuration');
215  if (self::$actionKeyProvider == null) {
216  self::$actionKeyProvider = new ConfigActionKeyProvider($config, 'views');
217  }
218 
219  $actionKey = ActionKey::getBestMatch(self::$actionKeyProvider, $controller, $context, $action);
220  if (self::$logger->isDebugEnabled()) {
221  self::$logger->debug('SmartyView::getTemplate: '.$controller."?".$context."?".$action.' -> '.$actionKey);
222  }
223  // get corresponding view
224  try {
225  $view = $config->getValue($actionKey, 'views', false);
226  }
227  catch (\Exception $ex) {
228  return false;
229  }
