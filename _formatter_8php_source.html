18  * Formatter is is the single entry point for request/response formatting.
19  * It chooses the configured formatter based on the format property of the message

26  /**
27  * Deserialize Request data into objects.
28  * @param $request A reference to the Request instance
29  */
30  public static function deserialize(Request $request) {
31 
32  // get the formatter that should be used for this request format
33  $format = $request->getFormat();
34  if ($format == null) {
35  // the format must be given!
36  throw new ConfigurationException("No content format defined for ".$request->__toString());
37  }
38  $format->deserialize($request);
39  }
40 
41  /**
42  * Serialize Response according to the output format.
43  * @param $response A reference to the Response instance
44  */
45  public static function serialize(Response $response) {
46  // get the formatter that should be used for this response format
47  $format = $response->getFormat();
48  if ($format == null) {
49  // the response format must be given!
50  throw new ConfigurationException("No response format defined for ".$response->__toString());
51  }
52 
53  if (!headers_sent()) {
54  header('HTTP/1.1 '.$response->getStatus());
55  header("Content-Type: ".$format->getMimeType()."; charset=utf-8");
56  foreach ($response->getHeaders() as $name => $value) {
57  header($name.': '.$value);
58  }
59  }
60  $format->serialize($response);
61  }
62 }
