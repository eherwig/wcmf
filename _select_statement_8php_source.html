37  /**
38  * Get the SelectStatement instance with the given id.
39  * If the id equals SelectStatement::NO_CACHE or is not cached, a new one will be created.
40  * @param $mapper RDBMapper instance used to retrieve the database adapter
41  * @param $id The statement id (optional, default: _SelectStatement::NO_CACHE_)
42  * @return SelectStatement
43  */
44  public static function get(RDBMapper $mapper, $id=self::NO_CACHE) {
45  $cache = ObjectFactory::getInstance('staticCache');
46  $cacheSection = self::getCacheSection($mapper->getType());
47  $cacheId = self::getCacheId($id);
48  if ($id == self::NO_CACHE || !$cache->exists($cacheSection, $cacheId)) {
49  $selectStmt = new SelectStatement($mapper, $id);
50  }
51  else {
52  $selectStmt = $cache->get($cacheSection, $cacheId);
53  }
54  return $selectStmt;
55  }

139  /**
140  * Execute a count query and return the row count
141  * @return Integer
142  */
143  public function getRowCount() {
144  $adapter = $this->getAdapter();
145  $sql = preg_replace('/^SELECT (.+) FROM/i', 'SELECT COUNT(*) AS nRows FROM', $this->getSql());
146  $sql = preg_replace('/LIMIT [0-9]+ OFFSET [0-9]+$/i', '', $sql);
147  $sql = preg_replace('/ORDER BY .+$/i', '', $sql);
148  $stmt = $adapter->getDriver()->getConnection()->prepare(trim($sql));
149  $result = $stmt->execute($this->getParameters())->getResource();
150  $row = $result->fetch();
151  $nRows = $row['nRows'];
152  return $nRows;
153  }

165  /**
166  * @see Select::join()
167  */
168  public function join($name, $on, $columns=self::SQL_STAR, $type=self::JOIN_INNER) {
169  // prevent duplicate joins
170  foreach ($this->joins->getJoins() as $join) {
171  if ($join['name'] == $name) {
172  return $this;
173  }
174  }
175  return parent::join($name, $on, $columns, $type);

178  /**
179  * Add columns to the statement
180  * @param $columns Array of columns (@see Select::columns())
181  * @param $joinName The name of the join to which the columns belong
182  */
183  public function addColumns(array $columns, $joinName=null) {
184  if ($joinName === null) {
185  // add normal column
186  $this->columns = $this->columns + $columns;
187  }
188  else {
189  // add column to join
190  $joins = array();
191  foreach ($this->joins->getJoins() as $join) {
192  if ($join['name'] == $joinName) {
193  $join['columns'] += $columns;
194  }
195  $joins[] = $join;
196  }
197  $this->joins->reset();
198  foreach ($joins as $join) {
199  parent::join($join['name'], $join['on'], $join['columns'], $join['type']);
200  }
201  }
202  }
