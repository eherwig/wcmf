391  /**
392  * Get the format instance for the given mime type.
393  * @param $mimeType The mime type
394  * @return Format instance
395  */
396  protected static function getFormatFromMimeType($mimeType) {
397  $formats = ObjectFactory::getInstance('formats');
398  $firstFormat = null;
399  foreach ($formats as $name => $instance) {
400  $firstFormat = $firstFormat == null ? $name : $firstFormat;
401  if (strpos($mimeType, $instance->getMimeType()) !== false) {
402  return $instance;
403  }
404  }
405  if (!isset($formats[$firstFormat])) {
406  throw new ConfigurationException("Configuration section 'Formats' does not contain a format definition for: ".$mimeType);
407  }
408  return $formats[$firstFormat];
409  }
