19  * PersistentObjectProxy is proxy for an PersistentObject instance.
20  *
21  * @author ingo herwig <ingo@wemove.com>
22  */
23 class PersistentObjectProxy {
24 
25  protected $_oid = null; // object identifier
26  protected $_realSubject = null; // the PersistentObject instance
27 
28  /**
29  * Constructor.
30  * @param $oid The object id of the PersistentObject instance.
31  */
32  public function __construct(ObjectId $oid) {
33  $this->_oid = $oid;
34  }
35 
36  /**
37  * Create a PersistenceProxy instance from a PersistentObject. This is useful
38  * if you want to prevent automatic loading of the subject if it is already loaded.
39  * Returns the argument, if already an PersistentObjectProxy instance.
40  * @param $object The PersistentObject or PersistentObjectProxy
41  * @return PersistentObjectProxy
42  */
43  public static function fromObject($object) {
44  if ($object instanceof PersistentObjectProxy) {
45  return $object;
46  }
47  else if ($object instanceof PersistentObject) {
48  $proxy = new PersistentObjectProxy($object->getOID());
49  $proxy->_realSubject = $object;
50  return $proxy;
51  }
52  else {
53  throw new IllegalArgumentException("Cannot create proxy from unknown object");
54  }
55  }
56 
57  /**
58  * Get the object id of the PersistentObject.
59  * @return ObjectId
60  */
61  public function getOID() {
62  return $this->_oid;
63  }
64 
65  /**
66  * Get the type of the PersistentObject.
67  * @return String
68  */
69  public function getType() {
70  return $this->_oid->getType();
71  }
72 
73  /**
74  * Get the value of a named item.
75  * @param $name The name of the item to query.
76  * @return The value of the item / null if it doesn't exits.
77  */
78  public function getValue($name) {
79  // return pk values as they are parts of the oid
80  $mapper = ObjectFactory::getInstance('persistenceFacade')->getMapper($this->getType());
81  $pkNames = $mapper->getPkNames();
82  for ($i=0, $count=sizeof($pkNames); $i<$count; $i++) {
83  if ($name == $pkNames[$i]) {
84  $ids = $this->_oid->getId();
85  return $ids[$i];
86  }
87  }
88  return $this->__call('getValue', array($name));
89  }
90 
91  /**
92  * Get the PersistentObject instance.
93  * @return PersistentObject
94  */
95  public function getRealSubject() {
96  if ($this->_realSubject == null) {
97  $this->resolve();
98  }
99  return $this->_realSubject;
100  }
101 
102  /**
103  * Delegate method call to the instance.
104  */
105  public function __call($name, array $arguments) {
106  if ($this->_realSubject == null) {
107  $this->resolve();
108  }
109  return call_user_func_array(array($this->_realSubject, $name), $arguments);
110  }
111 
112  /**
113  * Load the PersistentObject instance. Use this method if the subject should be loaded
114  * with a depth greater than BuildDepth::SINGLE
115  * @param $buildDepth One of the BUILDDEPTH constants or a number describing the number of generations to build
116  * (default: _BuildDepth::SINGLE_)
117  */
118  public function resolve($buildDepth=BuildDepth::SINGLE) {
119  $this->_realSubject = ObjectFactory::getInstance('persistenceFacade')->load($this->_oid, $buildDepth);
120  if ($this->_realSubject == null) {
121  throw new PersistenceException("Could not resolve oid: ".$this->_oid);
122  }
123  }
124 
125  /**
126  * Get a string representation of the instance.
127  * @return String
128  */
129  function __toString() {
130  return 'Proxy_'.$this->_oid->__toString();
131  }
132 }
