16 /**
17  * LoggingController is used to log a message in the backend log.
18  *
19  * The controller supports the following actions:
20  *
21  * <div class="controller-action">
22  * <div> __Action__ _default_ </div>
23  * <div>
24  * Log the message.
25  * | Parameter | Description
26  * |------------------------|-------------------------
27  * | _in_ `type` | The type of message. Must be one of: _DEBUG_, _INFO_, _WARNING_, _ERROR_, _FATAL_
28  * | _in_ `message` | The message
29  * | __Response Actions__ | |
30  * | `ok` | In all cases
31  * </div>
32  * </div>
33  *
34  * @author ingo herwig <ingo@wemove.com>
35  */
36 class LoggingController extends Controller {
37 
38  /**
39  * @see Controller::doExecute()
40  */
41  protected function doExecute() {
42 
43  $request = $this->getRequest();
44  $response = $this->getResponse();
45 
46  $logType = $request->getValue('type');
47  $message = $request->getValue('message');
48  switch($logType) {
49  case 'TRACE':
50  Log::trace($message, __CLASS__);
51  break;
52 
53  case 'DEBUG':
54  Log::debug($message, __CLASS__);
55  break;
56 
57  case 'INFO':
58  Log::info($message, __CLASS__);
59  break;
60 
61  case 'WARNING':
62  Log::warn($message, __CLASS__);
63  break;
64 
65  case 'ERROR':
66  Log::error($message, __CLASS__);
67  break;
68 
69  case 'FATAL':
70  Log::fatal($message, __CLASS__);
71  break;
72 
73  default:
74  Log::error("Unknown log message type: ".$logType, __CLASS__);
75  }
76 
77  $response->setAction('ok');
78  }
79 }
