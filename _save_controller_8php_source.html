27 /**
28  * SaveController is a controller that saves Node data.
29  *
30  * The controller supports the following actions:
31  *
32  * <div class="controller-action">
33  * <div> __Action__ _default_ </div>
34  * <div>
35  * Save the given Node values.
36  * | Parameter | Description
37  * |------------------------|-------------------------
38  * | _in_ / _out_ | Key/value pairs of serialized object ids and PersistentObject instances to save
39  * | _in_ `uploadDir` | The directory where attached files should be stored on the server (optional) (see SaveController::getUploadDir())
40  * | _out_ `oid` | The object id of the last newly created object
41  * | __Response Actions__ | |
42  * | `ok` | In all cases
43  * </div>
44  * </div>
45  *
46  * Errors concerning single input fields are added to the session (the keys are the input field names)
47  *
48  * @author ingo herwig <ingo@wemove.com>
49  */
50 class SaveController extends Controller {

65  /**
66  * @see Controller::doExecute()
67  */
68  protected function doExecute() {
69  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
70  $session = ObjectFactory::getInstance('session');
71  $request = $this->getRequest();
72  $response = $this->getResponse();
73 
74  // array of all involved nodes
75  $nodeArray = array();
76 
77  // array of oids to actually save
78  $saveOids = array();
79 
80  // array of oids to insert (subset of saveOids)
81  $insertOids = array();
82 
83  // start the persistence transaction
84  $transaction = $persistenceFacade->getTransaction();
85  $transaction->begin();
86  try {
87  // store all invalid parameters for later reference
88  $invalidOids = array();
89  $invalidAttributeNames = array();
90  $invalidAttributeValues = array();
91  $needCommit = false;
92  $curNode = null;
93 
94  // iterate over request values and check for oid/object pairs
95  $saveData = $request->getValues();
96  foreach ($saveData as $curOidStr => $curRequestObject) {
97  if ($curRequestObject instanceof PersistentObject && ($curOid = ObjectId::parse($curOidStr)) != null
98  && $curRequestObject->getOID() == $curOid) {
99 
100  // if the oid is a dummy, the object is supposed to be created instead of updated
101  $isNew = $curOid->containsDummyIds();
102 
103  // iterate over all values given in the node
104  $mapper = $curRequestObject->getMapper();
105  $pkValueNames = $curRequestObject->getPkNames();
106  foreach ($curRequestObject->getValueNames() as $curValueName) {
107  // check if the attribute exists
108  if ($mapper && !$mapper->hasAttribute($curValueName) && !$mapper->hasRelation($curValueName)) {
109  $invalidAttributeNames[] = $curValueName;
110  }
111  // ignore primary key values, because they are immutable
112  if (!$isNew && in_array($curValueName, $pkValueNames)) {
113  continue;
114  }
115 
116  $curRequestValue = $curRequestObject->getValue($curValueName);
117 
118  // save uploaded file/ process array values
119  $isFile = false;
120  if (is_array($curRequestValue)) {
121  if ($this->isFileUpload($curRequestValue)) {
122  // save file
123  $filename = $this->saveUploadFile($curOid, $curValueName, $curRequestValue);
124  if ($filename != null) {
125  // success with probably altered filename
126  $curRequestValue = $filename;
127  }
128  $isFile = true;
129  }
130  else {
131  // no upload
132  // connect array values to a comma separated string if it's no a relation
133  if (!$mapper->hasRelation($curValueName)) {
134  $curRequestValue = join($curRequestValue, ",");
135  }
136  }
137  }
138 
139  // get the requested node
140  // see if we have already handled values of the node before or
141  // if we have to initially load/create it
142  if (!isset($nodeArray[$curOidStr])) {
143  // load/create the node initially
144  if ($this->isLocalizedRequest()) {
145  // create a detached object, if this is a localization request in order to
146  // save it manually later
147  $curNode = $persistenceFacade->create($curOid->getType(), BuildDepth::SINGLE);
148  // don't store changes on the original object
149  $transaction->detach($curNode->getOID());
150  $curNode->setOID($curOid);
151  $nodeArray[$curOidStr] = &$curNode;
152  }
153  else {
154  if ($isNew) {
155  // create a new object, if this is an insert request. set the object id
156  // of the request object for correct assignement in save arrays
157  $curNode = $persistenceFacade->create($curOid->getType(), BuildDepth::SINGLE);
158  }
159  else {
160  // load the existing object, if this is a save request in order to merge
161  // the new with the existing values
162  $curNode = $persistenceFacade->load($curOid, BuildDepth::SINGLE);
163  }
164  $nodeArray[$curOidStr] = &$curNode;
165  }
166  // the node could not be created from the oid
167  if ($curNode == null) {
168  $invalidOids[] = $curOidStr;
169  continue;
170  }
171  }
172  else {
173  // take the existing node
174  $curNode = &$nodeArray[$curOidStr];
175  }
176 
177  // set data in node (prevent overwriting old image values, if no image is uploaded)
178  if (!$isFile || ($isFile && sizeof($curRequestValue) > 0)) {
179  try {
180  // validate the new value
181  $curNode->validateValue($curValueName, $curRequestValue);
182  if ($this->confirmSaveValue($curNode, $curValueName, $curRequestValue)) {
183  // set the new value
184  $oldValue = $curNode->getValue($curValueName);
185  $curNode->setValue($curValueName, $curRequestValue);
186  if ($oldValue != $curRequestValue) {
187  $needCommit = true;
188  }
189  }
190  }
191  catch(ValidationException $ex) {
192  $invalidAttributeValues[] = array('oid' => $curOidStr,
193  'parameter' => $curValueName, 'message' => $ex->getMessage());
194  // add error to session
195  $session->addError($curOidStr, $ex->getMessage());
196  }
197  }
198 
199  // add node to save array
200  if ($curNode->getState() != PersistentObject::STATE_CLEAN) {
201  // associative array to asure uniqueness
202  $saveOids[$curOidStr] = $curOidStr;
203  if ($isNew) {
204  $insertOids[$curOidStr] = $curOidStr;
205  }
206  }
207  }
208  }
209  }
210 
211  // add errors to the response
212  if (sizeof($invalidOids) > 0) {
213  $response->addError(ApplicationError::get('OID_INVALID',
214  array('invalidOids' => $invalidOids)));
215  }
216  if (sizeof($invalidAttributeNames) > 0) {
217  $response->addError(ApplicationError::get('ATTRIBUTE_NAME_INVALID',
218  array('invalidAttributeNames' => $invalidAttributeNames)));
219  }
220  if (sizeof($invalidAttributeValues) > 0) {
221  $response->addError(ApplicationError::get('ATTRIBUTE_VALUE_INVALID',
222  array('invalidAttributeValues' => $invalidAttributeValues)));
223  }
224 
225  // commit changes
226  if ($needCommit && !$response->hasErrors()) {
227  $localization = ObjectFactory::getInstance('localization');
228  $saveOids = array_keys($saveOids);
229  for ($i=0, $count=sizeof($saveOids); $i<$count; $i++) {
230  $curOidStr = $saveOids[$i];
231  $curObject = &$nodeArray[$curOidStr];
232  $curOid = $curObject->getOid();
233 
234  // ask for confirmation
235  if ($this->confirmSave($curObject)) {
236  $this->beforeSave($curObject);
237  if ($this->isLocalizedRequest()) {
238  if (isset($insertOids[$curOidStr])) {
239  // translations are only allowed for existing objects
240  $response->addError(ApplicationError::get('PARAMETER_INVALID',
241  array('invalidParameters' => array('language'))));
242  }
243  else {
244  // store a translation for localized data
245  $localization->saveTranslation($curObject, $request->getValue('language'));
246  }
247  }
248  $this->afterSave($curObject);
249  }
250  else {
251  // detach object if not confirmed
252  $transaction->detach($curOid);
253  }
254  }
255  $transaction->commit();
256  }
257  else {
258  $transaction->rollback();
259  }
260  }
261  catch (PessimisticLockException $ex) {
262  $lock = $ex->getLock();
263  $response->addError(ApplicationError::get('OBJECT_IS_LOCKED',
264  array('lockedOids' => array($lock->getObjectId()->__toString()))));
265  $transaction->rollback();
266  }
267  catch (OptimisticLockException $ex) {
268  $currentState = $ex->getCurrentState();
269  $response->addError(ApplicationError::get('CONCURRENT_UPDATE',
270  array('currentState' => $currentState)));
271  $transaction->rollback();
272  }
273  catch (Exception $ex) {
274  Log::error($ex, __CLASS__);
275  $response->addError(ApplicationError::fromException($ex));
276  $transaction->rollback();
277  }
278 
279  // return the saved nodes
280  foreach ($nodeArray as $oidStr => $node) {
281  $response->setValue($node->getOid()->__toString(), $node);
282  }
283 
284  // return oid of the lastly created node
285  if (sizeof($insertOids) > 0) {
286  $keys = array_keys($insertOids);
287  $lastCreatedNode = $nodeArray[array_pop($keys)];
288  $lastCreatedOid = $lastCreatedNode->getOid();
289  $response->setValue('oid', $lastCreatedOid);
290  }
291 
292  $response->setAction('ok');
293  }

295  /**
296  * Save uploaded file. This method calls checkFile which will prevent upload if returning false.
297  * @param $oid The ObjectId of the object to which the file is associated
298  * @param $valueName The name of the value to which the file is associated
299  * @param $data An assoziative array with keys 'name', 'type', 'tmp_name' as contained in the php $_FILES array.
300  * @return The final filename if the upload was successful, null on error
301  */
302  protected function saveUploadFile(ObjectId $oid, $valueName, array $data) {
303  $response = $this->getResponse();
304  if ($data['name'] != '') {
305  // upload request -> see if upload was succesfull
306  if ($data['tmp_name'] != 'none') {
307  // create FileUtil instance if not done already
308  if ($this->_fileUtil == null) {
309  $this->_fileUtil = new FileUtil();
310  }
311  // check if file was actually uploaded
312  if (!is_uploaded_file($data['tmp_name'])) {
313  $message = Message::get("Possible file upload attack: filename %0%.", array($data['name']));
314  $response->addError(ApplicationError::get('GENERAL_ERROR', array('message' => $message)));
315  return null;
316  }
317 
318  // get upload directory
319  $uploadDir = $this->getUploadDir($oid, $valueName);
320 
321  // get the name for the uploaded file
322  $uploadFilename = $uploadDir.$this->getUploadFilename($oid, $valueName, $data['name']);
323 
324  // check file validity
325  if (!$this->checkFile($oid, $valueName, $uploadFilename, $data['type'])) {
326  return null;
327  }
328 
329  // get upload parameters
330  $override = $this->shouldOverride($oid, $valueName, $uploadFilename);
331 
332  // upload file (mimeTypes parameter is set to null, because the mime type is already checked by checkFile method)
333  $filename = $this->_fileUtil->uploadFile($data, $uploadFilename, null, $override);
334  if (!$filename) {
335  $response->addError(ApplicationError::get('GENERAL_ERROR',
336  array('message' => $this->_fileUtil->getErrorMsg())));
337  return null;
338  }
339  else {
340  return $filename;
341  }
342  }
343  else {
344  $response->addError(ApplicationError::get('GENERAL_ERROR',
345  array('message' => Message::get("Upload failed for %0%.", array($data['name'])))));
346  return null;
347  }
348  }
349  return null;
350  }

403  /**
404  * Get the name of the directory to upload a file to and make shure that it exists.
405  * The default implementation will first look for a parameter 'uploadDir'
406  * and then, if it is not given, for an 'uploadDir'. _type_ key in the configuration file
407  * (section 'media') and finally for an 'uploadDir' key at the same place.
408  * @note subclasses will override this to implement special application requirements.
409  * @param $oid The ObjectId of the object which will hold the association to the file
410  * @param $valueName The name of the value which will hold the association to the file
411  * @return The directory name
412  */
413  protected function getUploadDir(ObjectId $oid, $valueName) {
414  $request = $this->getRequest();
415  if ($request->hasValue('uploadDir')) {
416  $uploadDir = FileUtil::realpath($request->getValue('uploadDir'));
417  }
418  else {
419  $config = ObjectFactory::getConfigurationInstance();
420  if (ObjectId::isValid($oid)) {
421  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
422  $type = $persistenceFacade->getSimpleType($oid->getType());
423  // check if uploadDir.type is defined in the configuration
424  if ($type && ($dir = $config->getDirectoryValue('uploadDir.'.$type, 'media')) !== false) {
425  $uploadDir = $dir;
426  }
427  else {
428  if(($dir = $config->getDirectoryValue('uploadDir', 'media')) !== false) {
429  $uploadDir = $dir;
430  }
431  }
432  }
433  }
434  // asure that the directory exists
435  FileUtil::mkdirRec($uploadDir);
436  return $uploadDir;
437  }
