42  /**
43  * Destructor
44  */
45  public function __destruct() {
46  // log resource usage
47  if (Log::isDebugEnabled(__CLASS__)) {
48  $timeDiff = microtime(true)-$this->_startTime;
49  $memory = number_format(memory_get_peak_usage()/(1024*1024), 2);
50  $msg = "Time[".round($timeDiff, 2)."s] Memory[".$memory."mb]";
51  if ($this->_initialRequest != null) {
52  $msg .= " Request[".$this->_initialRequest->getSender()."?".
53  $this->_initialRequest->getContext()."?".$this->_initialRequest->getAction()."]";
54  }
55  Log::debug($msg, __CLASS__);
56  }
57  ob_end_flush();
58  }

60  /**
61  * Initialize the application.
62  *
63  * @param $configPath The path where config files reside (as seen from main.php) (optional, default: 'config/')
64  * @param $mainConfigFile The main configuration file to use (optional, default: 'config.ini')
65  * @param $defaultController The controller to call if none is given in request parameters (optional, default: '')
66  * @param $defaultContext The context to set if none is given in request parameters (optional, default: '')
67  * @param $defaultAction The action to perform if none is given in request parameters (optional, default: 'login')
68  * @return Request instance representing the current HTTP request
69  */
70  public function initialize($configPath='../config/', $mainConfigFile='config.ini',
71  $defaultController='', $defaultContext='', $defaultAction='login') {
72 
73  Log::configure($configPath.'log4php.php');
74  $config = new InifileConfiguration($configPath);
75  $config->addConfiguration($mainConfigFile);
76  ObjectFactory::configure($config);
77 
78  // create the Request instance
79  $this->_initialRequest = ObjectFactory::getInstance('request');
80  $this->_initialRequest->initialize($defaultController, $defaultContext, $defaultAction);
81 
82  // initialize session
83  $session = ObjectFactory::getInstance('session');
84 
85  // clear errors
86  $session->clearErrors();
87 
88  // load user configuration
89  $authUser = $session->getAuthUser();
90  if ($authUser && strlen($authUser->getConfig()) > 0) {
91  $config->addConfiguration($authUser->getConfig(), true);
92  }
93 
94  // load event listeners
95  $listeners = $config->getValue('listeners', 'application');
96  foreach ($listeners as $key) {
97  $listener = ObjectFactory::getInstance($key);
98  }
99 
100  // set timezone
101  date_default_timezone_set($config->getValue('timezone', 'application'));
102 
103  // return the request
104  return $this->_initialRequest;
105  }
106 
107  /**
108  * Run the application with the given request
109  * @param $request
110  * @return Response instance
111  */
112  public function run(Request $request) {
113  // process the requested action
114  $response = ObjectFactory::getInstance('actionMapper')->processAction($request);
115 
116  // store the last successful request
117  $session = ObjectFactory::getInstance('session');
118  $session->set('lastRequest', $request);
119  return $response;
120  }
121 
122  /**
123  * Default exception handling method. Rolls back the transaction and
124  * executes 'failure' action.
125  * @param $exception The Exception instance
126  * @param $request The Request instance
127  */
128  public function handleException(\Exception $exception, Request $request=null) {
129  Log::error($exception->getMessage()."\n".$exception->getTraceAsString(), __CLASS__);
130 
131  // rollback current transaction
132  if (ObjectFactory::getConfigurationInstance() != null) {
133  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
134  $persistenceFacade->getTransaction()->rollback();
135 
136  // redirect to failure action
137  if ($request == null) {
138  $request = $this->_initialRequest;
139  }
140  $request->addError(ApplicationError::fromException($exception));
141  $request->setAction('failure');
142  ObjectFactory::getInstance('actionMapper')->processAction($request);
143  }
144  else {
145  throw $exception;
146  }
147  }
148 
149  /**
150  * This method is run as ob_start callback
151  * @note must be public
152  * @param $buffer The content to be returned to the client
153  * @return String
154  */
155  public function outputHandler($buffer) {
156  // log last error
157  $error = error_get_last();
158  if ($error !== NULL) {
159  $info = "Error: ".$error['message']." in ".$error['file']." on line ".$error['line'];
160  Log::error($info, __CLASS__);
161 
162  // suppress error message in browser
163  if (!$this->_debug) {
164  header('HTTP/1.1 500 Internal Server Error');
165  exit(0);
166  }
167  }
168  return $buffer;
169  }
