44  /**
45  * Initialize the application.
46  *
47  * @param $configPath The path where config files reside (as seen from main.php) (optional, default: 'config/')
48  * @param $mainConfigFile The main configuration file to use (optional, default: 'config.ini')
49  * @param $defaultController The controller to call if none is given in request parameters (optional, default: '')
50  * @param $defaultContext The context to set if none is given in request parameters (optional, default: '')
51  * @param $defaultAction The action to perform if none is given in request parameters (optional, default: 'login')
52  * @return Request instance representing the current HTTP request
53  */
54  public function initialize($configPath='../config/', $mainConfigFile='config.ini',
55  $defaultController='', $defaultContext='', $defaultAction='login') {
56 
57  Log::configure($configPath.'log4php.php');
58  $config = new InifileConfiguration($configPath);
59  $config->addConfiguration($mainConfigFile);
60  ObjectFactory::configure($config);
61 
62  // create the Request instance
63  $this->_initialRequest = Request::getDefault($defaultController, $defaultContext, $defaultAction);
64 
65  // initialize session
66  $session = ObjectFactory::getInstance('session');
67 
68  // clear errors
69  $session->clearErrors();
70 
71  // load user configuration
72  $permissionManager = ObjectFactory::getInstance('permissionManager');
73  $authUser = $permissionManager->getAuthUser();
74  if ($authUser && strlen($authUser->getConfig()) > 0) {
75  $config->addConfiguration($authUser->getConfig(), true);
76  }
77 
78  // load event listeners
79  $listeners = $config->getValue('listeners', 'application');
80  foreach ($listeners as $key) {
81  $listener = ObjectFactory::getInstance($key);
82  }
83 
84  // set timezone
85  date_default_timezone_set($config->getValue('timezone', 'application'));
86 
87  // return the request
88  return $this->_initialRequest;

91  /**
92  * Run the application with the given request
93  * @param $request
94  * @return Response instance
95  */
96  public function run(Request $request) {
97  // process the requested action
98  $response = ObjectFactory::getInstance('actionMapper')->processAction($request);
99 
100  // store the last successful request
101  $session = ObjectFactory::getInstance('session');
102  $session->set('lastRequest', $request);
103  return $response;
104  }

106  /**
107  * Default exception handling method. Rolls back the transaction and
108  * re-executes the last request (expected in the session variable 'lastRequest').
109  * @param $exception The Exception instance
110  * @param $request The Request instance
111  */
112  public function handleException(Exception $exception, Request $request=null) {
113  Log::error($exception->getMessage()."\n".$exception->getTraceAsString(), __CLASS__);
114 
115  // rollback current transaction
116  if (ObjectFactory::getConfigurationInstance() != null) {
117  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
118  $persistenceFacade->getTransaction()->rollback();
119 
120  // redirect to failure action
121  if ($request == null) {
122  $request = $this->_initialRequest;
123  }
124  $request->addError(ApplicationError::fromException($exception));
125  $request->setAction('failure');
126  ObjectFactory::getInstance('actionMapper')->processAction($request);
127  }
128  else {
129  throw $exception;
130  }
131  }

133  /**
134  * This method is called on script shutdown
135  * @note must be public
136  */
137  public function shutdown() {
138  // log resource usage
139  if (Log::isDebugEnabled(__CLASS__)) {
140  $timeDiff = microtime(true)-$this->_startTime;
141  $memory = number_format(memory_get_peak_usage()/(1024*1024), 2);
142  $msg = "Time[".round($timeDiff, 2)."s] Memory[".$memory."mb]";
143  if ($this->_initialRequest != null) {
144  $msg .= " Request[".$this->_initialRequest->getSender()."?".
145  $this->_initialRequest->getContext()."?".$this->_initialRequest->getAction()."]";
146  }
147  Log::debug($msg, __CLASS__);
148  }
149 
150  ob_end_flush();
151  }

153  /**
154  * This method is run as ob_start callback
155  * @note must be public
156  * @param $buffer The content to be returned to the client
157  * @return String
158  */
159  public function outputHandler($buffer) {
160  // log last error
161  $error = error_get_last();
162  if ($error !== NULL) {
163  $info = "Error: ".$error['message']." in ".$error['file']." on line ".$error['line'];
164  Log::error($info, __CLASS__);
165 
166  // suppress error message in browser
167  if (!$this->_debug) {
168  header('HTTP/1.1 500 Internal Server Error');
169  exit(0);
170  }
171  }
172  return $buffer;
173  }
