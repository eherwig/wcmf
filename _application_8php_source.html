47  /**
48  * Destructor
49  */
50  public function __destruct() {
51  // log resource usage
52  if (self::$logger->isDebugEnabled()) {
53  $timeDiff = microtime(true)-$this->startTime;
54  $memory = number_format(memory_get_peak_usage()/(1024*1024), 2);
55  $msg = "Time[".round($timeDiff, 2)."s] Memory[".$memory."mb]";
56  if ($this->request != null) {
57  $msg .= " Request[".$this->request->getSender()."?".
58  $this->request->getContext()."?".$this->request->getAction()."]";
59  }
60  $msg .= " URI[".$_SERVER['REQUEST_URI']."]";
61  self::$logger->debug($msg);
62  }
63  ob_end_flush();
64  }

66  /**
67  * Initialize the request.
68  *
69  * @param $defaultController The controller to call if none is given in request parameters (optional, default: '')
70  * @param $defaultContext The context to set if none is given in request parameters (optional, default: '')
71  * @param $defaultAction The action to perform if none is given in request parameters (optional, default: 'login')
72  * @return Request instance representing the current HTTP request
73  */
74  public function initialize($defaultController='', $defaultContext='', $defaultAction='login') {
75  $config = ObjectFactory::getInstance('configuration');
76  $this->debug = $config->getBooleanValue('debug', 'Application');
77 
78  // create the Request and Response instances
79  $this->request = ObjectFactory::getInstance('request');
80  $this->response = ObjectFactory::getInstance('response');
81  $this->request->setResponse($this->response);
82 
83  $this->request->initialize($defaultController, $defaultContext, $defaultAction);
84 
85  // initialize session
86  $session = ObjectFactory::getInstance('session');
87 
88  // load user configuration
89  $principalFactory = ObjectFactory::getInstance('principalFactory');
90  $authUser = $principalFactory->getUser($session->getAuthUser(), true);
91  if ($authUser && strlen($authUser->getConfig()) > 0) {
92  $config->addConfiguration($authUser->getConfig(), true);
93  }
94 
95  // load event listeners
96  $listeners = $config->getValue('listeners', 'application');
97  foreach ($listeners as $key) {
98  ObjectFactory::getInstance($key);
99  }
100 
101  // set timezone
102  date_default_timezone_set($config->getValue('timezone', 'application'));
103 
104  // return the request
105  return $this->request;
106  }

119  /**
120  * Default exception handling method. Rolls back the transaction and
121  * executes 'failure' action.
122  * @param $exception The Exception instance
123  */
124  public function handleException(\Exception $exception) {
125  // get error level
126  $logFunction = 'error';
127  if ($exception instanceof ApplicationException) {
128  $logFunction = $exception->getError()->getLevel() == ApplicationError::LEVEL_WARNING ?
129  'warn' : 'error';
130  }
131  self::$logger->$logFunction($exception);
132 
133  try {
134  if (ObjectFactory::getInstance('configuration') != null) {
135  // rollback current transaction
136  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
137  $persistenceFacade->getTransaction()->rollback();
138 
139  // redirect to failure action
140  if ($this->request) {
141  $error = ApplicationError::fromException($exception);
142  $this->request->addError($error);
143  $this->response->addError($error);
144  $this->request->setAction('failure');
145  $this->response->setAction('failure');
146  $this->response->setStatus($error->getStatusCode());
147  ObjectFactory::getInstance('actionMapper')->processAction($this->request, $this->response);
148  return;
149  }
150  }
151  throw $exception;
152  }
153  catch (Exception $ex) {
154  self::$logger->error($ex->getMessage()."\n".$ex->getTraceAsString());
155  }
156  }
157 
158  /**
159  * This method is run as ob_start callback
160  * @note must be public
161  * @param $buffer The content to be returned to the client
162  * @return String
163  */
164  public function outputHandler($buffer) {
165  // log last error, if it's level is enabled
166  $error = error_get_last();
167  if ($error !== null && $error['type'] == E_ERROR) {
168  $errorStr = $error['file']."::".$error['line'].": ".$error['type'].": ".$error['message'];
169  self::$logger->error($errorStr);
170  // suppress error message in browser
171  if (!$this->debug) {
172  header('HTTP/1.1 500 Internal Server Error');
173  $buffer = '';
174  }
175  }
176  return trim($buffer);
177  }
