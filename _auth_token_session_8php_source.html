20 /**
21  * AuthTokenSession is a DefaultSession, but additionally requires
22  * clients to send a token in the X-Auth-Token request header (Double Submit Cookie).
23  * The token is created, when the authenticated user is associated
24  * with the session and send to the client in a cookie named
25  * <em>application-title</em>-token.
26  *
27  * @author ingo herwig <ingo@wemove.com>
28  */
29 class AuthTokenSession extends DefaultSession implements Session {
30 
31  const TOKEN_HEADER = 'X-Auth-Token';
32 
33  private $tokenName = '';
34  private $isTokenValid = false;
35 
36  /**
37  * Constructor
38  * @param $configuration
39  * @param $request
40  */
41  public function __construct(Configuration $configuration, Request $request) {
42  parent::__construct($configuration);
43 
44  // check for valid auth-token in request
45  $this->tokenName = $this->getCookiePrefix().'-token';
46  $this->isTokenValid = $request->hasHeader(self::TOKEN_HEADER) &&
47  $request->getHeader(self::TOKEN_HEADER) == $_COOKIE[$this->tokenName];
48  }
49 
50  /**
51  * @see Session::setAuthUser()
52  */
53  public function setAuthUser($login) {
54  parent::setAuthUser($login);
55 
56  // set auth-token cookie for authenticated user
57  if ($login !== AnonymousUser::USER_GROUP_NAME) {
58  // generate a token, set it in the session and send it to the client
59  $token = base64_encode(openssl_random_pseudo_bytes(32));
60  $this->isTokenValid = true;
61  // NOTE: prevent "headers already sent" errors in phpunit tests
62  if (!headers_sent()) {
63  setcookie($this->tokenName, $token, 0, '/', '', URIUtil::isHttps(), false);
64  }
65  }
66  }
67 
68  /**
69  * @see Session::getAuthUser()
70  */
71  public function getAuthUser() {
72  $login = parent::getAuthUser();
73  return $this->isTokenValid ? $login : AnonymousUser::USER_GROUP_NAME;
74  }
75 }
