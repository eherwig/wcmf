20 /**
21  * AuthTokenSession is a DefaultSession, but additionally requires
22  * clients to send a token in the X-Auth-Token request header (Double Submit Cookie).
23  * The token is created, when the authenticated user is associated
24  * with the session and send to the client in a cookie named
25  * <em>application-title</em>-token.
26  *
27  * @author ingo herwig <ingo@wemove.com>
28  */
29 class AuthTokenSession extends DefaultSession implements Session {
30 
31  const TOKEN_HEADER = 'X-Auth-Token';
32 
33  private $tokenName = '';
34  private $isTokenValid = false;
35 
36  /**
37  * Constructor
38  * @param $configuration
39  */
40  public function __construct(Configuration $configuration) {
41  parent::__construct($configuration);
42 
43  $this->tokenName = $this->getCookiePrefix().'-token';
44  }
45 
46  /**
47  * @see Session::setAuthUser()
48  */
49  public function setAuthUser($login) {
50  parent::setAuthUser($login);
51 
52  // set auth-token cookie for authenticated user
53  if ($login !== AnonymousUser::USER_GROUP_NAME) {
54  // generate a token, set it in the session and send it to the client
55  $token = base64_encode(openssl_random_pseudo_bytes(32));
56  $this->isTokenValid = true;
57  // NOTE: prevent "headers already sent" errors in phpunit tests
58  if (!headers_sent()) {
59  setcookie($this->tokenName, $token, 0, '/', '', URIUtil::isHttps(), false);
60  }
61  }
62  }
63 
64  /**
65  * @see Session::getAuthUser()
66  */
67  public function getAuthUser() {
68  $login = parent::getAuthUser();
69  return $this->isTokenValid() ? $login : AnonymousUser::USER_GROUP_NAME;
70  }
71 
72  /**
73  * Check if the request contains a valid token
74  */
75  protected function isTokenValid() {
76  if ($this->isTokenValid) {
77  // already validated
78  return true;
79  }
80  $request = ObjectFactory::getInstance('request');
81  $token = $request->hasHeader(self::TOKEN_HEADER) ? $request->getHeader(self::TOKEN_HEADER) :
82  $request->getValue(self::TOKEN_HEADER, null);
83  $this->isTokenValid = $token != null && $token == $_COOKIE[$this->tokenName];
84  return $this->isTokenValid;
85  }
