19 /**
20  * AuthTokenSession is a DefaultSession, but additionally requires
21  * clients to send a token in the X-Auth-Token request header (Double Submit Cookie).
22  * The token is created, when the authenticated user is associated
23  * with the session and send to the client in a cookie named
24  * <em>application-title</em>-token.
25  *
26  * @author ingo herwig <ingo@wemove.com>
27  */
28 class AuthTokenSession extends DefaultSession implements TokenBasedSession {

59  /**
60  * @see Session::setAuthUser()
61  */
62  public function setAuthUser($login) {
63  parent::setAuthUser($login);
64 
65  // set auth-token cookie for authenticated user
66  if ($login !== AnonymousUser::USER_GROUP_NAME) {
67  // generate a token, set it in the session and send it to the client
68  $token = base64_encode(openssl_random_pseudo_bytes(32));
69  $this->isTokenValid = true;
70  // NOTE: prevent "headers already sent" errors in phpunit tests
71  if (!headers_sent()) {
72  setcookie($this->tokenName, $token, 0, '/', '', URIUtil::isHttps(), false);
73  }
74  }
75  }
76 
77  /**
78  * @see Session::getAuthUser()
79  */
80  public function getAuthUser() {
81  $login = parent::getAuthUser();
82  return $this->isTokenValid() ? $login : AnonymousUser::USER_GROUP_NAME;
83  }
84 
85  /**
86  * Check if the request contains a valid token
87  */
88  protected function isTokenValid() {
89  if ($this->isTokenValid) {
90  // already validated
91  return true;
92  }
93  $request = ObjectFactory::getInstance('request');
94  $token = $request->hasHeader(self::TOKEN_HEADER) ? $request->getHeader(self::TOKEN_HEADER) :
95  $request->getValue(self::TOKEN_HEADER, null);
96  $this->isTokenValid = $token != null && $token == $_COOKIE[$this->tokenName];
97  return $this->isTokenValid;
98  }
