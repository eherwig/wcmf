24 /**

58  /**
59  * @see PermissionManager::getPermissions()
60  */
61  public function getPermissions($resource, $context, $action) {
62  $result = null;
63  $actionKey = ActionKey::getBestMatch($this->_actionKeyProvider, $resource, $context, $action);
64  if (strlen($actionKey) > 0) {
65  $result = $this->deserializePermissions($this->_actionKeyProvider->getKeyValue($actionKey));
66  }
67  if (Log::isDebugEnabled(__CLASS__)) {
68  Log::debug("Permissions for $resource?$context?$action (->$actionKey): ".trim(StringUtil::getDump($result)), __CLASS__);
69  }
70  return $result;
71  }

73  /**
74  * @see PermissionManager::setPermissions()
75  */
76  public function setPermissions($resource, $context, $action, $permissions) {
77  $permissionInstance = $this->getPermissionInstance($resource, $context, $action);
78  $isChanged = false;
79 
80  if ($permissions != null) {
81  // set permissions
82  $rolesStr = $this->serializePermissions($permissions);
83  if (strlen($rolesStr)) {
84  if (!$permissionInstance) {
85  $this->createPermissionObject($resource, $context, $action, $rolesStr);
86  }
87  else {
88  $permissionInstance->setValue('roles', $rolesStr);
89  }
90  $isChanged = true;
91  }
92  }
93  else {
94  // delete permissions
95  if ($permissionInstance) {
96  $permissionInstance->delete();
97  $isChanged = true;
98  }
99  }
100  }

116  /**
117  * Modify a permission for the given role.
118  * @param $resource The resource (e.g. class name of the Controller or object id).
119  * @param $context The context in which the action takes place.
120  * @param $action The action to process.
121  * @param $role The role to authorize.
122  * @param $modifier One of the PERMISSION_MODIFIER_ constants, null, if the permission
123  * should be removed.
124  * @return boolean
125  */
126  protected function modifyPermission($resource, $context, $action, $role, $modifier) {
127  // define roles value (empty if no modifier is given)
128  $permVal = $modifier != null ? $modifier.$role : '';
129 
130  // check for existing permission
131  $permission = $this->getPermissionInstance($resource, $context, $action);
132  if (!$permission && $modifier != null) {
133  // create the permission, if it does not exist yet
134  $permission = $this->createPermissionObject($resource, $context, $action, $permVal);
135  }
136  elseif ($permission) {
137  $value = $permission->getValue('roles');
138  // remove role from value
139  $newValue = preg_replace('/ +/', ' ', str_replace(array(PermissionManager::PERMISSION_MODIFIER_ALLOW.$role,
140  PermissionManager::PERMISSION_MODIFIER_DENY.$role), "", $value));
141  if (strlen($newValue) > 0) {
142  $permission->setValue('roles', trim($newValue." ".$permVal));
143  }
144  else {
145  $permission->delete();
146  }
147  }
148  return true;
149  }

151  /**
152  * Get the permission object that matches the given parameters
153  * @param $resource Resource
154  * @param $context Context
155  * @param $action Action
156  * @return Instance of _permissionType or null
157  */
158  protected function getPermissionInstance($resource, $context, $action) {
159  $query = new ObjectQuery($this->_permissionType, __CLASS__.__METHOD__);
160  $tpl = $query->getObjectTemplate($this->_permissionType);
161  $tpl->setValue('resource', Criteria::asValue('=', $resource));
162  $tpl->setValue('context', Criteria::asValue('=', $context));
163  $tpl->setValue('action', Criteria::asValue('=', $action));
164  $permissions = $query->execute(BuildDepth::SINGLE);
165  return (sizeof($permissions) > 0) ? $permissions[0] : null;
166  }

168  /**
169  * Create a permission object with the given parameters
170  * @param $resource Resource
171  * @param $context Context
172  * @param $action Action
173  * @param $roles String representing the permissions as returned from serializePermissions()
174  * @return Instance of _permissionType
175  */
176  protected function createPermissionObject($resource, $context, $action, $roles) {
177  $persistenceFacade = ObjectFactory::getInstance('persistenceFacade');
178  $permission = $persistenceFacade->create($this->_permissionType);
179  $permission->setValue('resource', $resource);
180  $permission->setValue('context', $context);
181  $permission->setValue('action', $action);
182  $permission->setValue('roles', $roles);
183  return $permission;
184  }
