23 /**
24  * Render an responsive image tag using srcset and sizes attributes. The plugin
25  * will prepend the frontend cache directory (_FrontendCache_ configuration section)
26  * to the image locations in the srcset attribute.
27  *
28  * Example:
29  * @code
30  * {res_image src=$image->getFile() widths="1600,960,640" type="w"
31  * sizes="(min-width: 50em) 33vw, (min-width: 28em) 50vw, 100vw"
32  * alt="Image 1" default="images/blank.gif"}
33  * @endcode
34  *
35  * @note The plugin is capable of resizing the image according to the srcset sizes,
36  * but this feature is disabled by default, because it might be very memory consuming
37  * (see __generate__ option). A better option is to resize the images on demand
38  * by a separate script. To do this, add the following lines to the .htaccess file
39  * in the application's root directory:
40  *
41  * @code
42  * # responsive images (cache/ is the frontend cache directory)
43  * RewriteCond %{REQUEST_URI} cache/
44  * RewriteCond %{REQUEST_FILENAME} !-f
45  * RewriteRule ^cache/(.+)(\.(?:jpe?g|gif|png))$ image.php?file=$1$2 [NC,L]
46  * @endcode
47  *
48  * The resize script would could like this:
49  *
50  * @code
51  * <?php
52  * error_reporting(E_ALL);
53  *
54  * define('WCMF_BASE', realpath("./src/")."/");
55  * require_once(WCMF_BASE."/vendor/autoload.php");
56  *
57  * use wcmf\lib\core\ClassLoader;
58  * use wcmf\lib\io\FileUtil;
59  *
60  * new ClassLoader(WCMF_BASE);
61  *
62  * // get image file from 'file' request parameter and extract the width
63  * // filename is supposed to follow the pattern name-{width}.extension
64  * $requestedFile = filter_input(INPUT_GET, 'file', FILTER_SANITIZE_STRING);
65  * $extension = pathinfo($requestedFile, PATHINFO_EXTENSION);
66  * if(preg_match('/-([0-9]+)\.'.$extension.'$/', $requestedFile, $matches)) {
67  * $width = $matches[1];
68  * $originalFile = preg_replace('/-([0-9]+)\.'.$extension.'$/', '.'.$extension, $requestedFile);
69  * }
70  * else {
71  * $originalFile = $requestedFile;
72  * }
73  *
74  * // create the resized image file, if not existing
75  * $resizedFile = 'cache/'.$requestedFile;
76  * if (FileUtil::fileExists($originalFile) && !FileUtil::fileExists($resizedFile)) {
77  * $fixedFile = FileUtil::fixFilename($originalFile);
78  * FileUtil::mkdirRec(pathinfo($resizedFile, PATHINFO_DIRNAME));
79  * $image = new \Eventviva\ImageResize($fixedFile);
80  * $image->resizeToWidth($width);
81  * $image->save($resizedFile);
82  * }
83  *
84  * // return the image file
85  * $file = FileUtil::fileExists($resizedFile) ? $resizedFile : $originalFile;
86  * $imageInfo = getimagesize($file);
87  * $image = file_get_contents($file);
88  * header('Content-type: '.$imageInfo['mime'].';');
89  * header("Content-Length: ".strlen($image));
90  * echo $image;
91  * ?>
92  * @endcode
93  *
94  * @param $params Array with keys:
95  * - src: The image file
96  * - default: The default file, if src does not exist (optional)
97  * - widths: Comma separated, sorted list of width values to be used in the srcset attribute
98  * - type: Indicates how width values should be used (optional, default: w)
99  * - w: Values will be used as pixels, e.g. widths="1600,960" results in srcset="... 1600w, ... 960w"
100  * - x: Values will be used as pixel ration, e.g. widths="1600,960" results in srcset="... 2x, ... 1x"
101  * - sizes: Media queries to define image size in relation of the viewport (optional)
102  * - useDataAttributes: Boolean indicating whether to replace src, srcset, sizes by data-src, data-srcset, data-sizes (optional, default: __false__)
103  * - generate: Boolean indicating whether to generate the images or not (optional, default: __false__)
104  * - class: Image class (optional)
105  * - alt: Alternative text (optional)
106  * - title: Image title (optional)
107  * @param $template Smarty_Internal_Template
108  * @return String
109  */
110 function smarty_function_image($params, Smarty_Internal_Template $template) {
111  $file = $params['src'];
112  $default = isset($params['default']) ? $params['default'] : '';
113  $widths = $params['widths'];
114  $type = isset($params['type']) ? $params['type'] : 'w';
115  $sizes = isset($params['sizes']) ? $params['sizes'] : '';
116  $useDataAttributes = isset($params['useDataAttributes']) ? $params['useDataAttributes'] : false;
117  $generate = isset($params['generate']) ? $params['generate'] : false;
118  $class = isset($params['class']) ? $params['class'] : '';
119  $alt = isset($params['alt']) ? $params['alt'] : '';
120  $title = isset($params['title']) ? $params['title'] : '';
121 
122  if (strlen($file) == 0 && strlen($default) == 0) {
123  return;
124  }
125 
126  $fixedFile = FileUtil::fixFilename($file);
127 
128  // check if the file exists
129  if (!file_exists($fixedFile)) {
130  // try the default
131  $file = $fixedFile = $default;
132  if (!file_exists($fixedFile)) {
133  return;
134  }
135  }
136 
137  // get the image size in order to see if we have to resize
138  $imageInfo = getimagesize($fixedFile);
139  if ($imageInfo == false) {
140  // the file is no image
141  return;
142  }
143 
144  $config = ObjectFactory::getInstance('configuration');
145  $cacheRootAbs = $config->getDirectoryValue('cacheDir', 'FrontendCache');
146  if ($generate) {
147  FileUtil::mkdirRec(pathinfo($cacheRootAbs.$file, PATHINFO_DIRNAME));
148  }
149 
150  $extension = pathinfo($file, PATHINFO_EXTENSION);
151  $baseName = $cacheRootAbs.preg_replace('/\.'.$extension.'$/', '', $file).'-';
152 
153  $srcset = array();
154  $requestedWidths = array_map('trim', explode(',', $widths));
155  for ($i=0, $count=sizeof($requestedWidths); $i<$count; $i++) {
156  $width = $requestedWidths[$i];
157  $destNameAbs = $baseName.$width.'.'.$extension;
158  $destName = URIUtil::makeRelative($destNameAbs, dirname(FileUtil::realpath($_SERVER['SCRIPT_FILENAME'])).'/');
159 
160  // if 'width' differs from the image values, we have to resize the image
161  if ($width < $imageInfo[0] && $generate) {
162  // if the file does not exist in the cache, we have to create it
163  $dateOrig = @filemtime($fixedFile);
164  $dateCache = @filemtime($destName);
165  if (!file_exists($destName) || $dateOrig > $dateCache) {
166  $image = new \Eventviva\ImageResize($file);
167  $image->resizeToWidth($width);
168  $image->save($destName);
169  }
170 
171  // fallback to file, if cached file could not be created
172  if (!file_exists($destName)) {
173  $destName = $file;
174  }
175  }
176  $srcset[] = preg_replace('/ /', '%20', $destName).' '.($type === 'w' ? $width.'w' : ($count-$i).'x');
177  }
178 
179  $tag = '<img'.
180  ' '.($useDataAttributes ? 'data-' : '').'src="'.$file.'"'.
181  ' '.($useDataAttributes ? 'data-' : '').'srcset="'.join(', ', $srcset).'"'.
182  ' '.(strlen($sizes) > 0 ? ($useDataAttributes ? 'data-' : '').'sizes="'.$sizes.'"' : '').
183  (strlen($class) > 0 ? ' class="'.$class.'"' : '').
184  (strlen($alt) > 0 ? ' alt="'.$alt.'"' : '').
185  (strlen($title) > 0 ? ' title="'.$title.'"' : '').
186  '>';
187  return $tag;
188 }
